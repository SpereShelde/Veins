43 /**
44  * @brief Defines an extended DeciderResult for the 80211 protocol
45  * which stores the bit-rate of the transmission.
46  *
47  * @ingroup decider
48  * @ingroup ieee80211
49  */
50 class VEINS_API DeciderResult80211 : public DeciderResult {
51 protected:
52  /** @brief Stores the bit-rate of the transmission of the packet */
53  double bitrate;
54 
55  /** @brief Stores the signal to noise ratio of the transmission */
56  double snr;
57 
58  /** @brief Stores the received power in dBm
59  * Please note that this is NOT the RSSI. The RSSI is an indicator
60  * of the quality of the signal which is not standardized, and
61  * different vendors can define different indicators. This value
62  * indicates the power that the frame had when received by the
63  * NIC card, WITHOUT noise floor and WITHOUT interference
64  */
65  double recvPower_dBm;
66 
67  /** @brief Stores whether the uncorrect decoding was due to low power or collision */
68  bool collision;
69 
70 public:
71  /**
72  * @brief Initialises with the passed values.
73  *
74  * "bitrate" defines the bit-rate of the transmission of the packet.
75  */
76  DeciderResult80211(bool isCorrect, double bitrate, double snr, double recvPower_dBm = 0, bool collision = false)
77  : DeciderResult(isCorrect)
78  , bitrate(bitrate)
79  , snr(snr)
80  , recvPower_dBm(recvPower_dBm)
81  , collision(collision)
82  {
83  }
84 
85  /**
86  * @brief Returns the bit-rate of the transmission of the packet.
87  */
88  double getBitrate() const
89  {
90  return bitrate;
91  }
92 
93  /**
94  * @brief Returns the signal to noise ratio of the transmission.
95  */
96  double getSnr() const
97  {
98  return snr;
99  }
100 
101  /**
102  * @brief Returns whether drop was due to collision, if isCorrect is false
103  */
104  bool isCollision() const
105  {
106  return collision;
107  }
108 
109  /**
110  * @brief Returns the signal power in dBm.
111  */
112  double getRecvPower_dBm() const
113  {
114  return recvPower_dBm;
115  }
116 };
