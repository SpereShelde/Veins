74 std::vector<std::string> getMapping(std::string el)
75 {
76 
77  // search for string protection characters '
78  char protection = '\'';
79  size_t first = el.find(protection);
80  size_t second;
81  size_t eq;
82  std::string type, value;
83  std::vector<std::string> mapping;
84 
85  if (first == std::string::npos) {
86  // there's no string protection, simply split by '='
87  cStringTokenizer stk(el.c_str(), "=");
88  mapping = stk.asVector();
89  }
90  else {
91  // if there's string protection, we need to find a matching delimiter
92  second = el.find(protection, first + 1);
93  // ensure that a matching delimiter exists, and that it is at the end
94  if (second == std::string::npos || second != el.size() - 1) throw cRuntimeError("invalid syntax for mapping \"%s\"", el.c_str());
95 
96  // take the value of the mapping as the text within the quotes
97  value = el.substr(first + 1, second - first - 1);
98 
99  if (first == 0) {
100  // if the string starts with a quote, there's only the value
101  mapping.push_back(value);
102  }
103  else {
104  // search for the equal sign
105  eq = el.find('=');
106  // this must be the character before the quote
107  if (eq == std::string::npos || eq != first - 1) {
108  throw cRuntimeError("invalid syntax for mapping \"%s\"", el.c_str());
109  }
110  else {
111  type = el.substr(0, eq);
112  }
113  mapping.push_back(type);
114  mapping.push_back(value);
115  }
116  }
117  return mapping;
118 }

122 TraCIScenarioManager::TypeMapping TraCIScenarioManager::parseMappings(std::string parameter, std::string parameterName, bool allowEmpty)
123 {
124 
125  /**
126  * possible syntaxes
127  *
128  * "a" : assign module type "a" to all nodes (for backward compatibility)
129  * "a=b" : assign module type "b" to vehicle type "a". the presence of any other vehicle type in the simulation will cause the simulation to stop
130  * "a=b c=d" : assign module type "b" to vehicle type "a" and "d" to "c". the presence of any other vehicle type in the simulation will cause the simulation to stop
131  * "a=b c=d *=e": everything which is not of vehicle type "a" or "b", assign module type "e"
132  * "a=b c=0" : for vehicle type "c" no module should be instantiated
133  * "a=b c=d *=0": everything which is not of vehicle type a or c should not be instantiated
134  *
135  * For display strings key-value pairs needs to be protected with single quotes, as they use an = sign as the type mappings. For example
136  * *.manager.moduleDisplayString = "'i=block/process'"
137  * *.manager.moduleDisplayString = "a='i=block/process' b='i=misc/sun'"
138  *
139  * moduleDisplayString can also be left empty:
140  * *.manager.moduleDisplayString = ""
141  */
142 
143  unsigned int i;
144  TypeMapping map;
145 
146  // tokenizer to split into mappings ("a=b c=d", -> ["a=b", "c=d"])
147  cStringTokenizer typesTz(parameter.c_str(), " ");
148  // get all mappings
149  std::vector<std::string> typeMappings = typesTz.asVector();
150  // and check that there exists at least one
151  if (typeMappings.size() == 0) {
152  if (!allowEmpty)
153  throw cRuntimeError("parameter \"%s\" is empty", parameterName.c_str());
154  else
155  return map;
156  }
157 
158  // loop through all mappings
159  for (i = 0; i < typeMappings.size(); i++) {
160 
161  // tokenizer to find the mapping from vehicle type to module type
162  std::string typeMapping = typeMappings[i];
163 
164  std::vector<std::string> mapping = getMapping(typeMapping);
165 
166  if (mapping.size() == 1) {
167  // we are where there is no actual assignment
168  // "a": this is good
169  // "a b=c": this is not
170  if (typeMappings.size() != 1)
171  // stop simulation with an error
172  throw cRuntimeError("parameter \"%s\" includes multiple mappings, but \"%s\" is not mapped to any vehicle type", parameterName.c_str(), mapping[0].c_str());
173  else
174  // all vehicle types should be instantiated with this module type
175  map["*"] = mapping[0];
176  }
177  else {
178 
179  // check that mapping is valid (a=b and not like a=b=c)
180  if (mapping.size() != 2) throw cRuntimeError("invalid syntax for mapping \"%s\" for parameter \"%s\"", typeMapping.c_str(), parameterName.c_str());
181  // check that the mapping does not already exist
182  if (map.find(mapping[0]) != map.end()) throw cRuntimeError("duplicated mapping for vehicle type \"%s\" for parameter \"%s\"", mapping[0].c_str(), parameterName.c_str());
183 
184  // finally save the mapping
185  map[mapping[0]] = mapping[1];
186  }
187  }
188 
189  return map;
190 }

192 void TraCIScenarioManager::parseModuleTypes()
193 {
194 
195  TypeMapping::iterator i;
196  std::vector<std::string> typeKeys, nameKeys, displayStringKeys;
197 
198  std::string moduleTypes = par("moduleType").stdstringValue();
199  std::string moduleNames = par("moduleName").stdstringValue();
200  std::string moduleDisplayStrings = par("moduleDisplayString").stdstringValue();
201 
202  moduleType = parseMappings(moduleTypes, "moduleType", false);
203  moduleName = parseMappings(moduleNames, "moduleName", false);
204  moduleDisplayString = parseMappings(moduleDisplayStrings, "moduleDisplayString", true);
205 
206  // perform consistency check. for each vehicle type in moduleType there must be a vehicle type
207  // in moduleName (and in moduleDisplayString if moduleDisplayString is not empty)
208 
209  // get all the keys
210  for (i = moduleType.begin(); i != moduleType.end(); i++) typeKeys.push_back(i->first);
211  for (i = moduleName.begin(); i != moduleName.end(); i++) nameKeys.push_back(i->first);
212  for (i = moduleDisplayString.begin(); i != moduleDisplayString.end(); i++) displayStringKeys.push_back(i->first);
213 
214  // sort them (needed for intersection)
215  std::sort(typeKeys.begin(), typeKeys.end());
216  std::sort(nameKeys.begin(), nameKeys.end());
217  std::sort(displayStringKeys.begin(), displayStringKeys.end());
218 
219  std::vector<std::string> intersection;
220 
221  // perform set intersection
222  std::set_intersection(typeKeys.begin(), typeKeys.end(), nameKeys.begin(), nameKeys.end(), std::back_inserter(intersection));
223  if (intersection.size() != typeKeys.size() || intersection.size() != nameKeys.size()) throw cRuntimeError("keys of mappings of moduleType and moduleName are not the same");
224 
225  if (displayStringKeys.size() == 0) return;
226 
227  intersection.clear();
228  std::set_intersection(typeKeys.begin(), typeKeys.end(), displayStringKeys.begin(), displayStringKeys.end(), std::back_inserter(intersection));
229  if (intersection.size() != displayStringKeys.size()) throw cRuntimeError("keys of mappings of moduleType and moduleName are not the same");
230 }

289 void TraCIScenarioManager::init_traci()
290 {
291  auto* commandInterface = getCommandInterface();
292  {
293  auto apiVersion = commandInterface->getVersion();
294  EV_DEBUG << "TraCI server \"" << apiVersion.second << "\" reports API version " << apiVersion.first << endl;
295  commandInterface->setApiVersion(apiVersion.first);
296  }
297 
298  {
299  // query and set road network boundaries
300  auto networkBoundaries = commandInterface->initNetworkBoundaries(par("margin"));
301  if (world != nullptr && ((connection->traci2omnet(networkBoundaries.second).x > world->getPgs()->x) || (connection->traci2omnet(networkBoundaries.first).y > world->getPgs()->y))) {
302  EV_DEBUG << "WARNING: Playground size (" << world->getPgs()->x << ", " << world->getPgs()->y << ") might be too small for vehicle at network bounds (" << connection->traci2omnet(networkBoundaries.second).x << ", " << connection->traci2omnet(networkBoundaries.first).y << ")" << endl;
303  }
304  }
305 
306  {
307  // subscribe to list of departed and arrived vehicles, as well as simulation time
308  simtime_t beginTime = 0;
309  simtime_t endTime = SimTime::getMaxTime();
310  std::string objectId = "";
311  uint8_t variableNumber = 7;
312  uint8_t variable1 = VAR_DEPARTED_VEHICLES_IDS;
313  uint8_t variable2 = VAR_ARRIVED_VEHICLES_IDS;
314  uint8_t variable3 = commandInterface->getTimeStepCmd();
315  uint8_t variable4 = VAR_TELEPORT_STARTING_VEHICLES_IDS;
316  uint8_t variable5 = VAR_TELEPORT_ENDING_VEHICLES_IDS;
317  uint8_t variable6 = VAR_PARKING_STARTING_VEHICLES_IDS;
318  uint8_t variable7 = VAR_PARKING_ENDING_VEHICLES_IDS;
319  TraCIBuffer buf = connection->query(CMD_SUBSCRIBE_SIM_VARIABLE, TraCIBuffer() << beginTime << endTime << objectId << variableNumber << variable1 << variable2 << variable3 << variable4 << variable5 << variable6 << variable7);
320  processSubcriptionResult(buf);
321  ASSERT(buf.eof());
322  }
323 
324  {
325  // subscribe to list of vehicle ids
326  simtime_t beginTime = 0;
327  simtime_t endTime = SimTime::getMaxTime();
328  std::string objectId = "";
329  uint8_t variableNumber = 1;
330  uint8_t variable1 = ID_LIST;
331  TraCIBuffer buf = connection->query(CMD_SUBSCRIBE_VEHICLE_VARIABLE, TraCIBuffer() << beginTime << endTime << objectId << variableNumber << variable1);
332  processSubcriptionResult(buf);
333  ASSERT(buf.eof());
334  }
335 
336  if (!trafficLightModuleType.empty() && !trafficLightModuleIds.empty()) {
337  // initialize traffic lights
338  cModule* parentmod = getParentModule();
339  if (!parentmod) {
340  throw cRuntimeError("Parent Module not found (for traffic light creation)");
341  }
342  cModuleType* tlModuleType = cModuleType::get(trafficLightModuleType.c_str());
343 
344  // query traffic lights via TraCI
345  std::list<std::string> trafficLightIds = commandInterface->getTrafficlightIds();
346  size_t nrOfTrafficLights = trafficLightIds.size();
347  int cnt = 0;
348  for (std::list<std::string>::iterator i = trafficLightIds.begin(); i != trafficLightIds.end(); ++i) {
349  std::string tlId = *i;
350  if (std::find(trafficLightModuleIds.begin(), trafficLightModuleIds.end(), tlId) == trafficLightModuleIds.end()) {
351  continue; // filter only selected elements
352  }
353 
354  Coord position = commandInterface->junction(tlId).getPosition();
355 
356  cModule* module = tlModuleType->create(trafficLightModuleName.c_str(), parentmod, nrOfTrafficLights, cnt);
357  module->par("externalId") = tlId;
358  module->finalizeParameters();
359  module->getDisplayString().parse(trafficLightModuleDisplayString.c_str());
360  module->buildInside();
361  module->scheduleStart(simTime() + updateInterval);
362 
363  cModule* tlIfSubmodule = module->getSubmodule("tlInterface");
364  // initialize traffic light interface with current program
365  TraCITrafficLightInterface* tlIfModule = dynamic_cast<TraCITrafficLightInterface*>(tlIfSubmodule);
366  tlIfModule->preInitialize(tlId, position, updateInterval);
367 
368  // initialize mobility for positioning
369  cModule* mobiSubmodule = module->getSubmodule("mobility");
370  mobiSubmodule->par("x") = position.x;
371  mobiSubmodule->par("y") = position.y;
372  mobiSubmodule->par("z") = position.z;
373 
374  module->callInitialize();
375  trafficLights[tlId] = module;
376  subscribeToTrafficLightVariables(tlId); // subscribe after module is in trafficLights
377  cnt++;
378  }
379  }
380 
381  ObstacleControl* obstacles = ObstacleControlAccess().getIfExists();
382  if (obstacles) {
383  {
384  // get list of polygons
385  std::list<std::string> ids = commandInterface->getPolygonIds();
386  for (std::list<std::string>::iterator i = ids.begin(); i != ids.end(); ++i) {
387  std::string id = *i;
388  std::string typeId = commandInterface->polygon(id).getTypeId();
389  if (!obstacles->isTypeSupported(typeId)) continue;
390  std::list<Coord> coords = commandInterface->polygon(id).getShape();
391  std::vector<Coord> shape;
392  std::copy(coords.begin(), coords.end(), std::back_inserter(shape));
393  obstacles->addFromTypeAndShape(id, typeId, shape);
394  }
395  }
396  }
397 
398  traciInitialized = true;
399  emit(traciInitializedSignal, true);
400 
401  // draw and calculate area of rois
402  for (std::list<std::pair<TraCICoord, TraCICoord>>::const_iterator r = roi.getRectangles().begin(), end = roi.getRectangles().end(); r != end; ++r) {
403  TraCICoord first = r->first;
404  TraCICoord second = r->second;
405 
406  std::list<Coord> pol;
407 
408  Coord a = connection->traci2omnet(first);
409  Coord b = connection->traci2omnet(TraCICoord(first.x, second.y));
410  Coord c = connection->traci2omnet(second);
411  Coord d = connection->traci2omnet(TraCICoord(second.x, first.y));
412 
413  pol.push_back(a);
414  pol.push_back(b);
415  pol.push_back(c);
416  pol.push_back(d);
417 
418  // draw polygon for region of interest
419  if (annotations) {
420  annotations->drawPolygon(pol, "black");
421  }
422 
423  // calculate region area
424  double ab = a.distance(b);
425  double ad = a.distance(d);
426  double area = ab * ad;
427  areaSum += area;
428  }
429 }

483 void TraCIScenarioManager::addModule(std::string nodeId, std::string type, std::string name, std::string displayString, const Coord& position, std::string road_id, double speed, Heading heading, VehicleSignalSet signals, double length, double height, double width)
484 {
485 
486  if (hosts.find(nodeId) != hosts.end()) throw cRuntimeError("tried adding duplicate module");
487 
488  double option1 = hosts.size() / (hosts.size() + unEquippedHosts.size() + 1.0);
489  double option2 = (hosts.size() + 1) / (hosts.size() + unEquippedHosts.size() + 1.0);
490 
491  if (fabs(option1 - penetrationRate) < fabs(option2 - penetrationRate)) {
492  unEquippedHosts.insert(nodeId);
493  return;
494  }
495 
496  int32_t nodeVectorIndex = nextNodeVectorIndex++;
497 
498  cModule* parentmod = getParentModule();
499  if (!parentmod) throw cRuntimeError("Parent Module not found");
500 
501  cModuleType* nodeType = cModuleType::get(type.c_str());
502  if (!nodeType) throw cRuntimeError("Module Type \"%s\" not found", type.c_str());
503 
504  // TODO: this trashes the vectsize member of the cModule, although nobody seems to use it
505  cModule* mod = nodeType->create(name.c_str(), parentmod, nodeVectorIndex, nodeVectorIndex);
506  mod->finalizeParameters();
507  if (displayString.length() > 0) {
508  mod->getDisplayString().parse(displayString.c_str());
509  }
510  mod->buildInside();
511  mod->scheduleStart(simTime() + updateInterval);
512 
513  preInitializeModule(mod, nodeId, position, road_id, speed, heading, signals);
514 
515  mod->callInitialize();
516  hosts[nodeId] = mod;
517 
518  // post-initialize TraCIMobility
519  auto mobilityModules = getSubmodulesOfType<TraCIMobility>(mod);
520  for (auto mm : mobilityModules) {
521  mm->changePosition();
522  }
523 
524  if (vehicleObstacleControl) {
525  std::vector<AntennaPosition> initialAntennaPositions;
526  for (auto& caModule : getSubmodulesOfType<ChannelAccess>(mod, true)) {
527  initialAntennaPositions.push_back(caModule->getAntennaPosition());
528  }
529  ASSERT(mobilityModules.size() == 1);
530  auto mm = mobilityModules[0];
531  double offset = mm->getHostPositionOffset();
532  const MobileHostObstacle* vo = vehicleObstacleControl->add(MobileHostObstacle(initialAntennaPositions, mm, length, offset, width, height));
533  vehicleObstacles[mm] = vo;
534  }
535 
536  emit(traciModuleAddedSignal, mod);
537 }

551 void TraCIScenarioManager::deleteManagedModule(std::string nodeId)
552 {
553  cModule* mod = getManagedModule(nodeId);
554  if (!mod) throw cRuntimeError("no vehicle with Id \"%s\" found", nodeId.c_str());
555 
556  emit(traciModuleRemovedSignal, mod);
557 
558  auto cas = getSubmodulesOfType<ChannelAccess>(mod, true);
559  for (auto ca : cas) {
560  cModule* nic = ca->getParentModule();
561  auto connectionManager = ChannelAccess::getConnectionManager(nic);
562  connectionManager->unregisterNic(nic);
563  }
564  if (vehicleObstacleControl) {
565  for (cModule::SubmoduleIterator iter(mod); !iter.end(); iter++) {
566  cModule* submod = *iter;
567  TraCIMobility* mm = dynamic_cast<TraCIMobility*>(submod);
568  if (!mm) continue;
569  auto vo = vehicleObstacles.find(mm);
570  ASSERT(vo != vehicleObstacles.end());
571  vehicleObstacleControl->erase(vo->second);
572  }
573  }
