34 /**
35  * @brief HostState is published by the battery to announce host failure
36  *
37  * HostState information is published by the Battery to announce Host
38  * failure due to battery depletion. (Default state is ON, OFF is not
39  * used; existing modules basically ignore everything but FAIL.)
40  *
41  * All modules that generate messages and collect statistics should
42  * subscribe to this notification. Should eventually be generalized
43  * to handle more general HostState (e.g. stochastic failure,
44  * restart).
45  *
46  * @ingroup power
47  */
48 class VEINS_API HostState : public cObject {
49 public:
50  /**
51  * @brief Possible host states.
52  */
53  enum States {
54  ACTIVE, /** Host is active and fully working*/
55  FAILED, /** Host is not active because of some failure
56  * but might able to be restarted*/
57  BROKEN, /** Host is not active because of some failure
58  * and won't be able to be restarted */
59  SLEEP, /** Host is not active (sleeping) */
60  OFF /** Host is not active (shut down) */
61  };
62  // we could make a nice 'info' field here, to allow us to specify
63  // the cause of failure (e.g. battery, stochastic hardware failure)
64 
65 private:
66  /**
67  * @brief Host state.
68  */
69  States state;
70 
71 public:
72  /**
73  * @brief Constructor taking a state.
74  */
75  HostState(States state = ACTIVE)
76  : state(state)
77  {
78  }
79 
80  /** @brief Returns the host state */
81  States get() const
82  {
83  return state;
84  }
85  /** @brief Sets the host state */
86  void set(States s)
87  {
88  state = s;
89  }
90 
91  /**
92  * @brief Returns information about the current state.
93  */
94  std::string info() const override
95  {
96  std::ostringstream ost;
97  switch (state) {
98  case ACTIVE:
99  ost << "ACTIVE";
100  break;
101  case FAILED:
102  ost << "FAILED";
103  break;
104  case BROKEN:
105  ost << "BROKEN";
106  break;
107  case SLEEP:
108  ost << "SLEEP";
109  break;
110  case OFF:
111  ost << "OFF";
112  break;
113  default:
114  ost << "Unknown";
115  break;
116  }
117  return ost.str();
118  }
119 };
