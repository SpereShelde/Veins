101 template <>
102 std::string TraCIBuffer::read()
103 {
104  uint32_t length = read<uint32_t>();
105  if (length == 0) return std::string();
106  char obuf[length + 1];
107 
108  for (size_t i = 0; i < length; ++i) read<char>(obuf[i]);
109  obuf[length] = 0;
110 
111  return std::string(obuf, length);
112 }
