140  /**
141  * @brief Represents an minimalistic (hash)set of GridCoords.
142  *
143  * It is a workaround because c++ doesn't come with an hash set.
144  */
145  class VEINS_API CoordSet {
146  protected:
147  /** @brief Holds the hash table.*/
148  std::vector<GridCoord*> data;
149  /** @brief maximum size of the hash table.*/
150  unsigned maxSize;
151  /** @brief Current number of entries in the hash table.*/
152  unsigned size;
153  /** @brief Holds the current element when iterating over this table.*/
154  unsigned current;
155 
156  protected:
157  /**
158  * @brief Tries to insert a GridCoord at the specified position.
159  *
160  * If the same Coord already exists there nothing happens.
161  * If an other Coord already exists there calculate
162  * a new Position to insert end recursively call this Method again.
163  * If the spot is empty the Coord is inserted.
164  */
165  void insert(const GridCoord& c, unsigned pos)
166  {
167  if (data[pos] == nullptr) {
168  data[pos] = new GridCoord(c);
169  size++;
170  }
171  else {
172  if (*data[pos] != c) {
173  insert(c, (pos + 2) % maxSize);
174  }
175  }
176  }
177 
178  public:
179  /**
180  * @brief Initializes the set (hashtable) with the a specified size.
181  */
182  CoordSet(unsigned sz)
183  : maxSize(sz)
184  , size(0)
185  , current(0)
186  {
187  data.resize(maxSize);
188  }
189 
190  /**
191  * @brief Delete every created GridCoord
192  */
193  ~CoordSet()
194  {
195  for (unsigned i = 0; i < maxSize; i++) {
196  if (data[i] != nullptr) {
197  delete data[i];
198  }
199  }
200  }
201 
202  /**
203  * @brief Adds a GridCoord to the set.
204  * If a GridCoord with the same value already exists in the set
205  * nothing happens.
206  */
207  void add(const GridCoord& c)
208  {
209  unsigned hash = (c.x * 10000 + c.y * 100 + c.z) % maxSize;
210  insert(c, hash);
211  }
212 
213  /**
214  * @brief Returns the next GridCoord in the set.
215  * You can iterate through the set only one time with this function!
216  */
217  GridCoord* next()
218  {
219  for (; current < maxSize; current++) {
220  if (data[current] != nullptr) {
221  return data[current++];
222  }
223  }
224  return nullptr;
225  }
226 
227  /**
228  * @brief Returns the number of GridCoords currently saved in this set.
229  */
230  unsigned getSize()
231  {
232  return size;
233  }
234 
235  /**
236  * @brief Returns the maximum number of elements which can be stored inside
237  * this set.
238  * To prevent collisions the set should never be more than 75% filled.
239  */
240  unsigned getmaxSize()
241  {
242  return maxSize;
243  }
244  };
