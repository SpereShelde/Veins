35 /**
36  * Fast grid-based spatial datastructure to find obstacles (geometric shapes) in a bounding box.
37  *
38  * Stores bounding boxes for a set of obstacles and allows searching for them via another bounding box.
39  *
40  * Only considers a 2-dimensional plane (x and y coordinates).
41  *
42  * In principle, any kind (or implementation) of a obstacle/shape/polygon is possible.
43  * There only has to be a function to derive a bounding box for a given obstacle.
44  * Obstacle instances are stored as pointers, so the lifetime of the obstacle instances is not managed by this class.
45  */
46 class VEINS_API BBoxLookup {
47 public:
48  struct Point {
49  double x;
50  double y;
51  };
52  // array of stucts approach
53  // cache line size: 64byte = 8 x 8-byte double = 2 bboxes per line
54  // bbox coordinates are inherently local, if i check x1, i likely also check the other
55  struct Box {
56  Point p1;
57  Point p2;
58  };
59  struct BBoxCell {
60  size_t index; /**< index of the first element of this cell in bboxes */
61  size_t count; /**< number of elements in this cell; index + number = index of last element */
62  };
63 
64  BBoxLookup() = default;
65  BBoxLookup(const std::vector<Obstacle*>& obstacles, std::function<BBoxLookup::Box(Obstacle*)> makeBBox, double scenarioX, double scenarioY, int cellSize = 250);
66 
67  /**
68  * Return all obstacles which have their bounding box touched by the transmission from sender to receiver.
69  *
70  * The obstacles itself may not actually overlap with transmission (false positives are possible).
71  */
72  std::vector<Obstacle*> findOverlapping(Point sender, Point receiver) const;
73 
74 private:
75  // NOTE: obstacles may occur multiple times in bboxes/obstacleLookup (if they are in multiple cells)
76  std::vector<Box> bboxes; /**< ALL bboxes in one chunck of contiguos memory, ordered by cells */
77  std::vector<Obstacle*> obstacleLookup; /**< bboxes[i] belongs to instance in obstacleLookup[i] */
78  std::vector<BBoxCell> bboxCells; /**< flattened matrix of X * Y BBoxCell instances */
79  int cellSize = 0;
80  size_t numCols = 0; /**< X BBoxCell instances in a row */
81  size_t numRows = 0; /**< Y BBoxCell instances in a column */
82 };
