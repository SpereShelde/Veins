49 void PhyLayer80211p::initialize(int stage)
50 {
51  if (stage == 0) {
52  // get ccaThreshold before calling BasePhyLayer::initialize() which instantiates the deciders
53  ccaThreshold = pow(10, par("ccaThreshold").doubleValue() / 10);
54  allowTxDuringRx = par("allowTxDuringRx").boolValue();
55  collectCollisionStatistics = par("collectCollisionStatistics").boolValue();
56 
57  // Create frequency mappings and initialize spectrum for signal representation
58  Spectrum::Frequencies freqs;
59  for (auto& channel : IEEE80211ChannelFrequencies) {
60  freqs.push_back(channel.second - 5e6);
61  freqs.push_back(channel.second);
62  freqs.push_back(channel.second + 5e6);
63  }
64  overallSpectrum = Spectrum(freqs);
65  }
66  BasePhyLayer::initialize(stage);
67 }

98 {
99  double alpha1 = -1, alpha2 = -1, breakpointDistance = -1;
100  double L01 = -1, L02 = -1;

104

107  // set alpha1

109  EV_TRACE << "createPathLossModel(): alpha1 set from config.xml to " << alpha1 << endl;
110  // check whether alpha is not smaller than specified in ConnectionManager
111  if (cc->hasPar("alpha") && alpha1 < cc->par("alpha").doubleValue()) {
112  // throw error

115  }
116  }
117  it = params.find("L01");
118  if (it != params.end()) {
119  L01 = it->second.doubleValue();

122  if (it != params.end()) {
123  L02 = it->second.doubleValue();
124  }
125 
126  it = params.find("alpha2");
127  if (it != params.end()) { // parameter alpha1 has been specified in config.xml
128  // set alpha2
129  alpha2 = it->second.doubleValue();
130  EV_TRACE << "createPathLossModel(): alpha2 set from config.xml to " << alpha2 << endl;
131  // check whether alpha is not smaller than specified in ConnectionManager
132  if (cc->hasPar("alpha") && alpha2 < cc->par("alpha").doubleValue()) {
133  // throw error
134  throw cRuntimeError("TestPhyLayer::createPathLossModel(): alpha can't be smaller than specified in \
135  ConnectionManager. Please adjust your config.xml file accordingly");
136  }
137  }
138  it = params.find("breakpointDistance");
139  if (it != params.end()) { // parameter alpha1 has been specified in config.xml
140  breakpointDistance = it->second.doubleValue();
141  EV_TRACE << "createPathLossModel(): breakpointDistance set from config.xml to " << alpha2 << endl;
142  // check whether alpha is not smaller than specified in ConnectionManager
143  }
144 
145  if (alpha1 == -1 || alpha2 == -1 || breakpointDistance == -1 || L01 == -1 || L02 == -1) {
146  throw cRuntimeError("Undefined parameters for breakpointPathlossModel. Please check your configuration.");
147  }

149  return make_unique<BreakpointPathlossModel>(this, L01, L02, alpha1, alpha2, breakpointDistance, useTorus, playgroundSize);

271 void PhyLayer80211p::handleSelfMessage(cMessage* msg)
272 {
273 
274  switch (msg->getKind()) {
275  // transmission overBasePhyLayer::
276  case TX_OVER: {
277  ASSERT(msg == txOverTimer);
278  sendControlMsgToMac(new cMessage("Transmission over", TX_OVER));
279  // check if there is another packet on the chan, and change the chan-state to idle
280  Decider80211p* dec = dynamic_cast<Decider80211p*>(decider.get());
281  ASSERT(dec);
282  if (dec->cca(simTime(), nullptr)) {
283  // chan is idle
284  EV_TRACE << "Channel idle after transmit!\n";
285  dec->setChannelIdleStatus(true);
286  }
287  else {
288  EV_TRACE << "Channel not yet idle after transmit!\n";
289  }
290  break;
291  }
292  // radio switch over
293  case RADIO_SWITCHING_OVER:
294  ASSERT(msg == radioSwitchingOverTimer);
295  BasePhyLayer::finishRadioSwitching();
296  break;
297 
298  // AirFrame
299  case AIR_FRAME:
300  BasePhyLayer::handleAirFrame(static_cast<AirFrame*>(msg));
301  break;
302 
303  default:
304  break;
305  }
306 }

313 void PhyLayer80211p::attachSignal(AirFrame* airFrame, cObject* ctrlInfo)
314 {
315  const auto ctrlInfo11p = check_and_cast<MacToPhyControlInfo11p*>(ctrlInfo);
316 
317  const auto duration = getFrameDuration(airFrame->getEncapsulatedPacket()->getBitLength(), ctrlInfo11p->mcs);
318  ASSERT(duration > 0);
319  Signal signal(overallSpectrum, simTime(), duration);
320  auto freqIndex = overallSpectrum.indexOf(IEEE80211ChannelFrequencies.at(ctrlInfo11p->channelNr));
321  signal.at(freqIndex - 1) = ctrlInfo11p->txPower_mW;
322  signal.at(freqIndex) = ctrlInfo11p->txPower_mW;
323  signal.at(freqIndex + 1) = ctrlInfo11p->txPower_mW;
324  signal.setDataStart(freqIndex - 1);
325  signal.setDataEnd(freqIndex + 1);
326  signal.setCenterFrequencyIndex(freqIndex);
327  // copy the signal into the AirFrame
328  airFrame->setSignal(signal);
329  airFrame->setDuration(signal.getDuration());
330  airFrame->setMcs(static_cast<int>(ctrlInfo11p->mcs));
331 }

375 simtime_t PhyLayer80211p::getFrameDuration(int payloadLengthBits, MCS mcs) const
376 {
377  Enter_Method_Silent();
378  ASSERT(mcs != MCS::undefined);
379  auto ndbps = getNDBPS(mcs);
380  // calculate frame duration according to Equation (17-29) of the IEEE 802.11-2007 standard
381  return PHY_HDR_PREAMBLE_DURATION + PHY_HDR_PLCPSIGNAL_DURATION + T_SYM_80211P * ceil(static_cast<double>(16 + payloadLengthBits + 6) / (ndbps));
382 }
