118  /**
119  * @brief Multiplies a coordinate vector by another coordinate vector;
120  * @return Scalar value
121  * in Algebra: referred to as "dot product" or "scalar product";
122  * Takes two equal-length sequences of numbers (usually coordinate vectors) and returns a single number.
123  * in Euclidean geometry: the dot product of the Cartesian coordinates of two vectors (i.e. inner product).
124  */
125  friend double operator*(const Coord& a, const Coord& b)
126  {
127  return (a.x * b.x + a.y * b.y);
128  }
129 
130  /**
131  * @brief Returns the magnitude of the vector that would result from a regular 3D cross product
132  * of the input vectors; The values on the z-plane are assumed to be 0 (i.e. treating the 2D space as a plane in the 3D space).
133  * The 3D cross product will be perpendicular to that plane, and thus have 0 X & Y components
134  * (thus the scalar returned is the Z value of the 3D cross product vector).
135  */
136  double twoDimensionalCrossProduct(const Coord& a) const
137  {
138  return (x * a.y - y * a.x);
139  }

206  /**
207  * @brief Tests whether two coordinate vectors are equal.
208  *
209  * Because coordinates are floating point values, this is done using an epsilon comparison.

212  */

268  /**
269  * @brief Checks if this coordinate is inside a specified rectangle.
270  *
271  * @param lowerBound The upper bound of the rectangle.
272  * @param upperBound The lower bound of the rectangle.
273  */
274  bool isInBoundary(const Coord& lowerBound, const Coord& upperBound) const
275  {
276  return lowerBound.x <= x && x <= upperBound.x && lowerBound.y <= y && y <= upperBound.y && lowerBound.z <= z && z <= upperBound.z;
277  }
