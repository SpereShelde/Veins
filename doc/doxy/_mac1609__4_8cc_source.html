41 void Mac1609_4::initialize(int stage)
42 {

44  if (stage == 0) {
45 
46  phy11p = FindModule<Mac80211pToPhy11pInterface*>::findSubModule(getParentModule());
47  ASSERT(phy11p);
48

50  ASSERT(simTime().getScaleExp() == -12);
51 
52  txPower = par("txPower").doubleValue();
53  setParametersForBitrate(par("bitrate"));
54

56  dot11RTSThreshold = par("dot11RTSThreshold");
57  dot11ShortRetryLimit = par("dot11ShortRetryLimit");
58  dot11LongRetryLimit = par("dot11LongRetryLimit");
59  ackLength = par("ackLength");

61  ackErrorRate = par("ackErrorRate").doubleValue();

63  ignoreChannelState = false;
64  waitUntilAckRXorTimeout = false;
65  stopIgnoreChannelStateMsg = new cMessage("ChannelStateMsg");
66

68  // create two edca systems
69

71  myEDCA[ChannelType::control]->myId = myId;

79  myEDCA[ChannelType::service]->myId = myId;
80  myEDCA[ChannelType::service]->myId.append(" SCH");
81  myEDCA[ChannelType::service]->createQueue(2, (((CWMIN_11P + 1) / 4) - 1), (((CWMIN_11P + 1) / 2) - 1), AC_VO);
82  myEDCA[ChannelType::service]->createQueue(3, (((CWMIN_11P + 1) / 2) - 1), CWMIN_11P, AC_VI);
83  myEDCA[ChannelType::service]->createQueue(6, CWMIN_11P, CWMAX_11P, AC_BE);

88  if (useAcks) throw cRuntimeError("Unicast model does not support channel switching");

92  case 1:

94  break;
95  case 2:

97  break;
98  case 3:
99  mySCH = Channel::sch3;
100  break;

107  }
108  }
109 
110  headerLength = par("headerLength");
111 
112  nextMacEvent = new cMessage("next Mac Event");
113 
114  if (useSCH) {
115  uint64_t currenTime = simTime().raw();
116  uint64_t switchingTime = SWITCHING_INTERVAL_11P.raw();
117  double timeToNextSwitch = (double) (switchingTime - (currenTime % switchingTime)) / simTime().getScale();
118  if ((currenTime / switchingTime) % 2 == 0) {

121  else {

150

152  lastBusy = simTime();
153  channelIdle(true);
154  }

157 void Mac1609_4::handleSelfMsg(cMessage* msg)
158 {
159  if (msg == stopIgnoreChannelStateMsg) {
160  ignoreChannelState = false;
161  return;
162  }
163 
164  if (AckTimeOutMessage* ackTimeOutMsg = dynamic_cast<AckTimeOutMessage*>(msg)) {
165  handleAckTimeOut(ackTimeOutMsg);
166  return;
167  }
168 
169  if (msg == nextChannelSwitch) {
170  ASSERT(useSCH);
171 
172  scheduleAt(simTime() + SWITCHING_INTERVAL_11P, nextChannelSwitch);
173 
174  switch (activeChannel) {
175  case ChannelType::control:
176  EV_TRACE << "CCH --> SCH" << std::endl;
177  channelBusySelf(false);
178  setActiveChannel(ChannelType::service);
179  channelIdle(true);
180  phy11p->changeListeningChannel(mySCH);
181  break;
182  case ChannelType::service:
183  EV_TRACE << "SCH --> CCH" << std::endl;
184  channelBusySelf(false);
185  setActiveChannel(ChannelType::control);
186  channelIdle(true);
187  phy11p->changeListeningChannel(Channel::cch);
188  break;
189  }
190  // schedule next channel switch in 50ms
191  }
192  else if (msg == nextMacEvent) {
193 
194  // we actually came to the point where we can send a packet
195  channelBusySelf(true);
196  BaseFrame1609_4* pktToSend = myEDCA[activeChannel]->initiateTransmit(lastIdle);
197  ASSERT(pktToSend);
198 
199  lastAC = mapUserPriority(pktToSend->getUserPriority());
200  lastWSM = pktToSend;
201 
202  EV_TRACE << "MacEvent received. Trying to send packet with priority" << lastAC << std::endl;
203 
204  // send the packet
205  Mac80211Pkt* mac = new Mac80211Pkt(pktToSend->getName(), pktToSend->getKind());
206  if (pktToSend->getRecipientAddress() != LAddress::L2BROADCAST()) {
207  mac->setDestAddr(pktToSend->getRecipientAddress());
208  }
209  else {
210  mac->setDestAddr(LAddress::L2BROADCAST());
211  }
212  mac->setSrcAddr(myMacAddr);
213  mac->encapsulate(pktToSend->dup());
214 
215  MCS usedMcs = mcs;
216  double txPower_mW;
217  PhyControlMessage* controlInfo = dynamic_cast<PhyControlMessage*>(pktToSend->getControlInfo());
218  if (controlInfo) {
219  // if MCS is not specified, just use the default one
220  MCS explicitMcs = static_cast<MCS>(controlInfo->getMcs());
221  if (explicitMcs != MCS::undefined) {
222  usedMcs = explicitMcs;
223  }
224  // apply the same principle to tx power
225  txPower_mW = controlInfo->getTxPower_mW();
226  if (txPower_mW < 0) {
227  txPower_mW = txPower;
228  }
229  }
230  else {
231  txPower_mW = txPower;
232  }
233 
234  simtime_t sendingDuration = RADIODELAY_11P + phy11p->getFrameDuration(mac->getBitLength(), usedMcs);
235  EV_TRACE << "Sending duration will be" << sendingDuration << std::endl;
236  if ((!useSCH) || (timeLeftInSlot() > sendingDuration)) {
237  if (useSCH) EV_TRACE << " Time in this slot left: " << timeLeftInSlot() << std::endl;
238 
239  Channel channelNr = (activeChannel == ChannelType::control) ? Channel::cch : mySCH;
240  double freq = IEEE80211ChannelFrequencies.at(channelNr);
241 
242  EV_TRACE << "Sending a Packet. Frequency " << freq << " Priority" << lastAC << std::endl;
243  sendFrame(mac, RADIODELAY_11P, channelNr, usedMcs, txPower_mW);
244 
245  // schedule ack timeout for unicast packets
246  if (pktToSend->getRecipientAddress() != LAddress::L2BROADCAST() && useAcks) {
247  waitUntilAckRXorTimeout = true;
248  // PHY-RXSTART.indication should be received within ackWaitTime
249  // sifs + slot + rx_delay: see 802.11-2012 9.3.2.8 (32us + 13us + 49us = 94us)
250  simtime_t ackWaitTime(94, SIMTIME_US);
251  // update id in the retransmit timer
252  myEDCA[activeChannel]->myQueues[lastAC].ackTimeOut->setWsmId(pktToSend->getTreeId());
253  simtime_t timeOut = sendingDuration + ackWaitTime;
254  scheduleAt(simTime() + timeOut, myEDCA[activeChannel]->myQueues[lastAC].ackTimeOut);
255  }
256  }
257  else { // not enough time left now
258  EV_TRACE << "Too little Time left. This packet cannot be send in this slot." << std::endl;
259  statsNumTooLittleTime++;
260  // revoke TXOP
261  myEDCA[activeChannel]->revokeTxOPs();
262  delete mac;
263  channelIdle();
264  // do nothing. contention will automatically start after channel switch
265  }
266  }

274 void Mac1609_4::handleUpperMsg(cMessage* msg)
275 {
276 
277  BaseFrame1609_4* thisMsg = check_and_cast<BaseFrame1609_4*>(msg);
278 
279  t_access_category ac = mapUserPriority(thisMsg->getUserPriority());
280 
281  EV_TRACE << "Received a message from upper layer for channel " << thisMsg->getChannelNumber() << " Access Category (Priority): " << ac << std::endl;
282 
283  ChannelType chan;
284 
285  if (static_cast<Channel>(thisMsg->getChannelNumber()) == Channel::cch) {
286  chan = ChannelType::control;
287  }
288  else {
289  ASSERT(useSCH);
290  thisMsg->setChannelNumber(static_cast<int>(mySCH));
291  chan = ChannelType::service;
292  }
293 
294  int num = myEDCA[chan]->queuePacket(ac, thisMsg);
295 
296  // packet was dropped in Mac
297  if (num == -1) {
298  statsDroppedPackets++;
299  return;
300  }
301 
302  // if this packet is not at the front of a new queue we dont have to reevaluate times
303  EV_TRACE << "sorted packet into queue of EDCA " << static_cast<int>(chan) << " this packet is now at position: " << num << std::endl;
304 
305  if (chan == activeChannel) {
306  EV_TRACE << "this packet is for the currently active channel" << std::endl;
307  }
308  else {
309  EV_TRACE << "this packet is NOT for the currently active channel" << std::endl;
310  }
311 
312  if (num == 1 && idleChannel == true && chan == activeChannel) {
313 
314  simtime_t nextEvent = myEDCA[chan]->startContent(lastIdle, guardActive());
315 
316  if (nextEvent != -1) {
317  if ((!useSCH) || (nextEvent <= nextChannelSwitch->getArrivalTime())) {
318  if (nextMacEvent->isScheduled()) {
319  cancelEvent(nextMacEvent);
320  }
321  scheduleAt(nextEvent, nextMacEvent);
322  EV_TRACE << "Updated nextMacEvent:" << nextMacEvent->getArrivalTime().raw() << std::endl;
323  }
324  else {
325  EV_TRACE << "Too little time in this interval. Will not schedule nextMacEvent" << std::endl;
326  // it is possible that this queue has an txop. we have to revoke it
327  myEDCA[activeChannel]->revokeTxOPs();
328  statsNumTooLittleTime++;
329  }
330  }
331  else {
332  cancelEvent(nextMacEvent);
333  }
334  }
335  if (num == 1 && idleChannel == false && myEDCA[chan]->myQueues[ac].currentBackoff == 0 && chan == activeChannel) {
336  myEDCA[chan]->backoff(ac);
337  }
338 }

340 void Mac1609_4::handleLowerControl(cMessage* msg)
341 {
342  if (msg->getKind() == MacToPhyInterface::PHY_RX_START) {
343  rxStartIndication = true;
344  }
345  else if (msg->getKind() == MacToPhyInterface::PHY_RX_END_WITH_SUCCESS) {
346  // PHY_RX_END_WITH_SUCCESS will get packet soon! Nothing to do here
347  }
348  else if (msg->getKind() == MacToPhyInterface::PHY_RX_END_WITH_FAILURE) {
349  // RX failed at phy. Time to retransmit
350  phy11p->notifyMacAboutRxStart(false);
351  rxStartIndication = false;
352  handleRetransmit(lastAC);
353  }
354  else if (msg->getKind() == MacToPhyInterface::TX_OVER) {
355 
356  EV_TRACE << "Successfully transmitted a packet on " << lastAC << std::endl;
357 
358  phy->setRadioState(Radio::RX);
359 
360  if (!dynamic_cast<Mac80211Ack*>(lastMac.get())) {
361  // message was sent
362  // update EDCA queue. go into post-transmit backoff and set cwCur to cwMin
363  myEDCA[activeChannel]->postTransmit(lastAC, lastWSM, useAcks);
364  }
365  // channel just turned idle.
366  // don't set the chan to idle. the PHY layer decides, not us.
367 
368  if (guardActive()) {
369  throw cRuntimeError("We shouldnt have sent a packet in guard!");
370  }
371  }
372  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_BUSY) {
373  channelBusy();
374  }
375  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_IDLE) {
376  // Decider80211p::processSignalEnd() sends up the received packet to MAC followed by control message CHANNEL_IDLE in the same timestamp.
377  // If we received a unicast frame (first event scheduled by Decider), MAC immediately schedules an ACK message and wants to switch the radio to TX mode.
378  // So, the notification for channel idle from phy is undesirable and we skip it here.
379  // After ACK TX is over, PHY will inform the channel status again.
380  if (ignoreChannelState) {
381  // Skipping channelidle because we are about to send an ack regardless of the channel state
382  }
383  else {
384  channelIdle();
385  }
386  }
387  else if (msg->getKind() == Decider80211p::BITERROR || msg->getKind() == Decider80211p::COLLISION) {
388  statsSNIRLostPackets++;
389  EV_TRACE << "A packet was not received due to biterrors" << std::endl;
390  }
391  else if (msg->getKind() == Decider80211p::RECWHILESEND) {
392  statsTXRXLostPackets++;
393  EV_TRACE << "A packet was not received because we were sending while receiving" << std::endl;
394  }
395  else if (msg->getKind() == MacToPhyInterface::RADIO_SWITCHING_OVER) {
396  EV_TRACE << "Phylayer said radio switching is done" << std::endl;
397  }
398  else if (msg->getKind() == BaseDecider::PACKET_DROPPED) {
399  phy->setRadioState(Radio::RX);
400  EV_TRACE << "Phylayer said packet was dropped" << std::endl;
401  }
402  else {
403  EV_WARN << "Invalid control message type (type=NOTHING) : name=" << msg->getName() << " modulesrc=" << msg->getSenderModule()->getFullPath() << "." << std::endl;
404  ASSERT(false);
405  }
406 
407  if (msg->getKind() == Decider80211p::COLLISION) {
408  emit(sigCollision, true);
409  }
410 
411  delete msg;
412 }

461 void Mac1609_4::sendFrame(Mac80211Pkt* frame, simtime_t delay, Channel channelNr, MCS mcs, double txPower_mW)
462 {
463  phy->setRadioState(Radio::TX); // give time for the radio to be in Tx state before transmitting
464 
465  delay = std::max(delay, RADIODELAY_11P); // wait at least for the radio to switch
466 
467  attachControlInfo(frame, channelNr, mcs, txPower_mW);
468  check_and_cast<MacToPhyControlInfo11p*>(frame->getControlInfo());
469 
470  lastMac.reset(frame->dup());
471  sendDelayed(frame, delay, lowerLayerOut);
472 
473  if (dynamic_cast<Mac80211Ack*>(frame)) {
474  statsSentAcks += 1;
475  }
476  else {
477  statsSentPackets += 1;
478  }
479 }

553 void Mac1609_4::handleLowerMsg(cMessage* msg)
554 {
555  Mac80211Pkt* macPkt = check_and_cast<Mac80211Pkt*>(msg);
556 
557  // pass information about received frame to the upper layers
558  DeciderResult80211* macRes = check_and_cast<DeciderResult80211*>(PhyToMacControlInfo::getDeciderResult(msg));
559  DeciderResult80211* res = new DeciderResult80211(*macRes);
560 
561  long dest = macPkt->getDestAddr();
562 
563  EV_TRACE << "Received frame name= " << macPkt->getName() << ", myState= src=" << macPkt->getSrcAddr() << " dst=" << macPkt->getDestAddr() << " myAddr=" << myMacAddr << std::endl;
564 
565  if (dest == myMacAddr) {
566  if (auto* ack = dynamic_cast<Mac80211Ack*>(macPkt)) {
567  ASSERT(useAcks);
568  handleAck(ack);
569  delete res;
570  }
571  else {
572  unique_ptr<BaseFrame1609_4> wsm(check_and_cast<BaseFrame1609_4*>(macPkt->decapsulate()));
573  wsm->setControlInfo(new PhyToMacControlInfo(res));
574  handleUnicast(macPkt->getSrcAddr(), std::move(wsm));
575  }
576  }
577  else if (dest == LAddress::L2BROADCAST()) {
578  handleBroadcast(macPkt, res);
579  }
580  else {
581  EV_TRACE << "Packet not for me" << std::endl;
582  delete res;
583  }
584  delete macPkt;
585 
586  if (rxStartIndication) {
587  // We have handled/processed the incoming packet
588  // Since we reached here, we were expecting an ack but we didnt get it, so retransmission should take place
589  phy11p->notifyMacAboutRxStart(false);
590  rxStartIndication = false;
591  handleRetransmit(lastAC);
592  }

644 BaseFrame1609_4* Mac1609_4::EDCA::initiateTransmit(simtime_t lastIdle)
645 {
646 
647  // iterate through the queues to return the packet we want to send
648  BaseFrame1609_4* pktToSend = nullptr;
649 
650  simtime_t idleTime = simTime() - lastIdle;
651 
652  EV_TRACE << "Initiating transmit at " << simTime() << ". I've been idle since " << idleTime << std::endl;
653 
654  // As t_access_category is sorted by priority, we iterate back to front.
655  // This realizes the behavior documented in IEEE Std 802.11-2012 Section 9.2.4.2; that is, "data frames from the higher priority AC" win an internal collision.
656  // The phrase "EDCAF of higher UP" of IEEE Std 802.11-2012 Section 9.19.2.3 is assumed to be meaningless.
657  for (auto iter = myQueues.rbegin(); iter != myQueues.rend(); iter++) {
658  if (iter->second.queue.size() != 0 && !iter->second.waitForAck) {
659  if (idleTime >= iter->second.aifsn * SLOTLENGTH_11P + SIFS_11P && iter->second.txOP == true) {
660 
661  EV_TRACE << "Queue " << iter->first << " is ready to send!" << std::endl;
662 
663  iter->second.txOP = false;
664  // this queue is ready to send
665  if (pktToSend == nullptr) {
666  pktToSend = iter->second.queue.front();
667  }
668  else {
669  // there was already another packet ready. we have to go increase cw and go into backoff. It's called internal contention and its wonderful
670 
671  statsNumInternalContention++;
672  iter->second.cwCur = std::min(iter->second.cwMax, (iter->second.cwCur + 1) * 2 - 1);
673  iter->second.currentBackoff = owner->intuniform(0, iter->second.cwCur);
674  EV_TRACE << "Internal contention for queue " << iter->first << " : " << iter->second.currentBackoff << ". Increase cwCur to " << iter->second.cwCur << std::endl;
675  }
676  }
677  }
678  }
679 
680  if (pktToSend == nullptr) {
681  throw cRuntimeError("No packet was ready");
682  }
683  return pktToSend;

686 simtime_t Mac1609_4::EDCA::startContent(simtime_t idleSince, bool guardActive)
687 {
688 
689  EV_TRACE << "Restarting contention." << std::endl;
690 
691  simtime_t nextEvent = -1;
692 
693  simtime_t idleTime = SimTime().setRaw(std::max((int64_t) 0, (simTime() - idleSince).raw()));
694  ;
695 
696  lastStart = idleSince;
697 
698  EV_TRACE << "Channel is already idle for:" << idleTime << " since " << idleSince << std::endl;
699 
700  // this returns the nearest possible event in this EDCA subsystem after a busy channel
701 
702  for (auto&& p : myQueues) {
703  auto& accessCategory = p.first;
704  auto& edcaQueue = p.second;
705  if (edcaQueue.queue.size() != 0 && !edcaQueue.waitForAck) {
706 
707  /* 1609_4 says that when attempting to send (backoff == 0) when guard is active, a random backoff is invoked */
708 
709  if (guardActive == true && edcaQueue.currentBackoff == 0) {
710  // cw is not increased
711  edcaQueue.currentBackoff = owner->intuniform(0, edcaQueue.cwCur);
712  statsNumBackoff++;
713  }
714 
715  simtime_t DIFS = edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P;
716 
717  // the next possible time to send can be in the past if the channel was idle for a long time, meaning we COULD have sent earlier if we had a packet
718  simtime_t possibleNextEvent = DIFS + edcaQueue.currentBackoff * SLOTLENGTH_11P;
719 
720  EV_TRACE << "Waiting Time for Queue " << accessCategory << ":" << possibleNextEvent << "=" << edcaQueue.aifsn << " * " << SLOTLENGTH_11P << " + " << SIFS_11P << "+" << edcaQueue.currentBackoff << "*" << SLOTLENGTH_11P << "; Idle time: " << idleTime << std::endl;
721 
722  if (idleTime > possibleNextEvent) {
723  EV_TRACE << "Could have already send if we had it earlier" << std::endl;
724  // we could have already sent. round up to next boundary
725  simtime_t base = idleSince + DIFS;
726  possibleNextEvent = simTime() - simtime_t().setRaw((simTime() - base).raw() % SLOTLENGTH_11P.raw()) + SLOTLENGTH_11P;
727  }
728  else {
729  // we are gonna send in the future
730  EV_TRACE << "Sending in the future" << std::endl;
731  possibleNextEvent = idleSince + possibleNextEvent;
732  }
733  nextEvent == -1 ? nextEvent = possibleNextEvent : nextEvent = std::min(nextEvent, possibleNextEvent);
734  }
735  }
736  return nextEvent;

739 void Mac1609_4::EDCA::stopContent(bool allowBackoff, bool generateTxOp)
740 {
741  // update all Queues
742 
743  EV_TRACE << "Stopping Contention at " << simTime().raw() << std::endl;
744 
745  simtime_t passedTime = simTime() - lastStart;
746 
747  EV_TRACE << "Channel was idle for " << passedTime << std::endl;
748 
749  lastStart = -1; // indicate that there was no last start
750 
751  for (auto&& p : myQueues) {
752  auto& accessCategory = p.first;
753  auto& edcaQueue = p.second;
754  if ((edcaQueue.currentBackoff != 0 || edcaQueue.queue.size() != 0) && !edcaQueue.waitForAck) {
755  // check how many slots we already waited until the chan became busy
756 
757  int64_t oldBackoff = edcaQueue.currentBackoff;
758 
759  std::string info;
760  if (passedTime < edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P) {
761  // we didnt even make it one DIFS :(
762  info.append(" No DIFS");
763  }
764  else {
765  // decrease the backoff by one because we made it longer than one DIFS
766  edcaQueue.currentBackoff -= 1;
767 
768  // check how many slots we waited after the first DIFS
769  int64_t passedSlots = (int64_t)((passedTime - SimTime(edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P)) / SLOTLENGTH_11P);
770 
771  EV_TRACE << "Passed slots after DIFS: " << passedSlots << std::endl;
772 
773  if (edcaQueue.queue.size() == 0) {
774  // this can be below 0 because of post transmit backoff -> backoff on empty queues will not generate macevents,
775  // we dont want to generate a txOP for empty queues
776  edcaQueue.currentBackoff -= std::min(edcaQueue.currentBackoff, passedSlots);
777  info.append(" PostCommit Over");
778  }
779  else {
780  edcaQueue.currentBackoff -= passedSlots;
781  if (edcaQueue.currentBackoff <= -1) {
782  if (generateTxOp) {
783  edcaQueue.txOP = true;
784  info.append(" TXOP");
785  }
786  // else: this packet couldnt be sent because there was too little time. we could have generated a txop, but the channel switched
787  edcaQueue.currentBackoff = 0;
788  }
789  }
790  }
791  EV_TRACE << "Updating backoff for Queue " << accessCategory << ": " << oldBackoff << " -> " << edcaQueue.currentBackoff << info << std::endl;
792  }
793  }

1075 void Mac1609_4::handleRetransmit(t_access_category ac)
1076 {
1077  // cancel the acktime out
1078  if (myEDCA[ChannelType::control]->myQueues[ac].ackTimeOut->isScheduled()) {
1079  // This case is possible if we received PHY_RX_END_WITH_SUCCESS or FAILURE even before ack timeout
1080  cancelEvent(myEDCA[ChannelType::control]->myQueues[ac].ackTimeOut);
1081  }
1082  if (myEDCA[ChannelType::control]->myQueues[ac].queue.size() == 0) {
1083  throw cRuntimeError("Trying retransmission on empty queue...");
1084  }
1085  BaseFrame1609_4* appPkt = myEDCA[ChannelType::control]->myQueues[ac].queue.front();
1086  bool contend = false;
1087  bool retriesExceeded = false;
1088  // page 879 of IEEE 802.11-2012
1089  if (appPkt->getBitLength() <= dot11RTSThreshold) {
1090  myEDCA[ChannelType::control]->myQueues[ac].ssrc++;
1091  if (myEDCA[ChannelType::control]->myQueues[ac].ssrc <= dot11ShortRetryLimit) {
1092  retriesExceeded = false;
1093  }
1094  else {
1095  retriesExceeded = true;
1096  }
1097  }
1098  else {
1099  myEDCA[ChannelType::control]->myQueues[ac].slrc++;
1100  if (myEDCA[ChannelType::control]->myQueues[ac].slrc <= dot11LongRetryLimit) {
1101  retriesExceeded = false;
1102  }
1103  else {
1104  retriesExceeded = true;
1105  }
1106  }
1107  if (!retriesExceeded) {
1108  // try again!
1109  myEDCA[ChannelType::control]->myQueues[ac].cwCur = std::min(myEDCA[ChannelType::control]->myQueues[ac].cwMax, (myEDCA[ChannelType::control]->myQueues[ac].cwCur * 2) + 1);
1110  myEDCA[ChannelType::control]->backoff(ac);
1111  contend = true;
1112  // no need to reset wait on id here as we are still retransmitting same packet
1113  myEDCA[ChannelType::control]->myQueues[ac].waitForAck = false;
1114  }
1115  else {
1116  // enough tries!
1117  myEDCA[ChannelType::control]->myQueues[ac].queue.pop();
1118  if (myEDCA[ChannelType::control]->myQueues[ac].queue.size() > 0) {
1119  // start contention only if there are more packets in the queue
1120  contend = true;
1121  }
1122  delete appPkt;
1123  myEDCA[ChannelType::control]->myQueues[ac].cwCur = myEDCA[ChannelType::control]->myQueues[ac].cwMin;
1124  myEDCA[ChannelType::control]->backoff(ac);
1125  myEDCA[ChannelType::control]->myQueues[ac].waitForAck = false;
1126  myEDCA[ChannelType::control]->myQueues[ac].waitOnUnicastID = -1;
1127  myEDCA[ChannelType::control]->myQueues[ac].ssrc = 0;
1128  myEDCA[ChannelType::control]->myQueues[ac].slrc = 0;
1129  }
1130  waitUntilAckRXorTimeout = false;
1131  if (contend && idleChannel && !ignoreChannelState) {
1132  // reevaluate times -- if channel is not idle, then contention would start automatically
1133  cancelEvent(nextMacEvent);
1134  simtime_t nextEvent = myEDCA[ChannelType::control]->startContent(lastIdle, guardActive());
1135  scheduleAt(nextEvent, nextMacEvent);
1136  }
1137 }
