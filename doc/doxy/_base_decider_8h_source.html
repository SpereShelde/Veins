40 /**
41  * @brief Provides some base functionality for most common deciders.
42  *
43  * Forwards the AirFrame from "processSignal" to "processNewSignal",
44  * "processSignalHeader" or "processSignalEnd" depending on the
45  * state for that AirFrame returned by "getSignalState".
46  *
47  * If a subclassing Decider only tries to receive one signal at a time
48  * it can use BaseDeciders "currentSignal" member which is a pair of
49  * the signal to receive and the state for that signal. The state
50  * is then used by BaseDeciders "getSignalState" to decide to which
51  * "process***" method to forward the signal.
52  * If a subclassing Decider needs states for more than one Signal it
53  * has to store these states by itself and should probably override
54  * the "getSignalState" method.
55  *
56  * @ingroup decider
57  * @ingroup baseModules
58  */
59 class VEINS_API BaseDecider : public Decider {
60 public:
61  /**
62  * @brief The kinds of ControlMessages this Decider sends.
63  *
64  * Sub-classing decider should begin their own kind enumeration
65  * at the value of "LAST_BASE_DECIDER_CONTROL_KIND".
66  */
67  enum BaseDeciderControlKinds {
68  /** @brief The phy has recognized a bit error in the packet.*/
69  PACKET_DROPPED = 22100,
70  /** @brief Sub-classing decider should begin their own kinds at this
71  * value.*/
72  LAST_BASE_DECIDER_CONTROL_KIND
73  };
74 
75 protected:
76  /** @brief The current state of processing for a signal*/
77  enum SignalState {
78  /** @brief Signal is received the first time. */
79  NEW,
80  /** @brief Waiting for the header of the signal. */
81  EXPECT_HEADER,
82  /** @brief Waiting for the end of the signal. */
83  EXPECT_END,
84  };
85 
86  /** @brief minPowerLevel value for receiving an AirFrame */
87  double minPowerLevel;
88 
89  /** @brief Pair of a AirFrame and the state it is in. */
90  typedef std::pair<AirFrame*, int> ReceivedSignal;
91 
92  /** @brief pointer to the currently received AirFrame */
93  ReceivedSignal currentSignal;
94 
95  /** @brief Stores the idle state of the channel.*/
96  bool isChannelIdle;
97 
98  /** @brief index for this Decider-instance given by Phy-Layer (mostly
99  * Host-index) */
100  int myIndex;
101 
102 public:
103  /**
104  * @brief Initializes the decider with the passed values.
105  *
106  * Needs a pointer to its physical layer, the minPowerLevel, and the index of the host.
107  */
108  BaseDecider(cComponent* owner, DeciderToPhyInterface* phy, double minPowerLevel, int myIndex)
109  : Decider(owner, phy)
110  , minPowerLevel(minPowerLevel)
111  , isChannelIdle(true)
112  , myIndex(myIndex)
113  {
114  currentSignal.first = 0;
115  currentSignal.second = NEW;
116  }
117 
118  ~BaseDecider() override
119  {
120  }
121 
122 public:
123  /**
124  * @brief Processes an AirFrame given by the PhyLayer
125  *
126  * Returns the time point when the decider wants to be given the AirFrame
127  * again.
128  */
129  simtime_t processSignal(AirFrame* frame) override;
130 
131 protected:
132  /**
133  * @brief Processes a new Signal. Returns the time it wants to
134  * handle the signal again.
135  *
136  * Default implementation checks if the signals receiving power
137  * is above the minPowerLevel of the radio and we are not already trying
138  * to receive another AirFrame. If thats the case it waits for the end
139  * of the signal.
140  */
141  virtual simtime_t processNewSignal(AirFrame* frame);
142 
143  /**
144  * @brief Processes the end of the header of a received Signal.
145  *
146  * Returns the time it wants to handle the signal again.
147  *
148  * Default implementation does not handle signal headers.
149  */
150  virtual simtime_t processSignalHeader(AirFrame* frame)
151  {
152  throw cRuntimeError("BaseDecider does not handle Signal headers!");
153  return notAgain;
154  }
155 
156  /**
157  * @brief Processes the end of a received Signal.
158  *
159  * Returns the time it wants to handle the signal again
160  * (most probably notAgain).
161  *
162  * Default implementation just decides every signal as correct and passes it
163  * to the upper layer.
164  */
165  virtual simtime_t processSignalEnd(AirFrame* frame);
166 
167  /**
168  * @brief Processes any Signal for which no state could be found.
169  * (is an error case).
170  */
171  virtual simtime_t processUnknownSignal(AirFrame* frame);
172 
173  /**
174  * @brief Returns the SignalState for the passed AirFrame.
175  *
176  * The default implementation checks if the passed AirFrame
177  * is the "currentSignal" and returns its state or if not
178  * "NEW".
179  */
180  virtual int getSignalState(AirFrame* frame);
181 
182  /**
183  * @brief Changes the "isIdle"-status to the passed value.
184  */
185  virtual void setChannelIdleStatus(bool isIdle);
186 
187  /**
188  * @brief Collects the AirFrame on the channel during the passed interval.
189  *
190  * Forwards to DeciderToPhyInterfaces "getChannelInfo" method.
191  * Subclassing deciders can override this method to filter the returned
192  * AirFrames for their own criteria, for example by removing AirFrames on
193  * another not interferring channel.
194  *
195  * @param start The start of the interval to collect AirFrames from.
196  * @param end The end of the interval to collect AirFrames from.
197  * @param out The output vector in which to put the AirFrames.
198  */
199  virtual void getChannelInfo(simtime_t_cref start, simtime_t_cref end, AirFrameVector& out);
200 };
