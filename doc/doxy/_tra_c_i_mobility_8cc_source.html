81  currentAccelerationVec.setName("acceleration");
82  currentCO2EmissionVec.setName("co2emission");

84  statistics.initialize();
85  statistics.watch(*this);
86 
87  ASSERT(isPreInitialized);
88  isPreInitialized = false;
89 
90  Coord nextPos = calculateHostPosition(roadPosition);
91  nextPos.z = move.getStartPosition().z;
92 
93  move.setStart(nextPos);
94  move.setDirectionByVector(heading.toCoord());
95  move.setOrientationByVector(heading.toCoord());
96  if (this->setHostSpeed) {
97  move.setSpeed(speed);
98  }
99 
100  isParking = false;
101 
102  startAccidentMsg = nullptr;
103  stopAccidentMsg = nullptr;
104  manager = nullptr;
105  last_speed = -1;
106 
107  if (accidentCount > 0) {
108  simtime_t accidentStart = par("accidentStart");
109  startAccidentMsg = new cMessage("scheduledAccident");
110  stopAccidentMsg = new cMessage("scheduledAccidentResolved");
111  scheduleAt(simTime() + accidentStart, startAccidentMsg);
112  }
113  }
114  else if (stage == 1) {
115  // don't call BaseMobility::initialize(stage) -- our parent will take care to call changePosition later
116  }

119  }

134 void TraCIMobility::handleSelfMsg(cMessage* msg)
135 {
136  if (msg == startAccidentMsg) {
137  getVehicleCommandInterface()->setSpeed(0);
138  simtime_t accidentDuration = par("accidentDuration");
139  scheduleAt(simTime() + accidentDuration, stopAccidentMsg);
140  accidentCount--;
141  }
142  else if (msg == stopAccidentMsg) {
143  getVehicleCommandInterface()->setSpeed(-1);
144  if (accidentCount > 0) {
145  simtime_t accidentInterval = par("accidentInterval");
146  scheduleAt(simTime() + accidentInterval, startAccidentMsg);
147  }
148  }
149 }

189 void TraCIMobility::changePosition()
190 {
191  // ensure we're not called twice in one time step
192  ASSERT(lastUpdate != simTime());
193 
194  // keep statistics (for current step)
195  currentPosXVec.record(move.getStartPos().x);
196  currentPosYVec.record(move.getStartPos().y);
197 
198  Coord nextPos = calculateHostPosition(roadPosition);
199  nextPos.z = move.getStartPosition().z;
200 
201  // keep statistics (relative to last step)
202  if (statistics.startTime != simTime()) {
203  simtime_t updateInterval = simTime() - this->lastUpdate;
204 
205  double distance = move.getStartPos().distance(nextPos);
206  statistics.totalDistance += distance;
207  statistics.totalTime += updateInterval;
208  if (speed != -1) {
209  statistics.minSpeed = std::min(statistics.minSpeed, speed);
210  statistics.maxSpeed = std::max(statistics.maxSpeed, speed);
211  currentSpeedVec.record(speed);
212  if (last_speed != -1) {
213  double acceleration = (speed - last_speed) / updateInterval;
214  double co2emission = calculateCO2emission(speed, acceleration);
215  currentAccelerationVec.record(acceleration);
216  currentCO2EmissionVec.record(co2emission);
217  statistics.totalCO2Emission += co2emission * updateInterval.dbl();
218  }
219  last_speed = speed;
220  }
221  else {
222  last_speed = -1;
223  speed = -1;
224  }
225  }
226  this->lastUpdate = simTime();
227 
228  // Update display string to show node is getting updates
229  auto hostMod = getParentModule();
230  if (std::string(hostMod->getDisplayString().getTagArg("veins", 0)) == ". ") {
231  hostMod->getDisplayString().setTagArg("veins", 0, " .");
232  }
233  else {
234  hostMod->getDisplayString().setTagArg("veins", 0, ". ");
235  }
236 
237  move.setStart(Coord(nextPos.x, nextPos.y, move.getStartPosition().z)); // keep z position
238  move.setDirectionByVector(heading.toCoord());
239  move.setOrientationByVector(heading.toCoord());
240  if (this->setHostSpeed) {
241  move.setSpeed(speed);
242  }
243  fixIfHostGetsOutside();
244  updatePosition();
245 }

254 void TraCIMobility::fixIfHostGetsOutside()
255 {
256  Coord pos = move.getStartPos();
257  Coord dummy = Coord::ZERO;
258  double dum;
259 
260  bool outsideX = (pos.x < 0) || (pos.x >= playgroundSizeX());
261  bool outsideY = (pos.y < 0) || (pos.y >= playgroundSizeY());
262  bool outsideZ = (!world->use2D()) && ((pos.z < 0) || (pos.z >= playgroundSizeZ()));
263  if (outsideX || outsideY || outsideZ) {
264  throw cRuntimeError("Tried moving host to (%f, %f) which is outside the playground", pos.x, pos.y);
265  }
266 
267  handleIfOutside(RAISEERROR, pos, dummy, dummy, dum);
268 }

270 double TraCIMobility::calculateCO2emission(double v, double a) const
271 {
272  // Calculate CO2 emission parameters according to:
273  // Cappiello, A. and Chabini, I. and Nam, E.K. and Lue, A. and Abou Zeid, M., "A statistical model of vehicle emissions and fuel consumption," IEEE 5th International Conference on Intelligent Transportation Systems (IEEE ITSC), pp. 801-809, 2002
274 
275  double A = 1000 * 0.1326; // W/m/s
276  double B = 1000 * 2.7384e-03; // W/(m/s)^2
277  double C = 1000 * 1.0843e-03; // W/(m/s)^3
278  double M = 1325.0; // kg
279 
280  // power in W
281  double P_tract = A * v + B * v * v + C * v * v * v + M * a * v; // for sloped roads: +M*g*sin_theta*v
282 
283  /*
284  // "Category 7 vehicle" (e.g. a '92 Suzuki Swift)
285  double alpha = 1.01;
286  double beta = 0.0162;
287  double delta = 1.90e-06;
288  double zeta = 0.252;
289  double alpha1 = 0.985;
290  */
291 
292  // "Category 9 vehicle" (e.g. a '94 Dodge Spirit)
293  double alpha = 1.11;
294  double beta = 0.0134;
295  double delta = 1.98e-06;
296  double zeta = 0.241;
297  double alpha1 = 0.973;
298 
299  if (P_tract <= 0) return alpha1;
300  return alpha + beta * v * 3.6 + delta * v * v * v * (3.6 * 3.6 * 3.6) + zeta * a * v;
301 }

303 Coord TraCIMobility::calculateHostPosition(const Coord& vehiclePos) const
304 {
305  Coord corPos;
306  if (hostPositionOffset >= 0.001) {
307  // calculate antenna position of vehicle according to antenna offset
308  corPos = vehiclePos - (heading.toCoord() * hostPositionOffset);
309  }
310  else {
311  corPos = vehiclePos;
312  }
313  return corPos;
