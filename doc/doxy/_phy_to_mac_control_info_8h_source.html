33 /**
34  * @brief Controlinfo for packets which are send from Physical
35  * layer to the MAC layer.
36  *
37  * The ControlInfo contains the the DeciderResult of the Decider.
38  * @ingroup phyLayer
39  * @ingroup macLayer
40  */
41 class VEINS_API PhyToMacControlInfo : public cObject {
42 protected:
43  /** The result of the decider evaluation.*/
44  DeciderResult* result;
45 
46 public:
47  /**
48  * @brief Initializes the PhyToMacControlInfo with the passed DeciderResult.
49  *
50  * NOTE: PhyToMacControlInfo takes ownership of the passed DeciderResult!
51  */
52  PhyToMacControlInfo(DeciderResult* result)
53  : result(result)
54  {
55  }
56 
57  /**
58  * @brief Clean up the DeciderResult.
59  */
60  ~PhyToMacControlInfo() override
61  {
62  if (result) delete result;
63  }
64 
65  /**
66  * @brief Returns the result of the evaluation of the Decider.
67  */
68  DeciderResult* getDeciderResult() const
69  {
70  return result;
71  }
72 
73  /**
74  * @brief Attaches a "control info" structure (object) to the message pMsg.
75  *
76  * This is most useful when passing packets between protocol layers
77  * of a protocol stack, the control info will contain the decider result.
78  *
79  * The "control info" object will be deleted when the message is deleted.
80  * Only one "control info" structure can be attached (the second
81  * setL3ToL2ControlInfo() call throws an error).
82  *
83  * @param pMsg The message where the "control info" shall be attached.
84  * @param pDeciderResult The decider results.
85  */
86  static cObject* const setControlInfo(cMessage* const pMsg, DeciderResult* const pDeciderResult)
87  {
88  PhyToMacControlInfo* const cCtrlInfo = new PhyToMacControlInfo(pDeciderResult);
89  pMsg->setControlInfo(cCtrlInfo);
90 
91  return cCtrlInfo;
92  }
93  /**
94  * @brief extracts the decider result from message "control info".
95  */
96  static DeciderResult* const getDeciderResult(cMessage* const pMsg)
97  {
98  return getDeciderResultFromControlInfo(pMsg->getControlInfo());
99  }
100  /**
101  * @brief extracts the decider result from message "control info".
102  */
103  static DeciderResult* const getDeciderResultFromControlInfo(cObject* const pCtrlInfo)
104  {
105  PhyToMacControlInfo* const cCtrlInfo = dynamic_cast<PhyToMacControlInfo* const>(pCtrlInfo);
106 
107  if (cCtrlInfo) return cCtrlInfo->getDeciderResult();
108  return nullptr;
109  }
110 };
