32 /**
33  * Helper structure representing a wireless ray from a sender to a receiver.
34  *
35  * Contains pre-computed values to speed up calls to intersect with the same ray but different boxes.
36  */
37 struct Ray {
38  Point origin;
39  Point destination;
40  Point direction;
41  Point invDirection;
42  struct {
43  size_t x;
44  size_t y;
45  } sign;
46  double length;
47 };
48 
49 /**
50  * Return a Ray struct for fast intersection tests from sender to receiver.
51  */
52 Ray makeRay(const Point& sender, const Point& receiver)
53 {
54  const double dir_x = receiver.x - sender.x;
55  const double dir_y = receiver.y - sender.y;
56  Ray ray;
57  ray.origin = sender;
58  ray.destination = receiver;
59  ray.length = std::sqrt(dir_x * dir_x + dir_y * dir_y);
60  ray.direction.x = dir_x / ray.length;
61  ray.direction.y = dir_y / ray.length;
62  ray.invDirection.x = 1.0 / ray.direction.x;
63  ray.invDirection.y = 1.0 / ray.direction.y;
64  ray.sign.x = ray.invDirection.x < 0;
65  ray.sign.y = ray.invDirection.y < 0;
66  return ray;
67 }
68 /**
69  * Return whether ray intersects with box.
70  *
71  * Based on:
72  * Amy Williams, Steve Barrus, R. Keith Morley & Peter Shirley (2005) An Efficient and Robust Ray-Box Intersection Algorithm, Journal of Graphics Tools, 10:1, 49-54, DOI: 10.1080/2151237X.2005.10129188
73  */
74 bool intersects(const Ray& ray, const Box& box)
75 {
76  const double x[2]{box.p1.x, box.p2.x};
77  const double y[2]{box.p1.y, box.p2.y};
78  double tmin = (x[ray.sign.x] - ray.origin.x) * ray.invDirection.x;
79  double tmax = (x[1 - ray.sign.x] - ray.origin.x) * ray.invDirection.x;
80  double tymin = (y[ray.sign.y] - ray.origin.y) * ray.invDirection.y;
81  double tymax = (y[1 - ray.sign.y] - ray.origin.y) * ray.invDirection.y;
82 
83  if ((tmin > tymax) || (tymin > tmax)) return false;
84  if (tymin > tmin) tmin = tymin;
85  if (tymax < tmax) tmax = tymax;
86  return (tmin < ray.length) && (tmax > 0);
87 }

93 BBoxLookup::BBoxLookup(const std::vector<Obstacle*>& obstacles, std::function<BBoxLookup::Box(Obstacle*)> makeBBox, double scenarioX, double scenarioY, int cellSize)
94  : bboxes()
95  , obstacleLookup()
96  , bboxCells()
97  , cellSize(cellSize)
98  , numCols(std::floor(scenarioX / cellSize) + 1)
99  , numRows(std::floor(scenarioY / cellSize) + 1)
100 {
101  // phase 1: build unordered collection of cells
102  // initialize proto-cells (cells in non-contiguos memory)
103  ASSERT(scenarioX > 0);
104  ASSERT(scenarioY > 0);
105  ASSERT(numCols * cellSize >= scenarioX);
106  ASSERT(numRows * cellSize >= scenarioY);
107  const size_t numCells = numCols * numRows;
108  std::vector<std::vector<BBoxLookup::Box>> protoCells(numCells);
109  std::vector<std::vector<Obstacle*>> protoLookup(numCells);
110  // fill protoCells with boundingBoxes
111  size_t numEntries = 0;
112  for (const auto obstaclePtr : obstacles) {
113  auto bbox = makeBBox(obstaclePtr);
114  const size_t fromCol = std::max(0, int(bbox.p1.x / cellSize));
115  const size_t toCol = std::max(0, int(bbox.p2.x / cellSize));
116  const size_t fromRow = std::max(0, int(bbox.p1.y / cellSize));
117  const size_t toRow = std::max(0, int(bbox.p2.y / cellSize));
118  for (size_t row = fromRow; row <= toRow; ++row) {
119  for (size_t col = fromCol; col <= toCol; ++col) {
120  const size_t cellIndex = col + row * numCols;
121  protoCells[cellIndex].push_back(bbox);
122  protoLookup[cellIndex].push_back(obstaclePtr);
123  ++numEntries;
124  ASSERT(protoCells[cellIndex].size() == protoLookup[cellIndex].size());
125  }
126  }
127  }
128 
129  // phase 2: derive read-only data structure with fast lookup
130  bboxes.reserve(numEntries);
131  obstacleLookup.reserve(numEntries);
132  bboxCells.reserve(numCells);
133  size_t index = 0;
134  for (size_t row = 0; row < numRows; ++row) {
135  for (size_t col = 0; col < numCols; ++col) {
136  const size_t cellIndex = col + row * numCols;
137  auto& currentCell = protoCells.at(cellIndex);
138  auto& currentLookup = protoLookup.at(cellIndex);
139  ASSERT(currentCell.size() == currentLookup.size());
140  const size_t count = currentCell.size();
141  // copy over bboxes and obstacle lookups (in strict order)
142  for (size_t entryIndex = 0; entryIndex < count; ++entryIndex) {
143  bboxes.push_back(currentCell.at(entryIndex));
144  obstacleLookup.push_back(currentLookup.at(entryIndex));
145  }
146  // create lookup table for this cell
147  bboxCells.push_back({index, count});
148  // forward index to begin of next cell
149  index += count;
150  ASSERT(bboxes.size() == index);
151  }
152  }
153  ASSERT(bboxes.size() == numEntries);
154  ASSERT(bboxes.size() == obstacleLookup.size());
155 }

157 std::vector<Obstacle*> BBoxLookup::findOverlapping(Point sender, Point receiver) const
158 {
159  std::vector<Obstacle*> overlappingObstacles;
160  const Box bbox{
161  {std::min(sender.x, receiver.x), std::min(sender.y, receiver.y)},
162  {std::max(sender.x, receiver.x), std::max(sender.y, receiver.y)},
163  };
164 
165  // determine coordinates for all cells touched by bbox
166  const size_t firstCol = std::max(0, int(bbox.p1.x / cellSize));
167  const size_t lastCol = std::max(0, int(bbox.p2.x / cellSize));
168  const size_t firstRow = std::max(0, int(bbox.p1.y / cellSize));
169  const size_t lastRow = std::max(0, int(bbox.p2.y / cellSize));
170  ASSERT(lastCol < numCols && lastRow < numRows);
171  // precompute transmission ray properties
172  const Ray ray = makeRay(sender, receiver);
173  // iterate over cells
174  for (size_t row = firstRow; row <= lastRow; ++row) {
175  for (size_t col = firstCol; col <= lastCol; ++col) {
176  // skip cell if ray does not intersect with the cell.
177  const Box cellBox = {{static_cast<double>(col * cellSize), static_cast<double>(row * cellSize)}, {static_cast<double>((col + 1) * cellSize), static_cast<double>((row + 1) * cellSize)}};
178  if (!intersects(ray, cellBox)) continue;
179  // derive cell for current cell coordinates
180  const size_t cellIndex = col + row * numCols;
181  const BBoxCell& cell = bboxCells.at(cellIndex);
182  // iterate over bboxes in each cell
183  for (size_t bboxIndex = cell.index; bboxIndex < cell.index + cell.count; ++bboxIndex) {
184  const Box& current = bboxes.at(bboxIndex);
185  // check for overlap with bbox (fast rejection)
186  if (current.p2.x < bbox.p1.x) continue;
187  if (current.p1.x > bbox.p2.x) continue;
188  if (current.p2.y < bbox.p1.y) continue;
189  if (current.p1.y > bbox.p2.y) continue;
190  // derive corresponding obstacle
191  if (!intersects(ray, current)) continue;
192  overlappingObstacles.push_back(obstacleLookup.at(bboxIndex));
193  }
194  }
195  }
196  return overlappingObstacles;
197 }
