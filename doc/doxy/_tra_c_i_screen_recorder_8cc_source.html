45 void TraCIScreenRecorder::handleMessage(cMessage* msg)
46 {
47  ASSERT(msg == takeScreenshot);
48 
49  // get dirname
50  const char* dirname = par("dirname").stringValue();
51  if (std::string(dirname) == "") {
52  dirname = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RESULTDIR);
53  }
54 
55  // get absolute path of dirname (the TraCI server might be running in a different directory than OMNeT++)
56  std::string dirname_abs;
57  {
58  char* s = realpath(dirname, nullptr);
59  if (!s) {
60  perror("cannot open output directory");
61  throw cRuntimeError("cannot open output directory '%s'", dirname);
62  }
63  dirname_abs = s;
64  free(s);
65  }
66 
67  // get filename template
68  std::string filenameTemplate = par("filenameTemplate").stdstringValue();
69  if (filenameTemplate == "") {
70  const char* myRunID = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RUNID);
71  filenameTemplate = "screenshot-" + std::string(myRunID) + "-@%08.2f.png";
72 #ifdef _WIN32
73  // replace ':' with '-'
74  for (std::string::iterator i = filenameTemplate.begin(); i != filenameTemplate.end(); ++i)
75  if (*i == ':') *i = '-';
76 #endif /* _WIN32 */
77  }
78 
79  // assemble filename
80  std::string filename;
81  {
82  std::string tmpl = dirname_abs + "/" + filenameTemplate;
83  char buf[1024];
84  snprintf(buf, 1024, tmpl.c_str(), simTime().dbl());
85  buf[1023] = 0;
86  filename = buf;
87  }
88 
89  // take screenshot
90  TraCIScenarioManager* manager = TraCIScenarioManagerAccess().get();
91  ASSERT(manager);
92  TraCICommandInterface* traci = manager->getCommandInterface();
93  if (!traci) {
94  throw cRuntimeError("Cannot create screenshot: TraCI is not connected yet");
95  }
96  TraCICommandInterface::GuiView view = traci->guiView(par("viewName"));
97  view.takeScreenshot(filename);
98 
99  // schedule next screenshot
100  simtime_t stop = par("stop");
101  if ((stop == -1) || (simTime() < stop)) {
102  scheduleAt(simTime() + par("interval"), takeScreenshot);
103  }
104 }
