31 /**
32  * @brief
33  * This class represents an antenna whose gain is calculated from given samples in the horizontal plane.
34  * The respective gain is therefore dependent on the azimuth angle.
35  * The user has to provide the samples, which are assumed to be distributed equidistantly.
36  * As the power is assumed to be relative to an isotropic radiator, the values have to be given in dBi.
37  * The values are stored in a mapping automatically supporting linear interpolation between samples.
38  * Optional randomness in terms of sample offsets and antenna rotation is supported.
39  *
40  * * An example antenna.xml for this Antenna can be the following:
41  * @verbatim
42  <?xml version="1.0" encoding="UTF-8"?>
43  <root>
44  <Antenna type="SampledAntenna1D" id="antenna1">
45  <!-- Write the samples in the value attribute, separated by spaces. The values will be -->
46  <!-- distributed equidistantly, e.g. 4 values will be placed at 0°, 90°, 180° and 270° -->
47  <parameter name="samples" type="string" value="3 -3 3 -3"/>
48 
49  <!-- Options for random offsets are as follows. -->
50  <!-- The mean of the given distribution has to be 0 (so that the overall offset is close to 0dBi) -->
51  <!-- <parameter name="random-offsets" type="string" value="uniform a b"/> -->
52  <!-- <parameter name="random-offsets" type="string" value="normal mean stddev"/> -->
53  <!-- <parameter name="random-offsets" type="string" value="triang a b c"/> -->
54  <parameter name="random-offsets" type="string" value="uniform -0.01 0.01"/>
55 
56  <!-- Options for random rotation of the antennas are the same, but mean doesn't have to be 0. -->
57  <parameter name="random-rotation" type="string" value="uniform -1 1"/>
58  </Antenna>
59  </root>
60  @endverbatim
61  *
62  *
63  * @author Alexander Brummer
64  *
65  *
66  * @see Antenna
67  * @see BasePhyLayer
68  */
