72 void BaseMobility::initialize(int stage)
73 {
74  BaseModule::initialize(stage);
75  if (stage == 0) {
76  EV_TRACE << "initializing BaseMobility stage " << stage << endl;
77 
78  hasPar("scaleNodeByDepth") ? scaleNodeByDepth = par("scaleNodeByDepth").boolValue() : scaleNodeByDepth = true;
79 
80  // get utility pointers (world and host)
81  world = FindModule<BaseWorldUtility*>::findGlobalModule();
82  if (world == nullptr) throw cRuntimeError("Could not find BaseWorldUtility module");
83 
84  EV_TRACE << "initializing BaseUtility stage " << stage << endl; // for node position
85 
86  if (hasPar("updateInterval")) {
87  updateInterval = par("updateInterval");
88  }
89  else {
90  updateInterval = 0;
91  }
92 
93  // initialize Move parameter
94  bool use2D = world->use2D();
95 
96  // initalize position with random values
97  Coord pos = world->getRandomPosition();
98 
99  // read coordinates from parameters if available
100  double x = hasPar("x") ? par("x").doubleValue() : -1;
101  double y = hasPar("y") ? par("y").doubleValue() : -1;
102  double z = hasPar("z") ? par("z").doubleValue() : -1;
103 
104  // set position with values from parameters if available
105  if (x > -1) pos.x = x;
106  if (y > -1) pos.y = y;
107  if (!use2D && z > -1) pos.z = z;
108 
109  if (!hasPar("xOrientation") || !hasPar("yOrientation")) {
110  throw cRuntimeError("Orientation coordinates in x and y direction have to specified (necessary for antenna gain calculation)");
111  }
112  else {
113  double zOrient = hasPar("zOrientation") ? par("zOrientation").doubleValue() : 0;
114  Coord orient(par("xOrientation").doubleValue(), par("yOrientation").doubleValue(), zOrient);
115  move.setOrientationByVector(orient);
116  }
117 
118  // set start-position and start-time (i.e. current simulation-time) of the Move
119  move.setStart(pos);
120  EV_TRACE << "start pos: " << move.getStartPos().info() << endl;
121 
122  // check whether position is within the playground
123  if (!isInBoundary(move.getStartPos(), Coord::ZERO, *world->getPgs())) {
124  throw cRuntimeError("node position specified in omnetpp.ini exceeds playgroundsize");
125  }
126 
127  // set speed and direction of the Move
128  move.setSpeed(0);
129  move.setDirectionByVector(Coord::ZERO);
130  }
131  else if (stage == 1) {
132  EV_TRACE << "initializing BaseMobility stage " << stage << endl;
133 
134  // get playground scaling
135  if (world->getParentModule() != nullptr) {
136  const cDisplayString& dispWorldOwner = world->getParentModule()->getDisplayString();
137 
138  if (dispWorldOwner.containsTag("bgb")) {
139  double origPGWidth = 0.0;
140  double origPGHeight = 0.0;
141  // normally this should be equal to playground size
142  std::istringstream(dispWorldOwner.getTagArg("bgb", 0)) >> origPGWidth;
143  std::istringstream(dispWorldOwner.getTagArg("bgb", 1)) >> origPGHeight;
144 
145  // bgb of zero means size isn't set manually
146  if (origPGWidth > 0) {
147  playgroundScaleX = origPGWidth / playgroundSizeX();
148  }
149  if (origPGHeight > 0) {
150  playgroundScaleY = origPGHeight / playgroundSizeY();
151  }
152  }
153  }
154 
155  // get original display of host
156  cDisplayString& disp = findHost()->getDisplayString();
157 
158  // get host width and height
159  if (disp.containsTag("b")) {
160  std::istringstream(disp.getTagArg("b", 0)) >> origDisplayWidth;
161  std::istringstream(disp.getTagArg("b", 1)) >> origDisplayHeight;
162  }
163  // get hosts icon size
164  if (disp.containsTag("i")) {
165  // choose a appropriate icon size (only if a icon is specified)
166  origIconSize = iconSizeTagToSize(disp.getTagArg("is", 0));
167  }
168 
169  // print new host position on the screen and update bb info
170  updatePosition();
171 
172  if (move.getSpeed() > 0 && updateInterval > 0) {
173  EV_TRACE << "Host is moving, speed=" << move.getSpeed() << " updateInterval=" << updateInterval << endl;
174  moveMsg = new cMessage("move", MOVE_HOST);
175  // host moves the first time after some random delay to avoid synchronized movements
176  scheduleAt(simTime() + uniform(0, updateInterval), moveMsg);
177  }
178  }
179 }

249 void BaseMobility::handleBorderMsg(cMessage* msg)
250 {
251  EV_TRACE << "start MOVE_TO_BORDER:" << move.info() << endl;
252 
253  BorderMsg* bMsg = static_cast<BorderMsg*>(msg);
254 
255  switch (bMsg->getPolicy()) {
256  case REFLECT:
257  move.setStart(bMsg->getStartPos());
258  move.setDirectionByVector(bMsg->getDirection());
259  break;
260 
261  case WRAP:
262  move.setStart(bMsg->getStartPos());
263  break;
264 
265  case PLACERANDOMLY:
266  move.setStart(bMsg->getStartPos());
267  EV_TRACE << "new random position: " << move.getStartPos().info() << endl;
268  break;
269 
270  case RAISEERROR:
271  throw cRuntimeError("node moved outside the playground");
272  break;
273 
274  default:
275  throw cRuntimeError("Unknown BorderPolicy!");
276  break;
277  }
278 
279  fixIfHostGetsOutside();
280 
281  updatePosition();
282 
283  delete bMsg;
284 
285  EV_TRACE << "end MOVE_TO_BORDER:" << move.info() << endl;
286 }

288 void BaseMobility::updatePosition()
289 {
290  EV_DEBUG << "updatePosition: " << move.info() << endl;
291 
292  // publish the the new move
293  emit(mobilityStateChangedSignal, this);
294 
295  if (hasGUI()) {
296  std::ostringstream osDisplayTag;
297 #ifdef __APPLE__
298  const int iPrecis = 0;
299 #else
300  const int iPrecis = 5;
301 #endif
302  cDisplayString& disp = findHost()->getDisplayString();
303 
304  // setup output stream
305  osDisplayTag << std::fixed;
306  osDisplayTag.precision(iPrecis);
307 
308  if (playgroundScaleX != 1.0) {
309  osDisplayTag << (move.getStartPos().x * playgroundScaleX);
310  }
311  else {
312  osDisplayTag << (move.getStartPos().x);
313  }
314  disp.setTagArg("p", 0, osDisplayTag.str().data());
315 
316  osDisplayTag.str(""); // reset
317  if (playgroundScaleY != 1.0) {
318  osDisplayTag << (move.getStartPos().y * playgroundScaleY);
319  }
320  else {
321  osDisplayTag << (move.getStartPos().y);
322  }
323  disp.setTagArg("p", 1, osDisplayTag.str().data());
324 
325  if (!world->use2D() && scaleNodeByDepth) {
326  const double minScale = 0.25;
327  const double maxScale = 1.0;
328 
329  // scale host dependent on their z coordinate to simulate a depth
330  // effect
331  // z-coordinate of zero maps to a scale of maxScale (very close)
332  // z-coordinate of playground size z maps to size of minScale (far away)
333  double depthScale = minScale + (maxScale - minScale) * (1.0 - move.getStartPos().z / playgroundSizeZ());
334 
335  if (origDisplayWidth > 0.0 && origDisplayHeight > 0.0) {
336  osDisplayTag.str(""); // reset
337  osDisplayTag << (origDisplayWidth * depthScale);
338  disp.setTagArg("b", 0, osDisplayTag.str().data());
339 
340  osDisplayTag.str(""); // reset
341  osDisplayTag << (origDisplayHeight * depthScale);
342  disp.setTagArg("b", 1, osDisplayTag.str().data());
343  }
344 
345  if (origIconSize > 0) {
346  // choose a appropriate icon size (only if a icon is specified)
347  disp.setTagArg("is", 0, iconSizeToTag(origIconSize * depthScale));
348  }
349  }
350  }

353 void BaseMobility::reflectCoordinate(BorderHandling border, Coord& c)
354 {
355  switch (border) {
356  case X_SMALLER:
357  c.x = (-c.x);
358  break;
359  case X_BIGGER:
360  c.x = (2 * playgroundSizeX() - c.x);
361  break;
362 
363  case Y_SMALLER:
364  c.y = (-c.y);
365  break;
366  case Y_BIGGER:
367  c.y = (2 * playgroundSizeY() - c.y);
368  break;
369 
370  case Z_SMALLER:
371  c.z = (-c.z);
372  break;
373  case Z_BIGGER:
374  c.z = (2 * playgroundSizeZ() - c.z);
375  break;
376 
377  case NOWHERE:
378  default:
379  throw cRuntimeError("wrong border handling case!");
380  break;
381  }
382 }

385 {

390  switch (wo) {

392  case X_BIGGER:
393  step.x = (-step.x);

395  break;
396 
397  case Y_SMALLER:
398  case Y_BIGGER:
399  step.y = (-step.y);
400  angle = -angle;
401  break;
402 
403  case Z_SMALLER:
404  case Z_BIGGER:
405  step.z = (-step.z);
406  angle = -angle;
407  break;
408 
409  case NOWHERE:
410  default:
411  throw cRuntimeError("wrong border handling case!");
412  break;
413  }

449 BaseMobility::BorderHandling BaseMobility::checkIfOutside(Coord targetPos, Coord& borderStep)
450 {
451  BorderHandling outside = NOWHERE;
452 
453  // Testing x-value
454  if (targetPos.x < 0) {
455  borderStep.x = (-move.getStartPos().x);
456  outside = X_SMALLER;
457  }
458  else if (targetPos.x >= playgroundSizeX()) {
459  borderStep.x = (playgroundSizeX() - move.getStartPos().x);
460  outside = X_BIGGER;
461  }
462 
463  // Testing y-value
464  if (targetPos.y < 0) {
465  borderStep.y = (-move.getStartPos().y);
466 
467  if (outside == NOWHERE || fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.y / move.getDirection().y)) {
468  outside = Y_SMALLER;
469  }
470  }
471  else if (targetPos.y >= playgroundSizeY()) {
472  borderStep.y = (playgroundSizeY() - move.getStartPos().y);
473 
474  if (outside == NOWHERE || fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.y / move.getDirection().y)) {
475  outside = Y_BIGGER;
476  }
477  }
478 
479  // Testing z-value
480  if (!world->use2D()) {
481  // going to reach the lower z-border
482  if (targetPos.z < 0) {
483  borderStep.z = (-move.getStartPos().z);
484 
485  // no border reached so far
486  if (outside == NOWHERE) {
487  outside = Z_SMALLER;
488  }
489  // an y-border is reached earliest so far, test whether z-border
490  // is reached even earlier
491  else if ((outside == Y_SMALLER || outside == Y_BIGGER) && fabs(borderStep.y / move.getDirection().y) > fabs(borderStep.z / move.getDirection().z)) {
492  outside = Z_SMALLER;
493  }
494  // an x-border is reached earliest so far, test whether z-border
495  // is reached even earlier
496  else if ((outside == X_SMALLER || outside == X_BIGGER) && fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.z / move.getDirection().z)) {
497  outside = Z_SMALLER;
498  }
499  }
500  // going to reach the upper z-border
501  else if (targetPos.z >= playgroundSizeZ()) {
502  borderStep.z = (playgroundSizeZ() - move.getStartPos().z);
503 
504  // no border reached so far
505  if (outside == NOWHERE) {
506  outside = Z_BIGGER;
507  }
508  // an y-border is reached earliest so far, test whether z-border
509  // is reached even earlier
510  else if ((outside == Y_SMALLER || outside == Y_BIGGER) && fabs(borderStep.y / move.getDirection().y) > fabs(borderStep.z / move.getDirection().z)) {
511  outside = Z_BIGGER;
512  }
513  // an x-border is reached earliest so far, test whether z-border
514  // is reached even earlier
515  else if ((outside == X_SMALLER || outside == X_BIGGER) && fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.z / move.getDirection().z)) {
516  outside = Z_BIGGER;
517  }
518  }
519  }
520 
521  EV_TRACE << "checkIfOutside, outside=" << outside << " borderStep: " << borderStep.info() << endl;
522 
523  return outside;
524 }

526 void BaseMobility::goToBorder(BorderPolicy policy, BorderHandling wo, Coord& borderStep, Coord& borderStart)
527 {
528  double factor;
529 
530  EV_TRACE << "goToBorder: startPos: " << move.getStartPos().info() << " borderStep: " << borderStep.info() << " BorderPolicy: " << policy << " BorderHandling: " << wo << endl;
531 
532  switch (wo) {
533  case X_SMALLER:
534  factor = borderStep.x / move.getDirection().x;
535  borderStep.y = (factor * move.getDirection().y);
536  if (!world->use2D()) {
537  borderStep.z = (factor * move.getDirection().z); // 3D case
538  }
539 
540  if (policy == WRAP) {
541  borderStart.x = (playgroundSizeX());
542  borderStart.y = (move.getStartPos().y + borderStep.y);
543  if (!world->use2D()) {
544  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
545  }
546  }
547  break;
548 
549  case X_BIGGER:
550  factor = borderStep.x / move.getDirection().x;
551  borderStep.y = (factor * move.getDirection().y);
552  if (!world->use2D()) {
553  borderStep.z = (factor * move.getDirection().z); // 3D case
554  }
555 
556  if (policy == WRAP) {
557  borderStart.x = (0);
558  borderStart.y = (move.getStartPos().y + borderStep.y);
559  if (!world->use2D()) {
560  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
561  }
562  }
563  break;
564 
565  case Y_SMALLER:
566  factor = borderStep.y / move.getDirection().y;
567  borderStep.x = (factor * move.getDirection().x);
568  if (!world->use2D()) {
569  borderStep.z = (factor * move.getDirection().z); // 3D case
570  }
571 
572  if (policy == WRAP) {
573  borderStart.y = (playgroundSizeY());
574  borderStart.x = (move.getStartPos().x + borderStep.x);
575  if (!world->use2D()) {
576  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
577  }
578  }
579  break;
580 
581  case Y_BIGGER:
582  factor = borderStep.y / move.getDirection().y;
583  borderStep.x = (factor * move.getDirection().x);
584  if (!world->use2D()) {
585  borderStep.z = (factor * move.getDirection().z); // 3D case
586  }
587 
588  if (policy == WRAP) {
589  borderStart.y = (0);
590  borderStart.x = (move.getStartPos().x + borderStep.x);
591  if (!world->use2D()) {
592  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
593  }
594  }
595  break;
596 
597  case Z_SMALLER: // here we are definitely in 3D
598  factor = borderStep.z / move.getDirection().z;
599  borderStep.x = (factor * move.getDirection().x);
600  borderStep.y = (factor * move.getDirection().y);
601 
602  if (policy == WRAP) {
603  borderStart.z = (playgroundSizeZ());
604  borderStart.x = (move.getStartPos().x + borderStep.x);
605  borderStart.y = (move.getStartPos().y + borderStep.y);
606  }
607  break;
608 
609  case Z_BIGGER: // here we are definitely in 3D
610  factor = borderStep.z / move.getDirection().z;
611  borderStep.x = (factor * move.getDirection().x);
612  borderStep.y = (factor * move.getDirection().y);
613 
614  if (policy == WRAP) {
615  borderStart.z = (0);
616  borderStart.x = (move.getStartPos().x + borderStep.x);
617  borderStart.y = (move.getStartPos().y + borderStep.y);
618  }
619  break;
620 
621  default:
622  throw cRuntimeError("invalid state in goToBorder switch!");
623  break;
624  }
625 
626  EV_TRACE << "goToBorder: startPos: " << move.getStartPos().info() << " borderStep: " << borderStep.info() << " borderStart: " << borderStart.info() << " factor: " << factor << endl;
627 }

629 bool BaseMobility::handleIfOutside(BorderPolicy policy, Coord& stepTarget, Coord& targetPos, Coord& step, double& angle)
630 {
631  // where did the host leave the playground?
632  BorderHandling wo;
633 
634  // step to reach the border
635  Coord borderStep;
636 
637  wo = checkIfOutside(stepTarget, borderStep);
638 
639  // just return if the next step is not outside the playground
640  if (wo == NOWHERE) return false;
641 
642  EV_TRACE << "handleIfOutside:stepTarget = " << stepTarget.info() << endl;
643 
644  // new start position after the host reaches the border
645  Coord borderStart;
646  // new direction the host has to move to
647  Coord borderDirection;
648  // time to reach the border
649  simtime_t borderInterval;
650 
651  EV_TRACE << "old values: stepTarget: " << stepTarget.info() << " step: " << step.info() << " targetPos: " << targetPos.info() << " angle: " << angle << endl;
652 
653  // which border policy is to be followed
654  switch (policy) {
655  case REFLECT:
656  reflectIfOutside(wo, stepTarget, targetPos, step, angle);
657  break;
658  case WRAP:
659  wrapIfOutside(wo, stepTarget, targetPos);
660  break;
661  case PLACERANDOMLY:
662  placeRandomlyIfOutside(targetPos);
663  break;
664  case RAISEERROR:
665  break;
666  }
667 
668  EV_TRACE << "new values: stepTarget: " << stepTarget.info() << " step: " << step.info() << " angle: " << angle << endl;
669 
670  // calculate the step to reach the border
671  goToBorder(policy, wo, borderStep, borderStart);
672 
673  // calculate the time to reach the border
674  borderInterval = (borderStep.length()) / move.getSpeed();
675 
676  // calculate new start position
677  // NOTE: for WRAP this is done in goToBorder
678  switch (policy) {
679  case REFLECT: {
680  borderStart = move.getStartPos() + borderStep;
681  double d = stepTarget.distance(borderStart);
682  borderDirection = (stepTarget - borderStart) / d;
683  break;
684  }
685  case PLACERANDOMLY:
686  borderStart = targetPos;
687  stepTarget = targetPos;
688  break;
689 
690  case WRAP:
691  case RAISEERROR:
692  break;
693 
694  default:
