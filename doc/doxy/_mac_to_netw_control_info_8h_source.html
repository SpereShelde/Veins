126  /**
127  * @brief Attaches a "control info" structure (object) to the message pMsg.
128  *
129  * This is most useful when passing packets between protocol layers
130  * of a protocol stack, the control info will contain the source MAC address.
131  *
132  * The "control info" object will be deleted when the message is deleted.
133  * Only one "control info" structure can be attached (the second
134  * setL3ToL2ControlInfo() call throws an error).
135  *
136  * @param pMsg The message where the "control info" shall be attached.
137  * @param pSrcAddr The MAC address of the message sender.
138  */
139  static cObject* const setControlInfo(cMessage* const pMsg, const LAddress::L2Type& pSrcAddr)
140  {
141  MacToNetwControlInfo* const cCtrlInfo = new MacToNetwControlInfo(pSrcAddr);
142  pMsg->setControlInfo(cCtrlInfo);
143 
144  return cCtrlInfo;
145  }
146  static const LAddress::L2Type& getAddress(cMessage* const pMsg)
147  {
148  return getAddressFromControlInfo(pMsg->getControlInfo());
149  }
150  static const LAddress::L2Type& getAddressFromControlInfo(cObject* const pCtrlInfo)
151  {
152  MacToNetwControlInfo* const cCtrlInfo = dynamic_cast<MacToNetwControlInfo* const>(pCtrlInfo);
153 
154  if (cCtrlInfo) return cCtrlInfo->getLastHopMac();
155 
156  return LAddress::L2NULL();
157  }
158 };
