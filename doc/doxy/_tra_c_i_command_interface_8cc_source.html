60 std::pair<uint32_t, std::string> TraCICommandInterface::getVersion()
61 {
62  TraCIConnection::Result result;
63  TraCIBuffer buf = connection.query(CMD_GETVERSION, TraCIBuffer(), &result);
64 
65  if (!result.success) {
66  ASSERT(buf.eof());

68  }
69 
70  uint8_t cmdLength;
71  buf >> cmdLength;
72  uint8_t commandResp;
73  buf >> commandResp;
74  ASSERT(commandResp == CMD_GETVERSION);
75  uint32_t apiVersion;
76  buf >> apiVersion;
77  std::string serverVersion;
78  buf >> serverVersion;
79  ASSERT(buf.eof());
80 
81  return std::pair<uint32_t, std::string>(apiVersion, serverVersion);

95 std::pair<TraCICoord, TraCICoord> TraCICommandInterface::initNetworkBoundaries(int margin)
96 {
97  // query road network boundaries
98  TraCIBuffer buf = connection.query(CMD_GET_SIM_VARIABLE, TraCIBuffer() << static_cast<uint8_t>(VAR_NET_BOUNDING_BOX) << std::string("sim0"));
99  uint8_t cmdLength_resp;
100  buf >> cmdLength_resp;
101  uint8_t commandId_resp;
102  buf >> commandId_resp;
103  ASSERT(commandId_resp == RESPONSE_GET_SIM_VARIABLE);
104  uint8_t variableId_resp;
105  buf >> variableId_resp;
106  ASSERT(variableId_resp == VAR_NET_BOUNDING_BOX);
107  std::string simId;
108  buf >> simId;
109  uint8_t typeId_resp;
110  buf >> typeId_resp;
111  ASSERT(typeId_resp == getNetBoundaryType());
112  if (getNetBoundaryType() == TYPE_POLYGON) {
113  // Polygons can have an arbitrary number of tuples, so check that it is actually 2
114  uint8_t npoints;
115  buf >> npoints;
116  ASSERT(npoints == 2);
117  }
118  double x1;
119  buf >> x1;
120  double y1;
121  buf >> y1;
122  double x2;
123  buf >> x2;
124  double y2;
125  buf >> y2;
126  ASSERT(buf.eof());
127  EV_DEBUG << "TraCI reports network boundaries (" << x1 << ", " << y1 << ")-(" << x2 << ", " << y2 << ")" << endl;
128  TraCICoord nb1(x1, y1);
129  TraCICoord nb2(x2, y2);
130  connection.setNetbounds(nb1, nb2, margin);
131  return {nb1, nb2};
132 }

483 std::list<std::list<TraCITrafficLightLink>> TraCICommandInterface::Trafficlight::getControlledLinks() const
484 {
485  uint8_t resultTypeId = TYPE_COMPOUND;
486  uint8_t commandId = CMD_GET_TL_VARIABLE;
487  uint8_t variableId = TL_CONTROLLED_LINKS;
488  std::string objectId = trafficLightId;
489  uint8_t responseId = RESPONSE_GET_TL_VARIABLE;
490 
491  TraCIBuffer buf = connection->query(commandId, TraCIBuffer() << variableId << objectId);
492  // generic header
493  uint8_t cmdLength;
494  buf >> cmdLength;
495  if (cmdLength == 0) {
496  uint32_t cmdLengthX;
497  buf >> cmdLengthX;
498  }
499  uint8_t commandId_r;
500  buf >> commandId_r;
501  ASSERT(commandId_r == responseId);
502  uint8_t varId;
503  buf >> varId;
504  ASSERT(varId == variableId);
505  std::string objectId_r;
506  buf >> objectId_r;
507  ASSERT(objectId_r == objectId);
508  uint8_t resType_r;
509  buf >> resType_r;
510  ASSERT(resType_r == resultTypeId);
511 
512  int32_t compoundSize;
513  buf >> compoundSize; // nr of fields in the compound
514  int32_t nrOfSignals = buf.readTypeChecked<int32_t>(TYPE_INTEGER);
515  std::list<std::list<TraCITrafficLightLink>> controlledLinks;
516  for (int32_t i = 0; i < nrOfSignals; ++i) {
517  int32_t nrOfLinks = buf.readTypeChecked<int32_t>(TYPE_INTEGER);
518  std::list<TraCITrafficLightLink> linksOfSignal;
519  uint8_t stringListType;
520  buf >> stringListType;
521  ASSERT(stringListType == static_cast<uint8_t>(TYPE_STRINGLIST));
522  for (int32_t j = 0; j < nrOfLinks; ++j) {
523  int32_t stringCount;
524  buf >> stringCount;
525  ASSERT(stringCount == 3);
526  TraCITrafficLightLink link;
527  buf >> link.incoming;
528  buf >> link.outgoing;
529  buf >> link.internal;
530  linksOfSignal.push_back(link);
531  }
532  controlledLinks.push_back(linksOfSignal);
533  }
534  return controlledLinks;
535 }

579  ASSERT(resType_r == resultTypeId);

583  int32_t nrOfLogics = buf.readTypeChecked<int32_t>(TYPE_INTEGER); // nr of logics in the compound
584 
585  for (int32_t i = 0; i < nrOfLogics; ++i) {
586  TraCITrafficLightProgram::Logic logic;

588  logic.type = buf.readTypeChecked<int32_t>(TYPE_INTEGER); // (sub)type - currently just a 0
589  logic.parameter = buf.readTypeChecked<int32_t>(TYPE_COMPOUND); // (sub)parameter - currently just a 0
590  logic.currentPhase = buf.readTypeChecked<int32_t>(TYPE_INTEGER); // phase index
591

593  for (int32_t j = 0; j < nrOfPhases; ++j) {
594  TraCITrafficLightProgram::Phase phase;
595  phase.duration = buf.readTypeChecked<simtime_t>(traci->getTimeType()); // default duration of phase
596  phase.minDuration = buf.readTypeChecked<simtime_t>(traci->getTimeType()); // minimum duration of phase
597  phase.maxDuration = buf.readTypeChecked<simtime_t>(traci->getTimeType()); // maximum duration of phase
598  phase.state = buf.readTypeChecked<std::string>(TYPE_STRING); // phase definition (like "[ryg]*")
599  logic.phases.push_back(phase);
600  }
601  program.addLogic(logic);
602  }
603  }
604  else if (apiVersion == 19 || apiVersion == 20) {
605  uint8_t commandId = CMD_GET_TL_VARIABLE;
606  uint8_t variableId = TL_COMPLETE_DEFINITION_RYG;
607  std::string objectId = trafficLightId;

617  buf >> cmdLengthX;
618  }

623  buf >> varId;

627  ASSERT(objectId_r == objectId);
628

630  for (int32_t i = 0; i < nrOfLogics; ++i) {
631  int32_t nrOfComps = buf.readTypeChecked<int32_t>(TYPE_COMPOUND);
632  ASSERT(nrOfComps == 5);

635  logic.id = buf.readTypeChecked<std::string>(TYPE_STRING); // program ID
636  logic.type = buf.readTypeChecked<int32_t>(TYPE_INTEGER); // (sub)type - currently just a 0
637  logic.currentPhase = buf.readTypeChecked<int32_t>(TYPE_INTEGER); // phase index
638

641  TraCITrafficLightProgram::Phase phase;
