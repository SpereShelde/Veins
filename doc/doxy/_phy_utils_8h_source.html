38 /**
39  * @brief The class that represents the Radio as a state machine.
40  *
41  * For this basic version we assume a minimal attenuation when the Radio is in
42  * state RX, and a maximum attenuation otherwise.
43  *
44  * @ingroup phyLayer
45  */
46 class VEINS_API Radio {
47 public:
48  /**
49  * @brief The state of the radio of the nic.
50  */
51  enum RadioState {
52  /** @brief receiving state*/
53  RX = 0,
54  /** @brief transmitting state*/
55  TX,
56  /** @brief sleeping*/
57  SLEEP,
58  /** @brief switching between two states*/
59  SWITCHING,
60 
61  /**
62  * @brief No real radio state just a counter constant for the amount of states.
63  *
64  * Sub-classing Radios which want to add more states can add their own
65  * states in their own enum beginning at the value of NUM_RADIO_STATES.
66  * They should also remember to update the "numRadioStates" member accordingly.
67  *
68  * @see RadioUWBIR for an example.
69  */
70  NUM_RADIO_STATES
71  };
72 
73 protected:
74  /** @brief Output vector for radio states.*/
75  cOutVector radioStates;
76  /** @brief Output vector for radio channels.*/
77  cOutVector radioChannels;
78 
79  /** @brief The current state the radio is in.*/
80  int state;
81  /** @brief The state the radio is currently switching to.*/
82  int nextState;
83 
84  /** @brief The number of radio states this Radio can be in.*/
85  const int numRadioStates;
86  /** @brief Array for storing switch-times between states.*/
87  simtime_t** swTimes;
88 
89  /** @brief Currently selected channel (varies between 0 and nbChannels-1).*/
90  int currentChannel;
91 
92  /** @brief Number of available channels. */
93  int nbChannels;
94 
95 public:
96  /**
97  * @brief Creates a new instance of this class.
98  *
99  * Since Radio hasn't a public constructor this is the only
100  * way to create an instance of this class.
101  *
102  * This method assures that the radio is initialized with the
103  * correct number of radio states. Sub classing Radios should also
104  * define a factory method like this instead of an public constructor.
105  */
106  static std::unique_ptr<Radio> createNewRadio(bool recordStats = false, int initialState = RX, int currentChannel = 0, int nbChannels = 1)
107  {
108  return std::unique_ptr<Radio>(new Radio(NUM_RADIO_STATES, recordStats, initialState, currentChannel, nbChannels));
109  }
110 
111  /**
112  * @brief Destructor for the Radio class
113  */
114  virtual ~Radio();
115 
116  /**
117  * @brief A function called by the Physical Layer to start the switching process to a new RadioState
118  *
119  * @return -1: Error code if the Radio is currently switching
120  * else: switching time from the current RadioState to the new RadioState
121  *
122  *
123  * The actual simtime must be passed, to create properly RSAMEntry
124  */
125  virtual simtime_t switchTo(int newState, simtime_t_cref now);
126 
127  /**
128  * @brief function called by PhyLayer in order to make an entry in the switch times matrix,
129  * i.e. set the time for switching from one state to another
130  *
131  */
132  virtual void setSwitchTime(int from, int to, simtime_t_cref time);
133 
134  /**
135  * @brief Returns the state the Radio is currently in
136  */
137  virtual int getCurrentState() const
138  {
139  return state;
140  }
141 
142  /**
143  * @brief called by PhyLayer when duration-time for the
144  * current switching process is up
145  *
146  * Radio checks whether it is in switching state (pre-condition)
147  * and switches to the target state
148  *
149  * The actual simtime must be passed, to create properly RSAMEntry
150  */
151  virtual void endSwitch(simtime_t_cref now);
152 
153  /**
154  * @brief Changes the current channel of the radio.
155  *
156  * @param newChannel The new channel to switch to, between 0 and
157  * nbChannels-1
158  */
159  void setCurrentChannel(int newChannel)
160  {
161  ASSERT(newChannel > -1);
162  ASSERT(newChannel < nbChannels);
163  currentChannel = newChannel;
164  radioChannels.record(currentChannel);
165  }
166 
167  /**
168  * @brief Returns the channel the radio is currently set to.
169  * @return The current channel of the radio, between 0 and nbChannels-1.
170  */
171  int getCurrentChannel()
172  {
173  return currentChannel;
174  }
175 
176 protected:
177  /**
178  * @brief Intern constructor to initialize the radio.
179  *
180  * By defining no default constructor we assure that sub classing radios
181  * have to explicitly call this constructor which assures they pass
182  * the correct number of radio states.
183  *
184  * The protected constructor + factory method solution assures that
185  * while sub-classing Radios HAVE to explicitly pass their correct amount
186  * of radio states to this constructor, the user (creator) of the Radio
187  * doesn't has to pass it or even know about it (which wouldn't be possible
188  * with a public constructor).
189  * Therefore sub classing Radios which could be sub-classed further should
190  * also do it this way.
191  */
192  Radio(int numRadioStates, bool recordStats, int initialState = RX, int currentChannel = 0, int nbChannels = 1);
193 
194 }; // end class Radio
