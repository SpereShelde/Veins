29 SampledAntenna1D::SampledAntenna1D(std::vector<double>& values, std::string offsetType, std::vector<double>& offsetParams, std::string rotationType, std::vector<double>& rotationParams, cRNG* rng)
30  : antennaGains(values.size() + 1)
31 {
32  distance = (2 * M_PI) / values.size();
33 
34  // instantiate a random number generator for sample offsets if one is specified
35  cRandom* offsetGen = nullptr;
36  if (offsetType == "uniform") {
37  if (!math::almost_equal(offsetParams[0], -offsetParams[1])) {
38  throw cRuntimeError("SampledAntenna1D::SampledAntenna1D(): The mean of the random distribution for the samples' offsets has to be 0.");
39  }
40  offsetGen = new cUniform(rng, offsetParams[0], offsetParams[1]);
41  }
42  else if (offsetType == "normal") {
43  if (!math::almost_equal<double>(offsetParams[0], 0)) {
44  throw cRuntimeError("SampledAntenna1D::SampledAntenna1D(): The mean of the random distribution for the samples' offsets has to be 0.");
45  }
46  offsetGen = new cNormal(rng, offsetParams[0], offsetParams[1]);
47  }
48  else if (offsetType == "triang") {
49  if (!math::almost_equal<double>((offsetParams[0] + offsetParams[1] + offsetParams[2]) / 3, 0)) {
50  throw cRuntimeError("SampledAntenna1D::SampledAntenna1D(): The mean of the random distribution for the samples' offsets has to be 0.");
51  }
52  offsetGen = new cTriang(rng, offsetParams[0], offsetParams[1], offsetParams[2]);
53  }
54 
55  // determine random rotation of the antenna if specified
56  cRandom* rotationGen = nullptr;
57  if (rotationType == "uniform") {
58  rotationGen = new cUniform(rng, rotationParams[0], rotationParams[1]);
59  }
60  else if (rotationType == "normal") {
61  rotationGen = new cNormal(rng, rotationParams[0], rotationParams[1]);
62  }
63  else if (rotationType == "triang") {
64  rotationGen = new cTriang(rng, rotationParams[0], rotationParams[1], rotationParams[2]);
65  }
66  rotation = (rotationGen == nullptr) ? 0 : rotationGen->draw();
67  if (rotationGen != nullptr) delete rotationGen;
68 
69  // transform to rad
70  rotation *= (M_PI / 180);
71 
72  // copy values and apply offset
73  for (unsigned int i = 0; i < values.size(); i++) {
74  double offset = 0;
75  if (offsetGen != nullptr) {
76  offset = offsetGen->draw();
77  // transform to rad
78  offset *= (M_PI / 180);
79  }
80  antennaGains[i] = values[i] + offset;
81  }
82  if (offsetGen != nullptr) delete offsetGen;
83 
84  // assign the value of 0 degrees to 360 degrees as well to assure correct interpolation (size allocated already before)
85  antennaGains[values.size()] = antennaGains[0];
86 }

92 double SampledAntenna1D::getGain(Coord ownPos, Coord ownOrient, Coord otherPos)
93 {
94  // get the line of sight vector
95  Coord los = otherPos - ownPos;
96  // calculate angle using atan2
97  double angle = atan2(los.y, los.x) - atan2(ownOrient.y, ownOrient.x);
98 
99  // apply possible rotation
100  angle -= rotation;
101 
102  // make sure angle is within [0, 2*M_PI)
103  angle = fmod(angle, 2 * M_PI);
104  if (angle < 0) angle += 2 * M_PI;
105 
106  // calculate antennaGain
107  size_t baseElement = angle / distance;
108  double offset = (angle - (baseElement * distance)) / distance;
109 
110  // make sure to not address an element out of antennaGains (baseElement == lastElement implies that offset is zero)
111  ASSERT((baseElement < antennaGains.size()) && (baseElement != antennaGains.size() - 1 || offset == 0));
112 
113  double gainValue = antennaGains[baseElement];
114  if (offset > 0) {
115  gainValue += offset * (antennaGains[baseElement + 1] - antennaGains[baseElement]);
116  }
117 
118  return FWMath::dBm2mW(gainValue);
119 }
