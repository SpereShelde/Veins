53 void TraCIDemo11p::onWSM(BaseFrame1609_4* frame)
54 {
55  TraCIDemo11pMessage* wsm = check_and_cast<TraCIDemo11pMessage*>(frame);
56 
57  findHost()->getDisplayString().setTagArg("i", 1, "green");
58 
59  if (mobility->getRoadId()[0] != ':') traciVehicle->changeRoute(wsm->getDemoData(), 9999);
60  if (!sentMessage) {
61  sentMessage = true;
62  // repeat the received traffic update once in 2 seconds plus some random delay
63  wsm->setSenderAddress(myId);

69 void TraCIDemo11p::handleSelfMsg(cMessage* msg)
70 {
71  if (TraCIDemo11pMessage* wsm = dynamic_cast<TraCIDemo11pMessage*>(msg)) {
72  // send this message on the service channel until the counter is 3 or higher.
73  // this code only runs when channel switching is enabled
74  sendDown(wsm->dup());
75  wsm->setSerial(wsm->getSerial() + 1);
76  if (wsm->getSerial() >= 3) {
77  // stop service advertisements
78  stopService();
79  delete (wsm);
80  }
81  else {
82  scheduleAt(simTime() + 1, wsm);
83  }
84  }
85  else {
86  DemoBaseApplLayer::handleSelfMsg(msg);
87  }
88 }

94  // stopped for for at least 10s?

97  findHost()->getDisplayString().setTagArg("i", 1, "red");

103

107  // started service and server advertising, schedule message to self to send later

109  }
110  else {

112  sendDown(wsm);
