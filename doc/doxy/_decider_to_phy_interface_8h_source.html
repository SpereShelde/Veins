43 /**
44  * @brief This class is the interface for a Decider to the BasePhyLayer.
45  *
46  * The Decider can do the following things using it:
47  *
48  * - get the current simulation time
49  * - get the list of AirFrames that intersect with a specific time interval (to
50  * calculate SNR)
51  * - tell the BasePhyLayer to hand an AirFrame up to the MACLayer
52  * - tell the BasePhyLayer to send a control message to the MACLayer
53  *
54  * @ingroup decider
55  */
56 class VEINS_API DeciderToPhyInterface {
57 public:
58  /**
59  * @brief Type for container of AirFrames.
60  *
61  * Used as out-value in "getChannelInfo" method.
62  */
63  using AirFrameVector = std::list<AirFrame*>;
64 
65  virtual ~DeciderToPhyInterface()
66  {
67  }
68 
69  /**
70  * @brief Fills the passed AirFrameVector with all AirFrames that intersect
71  * with the time interval [from, to]
72  */
73  virtual void getChannelInfo(simtime_t_cref from, simtime_t_cref to, AirFrameVector& out) = 0;
74 
75  /**
76  * @brief Returns a constant which defines the noise floor in
77  * the passed time frame (in mW).
78  */
79  virtual double getNoiseFloorValue() = 0;
80 
81  /**
82  * @brief Called by the Decider to send a control message to the MACLayer
83  */
84  virtual void sendControlMsgToMac(cMessage* msg) = 0;
85 
86  /**
87  * @brief Called to send an AirFrame with DeciderResult to the MACLayer
88  *
89  * When a packet is completely received and not noise, the Decider
90  * call this function to send the packet together with
91  * the corresponding DeciderResult up to MACLayer
92  *
93  */
94  virtual void sendUp(AirFrame* packet, DeciderResult* result) = 0;
95 
96  /**
97  * @brief Returns a pointer to the simulations world-utility-module.
98  */
99  virtual BaseWorldUtility* getWorldUtility() = 0;
100 
101  /**
102  * @brief Utility method to enable a Decider, which isn't an OMNeT-module, to
103  * use the OMNeT-method 'recordScalar' with the help of and through its interface to BasePhyLayer.
104  *
105  * The method-signature is taken from OMNeTs 'ccomponent.h' but made pure virtual here.
106  * See the original method-description below:
107  *
108  * Records a double into the scalar result file.
109  */
110  virtual void recordScalar(const char* name, double value, const char* unit = nullptr) = 0;
111 
112  /** @brief Returns the channel currently used by the radio. */
113  virtual int getCurrentRadioChannel() = 0;
114 };
