41 /*
42  * Bitset template for scoped (class) enums.
43  *
44  * Wrapper around the original std::bitset to use a user-defined scoped
45  * enum as the type for the bitset.
46  * Allows implementation of typesafe sets.
47  */
48 template <typename T>
49 class VEINS_API EnumBitset {
50 private:
51  using EnumUnderlyingType = typename std::underlying_type<T>::type;
52  std::bitset<static_cast<EnumUnderlyingType>(EnumTraits<T>::max) + 1> bits;
53 
54  EnumUnderlyingType get_value(T v) const
55  {
56  return static_cast<EnumUnderlyingType>(v);
57  }
58 
59 public:
60  EnumBitset()
61  : bits()
62  {
63  }
64 
65  explicit EnumBitset(EnumUnderlyingType val)
66  : bits(val)
67  {
68  }
69 
70  EnumBitset(T val)
71  : bits()
72  {
73  set(std::move(val));
74  }
75 
76  EnumBitset(std::initializer_list<T> vals)
77  : bits()
78  {
79  for (auto val : vals) {
80  set(val);
81  }
82  }
83 
84  bool operator==(const EnumBitset<T>& rhs) const
85  {
86  return bits == rhs.bits;
87  }
88
