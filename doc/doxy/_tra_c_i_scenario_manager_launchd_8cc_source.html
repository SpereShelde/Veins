51 void TraCIScenarioManagerLaunchd::initialize(int stage)
52 {
53  if (stage != 1) {
54  TraCIScenarioManager::initialize(stage);
55  return;
56  }
57  launchConfig = par("launchConfig").xmlValue();
58  seed = par("seed");
59  cXMLElementList basedir_nodes = launchConfig->getElementsByTagName("basedir");
60  if (basedir_nodes.size() == 0) {
61  // default basedir is where current network file was loaded from
62  std::string basedir = cSimulation::getActiveSimulation()->getEnvir()->getConfig()->getConfigEntry("network").getBaseDirectory();
63  cXMLElement* basedir_node = new cXMLElement("basedir", __FILE__, launchConfig);
64  basedir_node->setAttribute("path", basedir.c_str());
65  launchConfig->appendChild(basedir_node);
66  }
67  cXMLElementList seed_nodes = launchConfig->getElementsByTagName("seed");
68  if (seed_nodes.size() == 0) {
69  if (seed == -1) {
70  // default seed is current repetition
71  const char* seed_s = cSimulation::getActiveSimulation()->getEnvir()->getConfigEx()->getVariable(CFGVAR_RUNNUMBER);
72  seed = atoi(seed_s);
73  }
74  std::stringstream ss;
75  ss << seed;
76  cXMLElement* seed_node = new cXMLElement("seed", __FILE__, launchConfig);
77  seed_node->setAttribute("value", ss.str().c_str());
78  launchConfig->appendChild(seed_node);
79  }
80  TraCIScenarioManager::initialize(stage);
81 }

88 void TraCIScenarioManagerLaunchd::init_traci()
89 {
90  {
91  std::pair<uint32_t, std::string> version = getCommandInterface()->getVersion();
92  uint32_t apiVersion = version.first;
93  std::string serverVersion = version.second;
94 
95  if (apiVersion == 1) {
96  EV_DEBUG << "TraCI server \"" << serverVersion << "\" reports API version " << apiVersion << endl;
97  }
98  else {
99  throw cRuntimeError("TraCI server \"%s\" reports API version %d, which is unsupported. We recommend using the version of sumo-launchd that ships with Veins.", serverVersion.c_str(), apiVersion);
100  }
101  }
102 
103 #if OMNETPP_VERSION <= 0x500
104  std::string contents = launchConfig->tostr(0);
105 #else
106  std::string contents = launchConfig->getXML();
107 #endif
108 
109  TraCIBuffer buf;
110  buf << std::string("sumo-launchd.launch.xml") << contents;
111  connection->sendMessage(makeTraCICommand(CMD_FILE_SEND, buf));
112 
113  TraCIBuffer obuf(connection->receiveMessage());
114  uint8_t cmdLength;
115  obuf >> cmdLength;
116  uint8_t commandResp;
117  obuf >> commandResp;
118  if (commandResp != CMD_FILE_SEND) throw cRuntimeError("Expected response to command %d, but got one for command %d", CMD_FILE_SEND, commandResp);
119  uint8_t result;
120  obuf >> result;
121  std::string description;
122  obuf >> description;
123  if (result != RTYPE_OK) {
124  EV << "Warning: Received non-OK response from TraCI server to command " << CMD_FILE_SEND << ":" << description.c_str() << std::endl;
125  }
126 
127  TraCIScenarioManager::init_traci();
128 }
