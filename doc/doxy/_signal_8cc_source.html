180 bool Signal::greaterAtCenterFrequency(double threshold)
181 {
182  if (values[centerFrequencyIndex] < threshold) return false;
183 
184  uint16_t maxAnalogueModels = analogueModelList->size();
185 
186  while (numAnalogueModelsApplied < maxAnalogueModels) {
187  // Apply filter here
188  (*analogueModelList)[numAnalogueModelsApplied]->filterSignal(this);
189  numAnalogueModelsApplied++;
190 
191  if (values[centerFrequencyIndex] < threshold) return false;
192  }
193  return true;
194 }
