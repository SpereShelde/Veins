64 /**
65  * The basic handle message function.
66  *
67  * Depending on the gate a message arrives handleMessage just calls
68  * different handle*Msg functions to further process the message.
69  *
70  * You should not make any changes in this function but implement all
71  * your functionality into the handle*Msg functions called from here.
72  *
73  * @sa handleUpperMsg, handleLowerMsg, handleSelfMsg
74  **/
75 void BaseLayer::handleMessage(cMessage* msg)
76 {
77  if (msg->isSelfMessage()) {
78  handleSelfMsg(msg);
79  }
80  else if (msg->getArrivalGateId() == upperLayerIn) {
81  recordPacket(PassedMessage::INCOMING, PassedMessage::UPPER_DATA, msg);
82  handleUpperMsg(msg);
83  }
84  else if (msg->getArrivalGateId() == upperControlIn) {
85  recordPacket(PassedMessage::INCOMING, PassedMessage::UPPER_CONTROL, msg);
86  handleUpperControl(msg);
87  }
88  else if (msg->getArrivalGateId() == lowerControlIn) {
89  recordPacket(PassedMessage::INCOMING, PassedMessage::LOWER_CONTROL, msg);
90  handleLowerControl(msg);
91  }
92  else if (msg->getArrivalGateId() == lowerLayerIn) {
93  recordPacket(PassedMessage::INCOMING, PassedMessage::LOWER_DATA, msg);
94  handleLowerMsg(msg);
95  }
96  else if (msg->getArrivalGateId() == -1) {
97  /* Classes extending this class may not use all the gates, f.e.
98  * BaseApplLayer has no upper gates. In this case all upper gate-
99  * handles are initialized to -1. When getArrivalGateId() equals -1,
100  * it would be wrong to forward the message to one of these gates,
101  * as they actually don't exist, so raise an error instead.
102  */
103  throw cRuntimeError("No self message and no gateID?? Check configuration.");
104  }
105  else {
106  /* msg->getArrivalGateId() should be valid, but it isn't recognized
107  * here. This could signal the case that this class is extended
108  * with extra gates, but handleMessage() isn't overridden to
109  * check for the new gate(s).
110  */
111  throw cRuntimeError("Unknown gateID?? Check configuration or override handleMessage().");
112  }
113 }

149 void BaseLayer::recordPacket(PassedMessage::direction_t dir, PassedMessage::gates_t gate, const cMessage* msg)
150 {
151  if (passedMsg == nullptr) return;
152  passedMsg->direction = dir;
153  passedMsg->gateType = gate;
154  passedMsg->kind = msg->getKind();
155  passedMsg->name = msg->getName();
156  emit(catPassedMsgSignal, passedMsg);
