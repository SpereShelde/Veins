32 /**
33  * Abstraction for (recurring) Timers for cSimpleModule.
34  *
35  * This abstraction takes care of managing the required self-messages to (repeatedly) execute a piece of code after a certain time.
36  * To use, instantiate one TimerManager per cSimpleModule, then call its handleMessage method from that of the cSimpleModule.
37  *
38  * In order to schedule a timer, create a TimerSpecification object using the corresponding methods.
39  * After configuration, use the create function from the TimerManager to actually schedule the configured timer.
40  */
41 class TimerManager;
42

44  * A message which is used for triggering Timers.
45  *
46  * Its implementation is empty as it is only used to differentiate from other
47  * messages.
48  */

51 /**
52  * A class which specifies a Timer.
53  *
54  * This includes timing information as well as its callback.
55  */
56 struct VEINS_API TimerSpecification {
57 public:
58  /**
59  * Create a new TimerSpecification.
60  *
61  * The created timer is invalid, an interval is missing for it to be usable.
62  * By default, the timer starts running immediately and triggers first after the first time after the interval.
63  * After that, it will continue to run until the simulation ends, calling the callback after the interval has elapsed.
64  * In order to create a timer, this needs to be passed to TimerManager::create.
65  *
66  * @param callback The callback which is executed when the timer is triggered.
67  *
68  * @see TimerManager
69  */
70  TimerSpecification(std::function<void()> callback);
71 
72  /**
73  * Set the period between two timer occurences.
74  */
75  TimerSpecification& interval(omnetpp::simtime_t interval);
76 
77  /**
78  * Set the number of repetitions.
79  *
80  * @note You cannot use both this and absoluteEnd or relativeEnd.
81  */
82  TimerSpecification& repetitions(size_t n);
83 
84  /**
85  * Set the timer's start time.
86  *
87  * Any previously set start time will be overwritten.
88  *
89  * @param start Time of first execution relative to the current simulation time. E.g., passing simtime_t(1, SIMTIME_S) will execute the timer in one second.
90  *
91  * @note You cannot use this in conjunction with repetition().
92  */
93  TimerSpecification& relativeStart(omnetpp::simtime_t start);
94 
95  /**
96  * Set the timer's start time.
97  *
98  * Any previously set start time will be overwritten.
99  *
100  * @param start The absolute start time. The first occurence will be exactly at this time. Passing a value earlier than the current simtime will result in an error.
101  *
102  * @note You cannot use this in conjunction with repetition().
103  */
104  TimerSpecification& absoluteStart(omnetpp::simtime_t start);
105 
106  /**
107  * Set the timer's end time.
108  *
109  * Any previously set end time will be overwritten.
110  *
111  * @param end Time after which this timer will no longer be executed, relative to the current simulation time. E.g., passing simtime_t(1, SIMTIME_S) will stop the execution of the time after one second has passed.
112  */
113  TimerSpecification& relativeEnd(omnetpp::simtime_t end);
114 
115  /**
116  * Set the timer's end time.
117  *
118  * Any previously set end time will be overwritten.
119  *
120  * @param end The absolute end time. The latest possible occurence is at this time. Values before the current start time will prevent any executions.
121  */
122  TimerSpecification& absoluteEnd(omnetpp::simtime_t end);
123 
124  /**
125  * Set the timer to be open ended.
126  *
127  * Any previously set end time will be overwritten.
128  */
129  TimerSpecification& openEnd();
130 
131  /**
132  * Set the timer to execute once in a given time.
133  *
134  * Any previously set start time, end time, and interval will be overwritten.
135  */
136  TimerSpecification& oneshotIn(omnetpp::simtime_t in);
137 
138  /**
139  * Set the timer to execute once at a given time.
140  *
141  * Any previously set start time, end time, and interval will be overwritten.
142  */
143  TimerSpecification& oneshotAt(omnetpp::simtime_t at);
144 
145 private:
146  friend TimerManager;

149  relative,
150  absolute,
151  immediate
152  };
153  enum class EndMode {
154  relative,
155  absolute,

160  /**
161  * Finalizes this instance such that its values are independent of current simulation time.
162  *
163  * After calling this function, start_mode_ is guaranteed to be StartMode::absolute and end_mode_ to be EndMode::absolute or EndMode::open.
164  */
165  void finalize();
166 
167  /**
168  * Checks validity of this specification, i.e., whether all necessary information is set.
169  */
170  bool valid() const

172  return period_ != -1;
173  }
174 
175  /**
176  * Check that the given time is a valid occurence for this timer.
177  */
178  bool validOccurence(omnetpp::simtime_t time) const;
179 
180  StartMode start_mode_; ///< Interpretation of start time._
181  omnetpp::simtime_t start_; ///< Time of the Timer's first occurence. Interpretation depends on start_mode_.
182  EndMode end_mode_; ///< Interpretation of end time._
183  unsigned end_count_; ///< Number of repetitions of the timer. Only valid when end_mode_ == repetition.
184  omnetpp::simtime_t end_time_; ///< Last possible occurence of the timer. Only valid when end_mode_ != repetition.
185  omnetpp::simtime_t period_; ///< Time between events.
186  std::function<void()> callback_; ///< The function to be called when the Timer is triggered.
187 };
