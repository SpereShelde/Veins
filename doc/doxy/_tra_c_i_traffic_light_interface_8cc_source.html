128 void TraCITrafficLightInterface::setCurrentLogicById(const std::string& logicId, bool setSumo)
129 {
130  if (setSumo) {
131  ASSERT(logicId != "online");
132  if (!programDefinition.hasLogic(logicId)) {
133  throw cRuntimeError("Logic '%s' not found in program of TraCITrafficLightInterface %s", logicId.c_str(), external_id.c_str());
134  }
135  tlCommandInterface->setProgram(logicId);
136  const std::string newValueInSumo = tlCommandInterface->getCurrentProgramID();
137  ASSERT(newValueInSumo == logicId);
138  }
139  if (currentLogicId != logicId || (isInOnlineSignalState() && logicId != "online")) {
140  sendChangeMsg(TrafficLightAtrributeType::LOGICID, logicId, currentLogicId);
141  }
142  if (logicId != "online") {
143  inOnlineSignalState = false;
144  this->currentLogicId = logicId;
145  }
146 }

148 void TraCITrafficLightInterface::setCurrentPhaseByNr(const unsigned int phaseNr, bool setSumo)
149 {
150  if (setSumo) {
151  if (phaseNr >= getCurrentLogic().phases.size()) {
152  throw cRuntimeError("Cannot set current phase to %d: current logic has only %d Phases (TraCITrafficLightInterface %s)", phaseNr, getCurrentLogic().phases.size(), external_id.c_str());
153  }
154  tlCommandInterface->setPhaseIndex(phaseNr);
155  const unsigned int newValueInSumo = tlCommandInterface->getCurrentPhaseIndex();
156  ASSERT(newValueInSumo == phaseNr);
157  }
158  if (currentPhaseNr != static_cast<int>(phaseNr) || isInOnlineSignalState()) {
159  sendChangeMsg(TrafficLightAtrributeType::PHASEID, std::to_string(phaseNr), std::to_string(currentPhaseNr));
160  }
161  inOnlineSignalState = false;
162  currentPhaseNr = phaseNr;
163 }

179 void TraCITrafficLightInterface::setNextSwitch(const simtime_t& newNextSwitch, bool setSumo)
180 {
181  // FIXME: not working reliably - use setRemainingDuration instead!
182  // round to be feasible for SUMO
183  simtime_t nextSwitch = ceil(newNextSwitch, updateInterval, 0);
184  if (setSumo) {
185  simtime_t remainingDuration = ceil(nextSwitch - simTime(), updateInterval, 0);
186  if (remainingDuration < 0) {
187  EV << "Warning: remaining duration for switch below 0: " << remainingDuration << std::endl;
188  // maybe issue even an error if this occurs
189  remainingDuration = 0;
190  }
191  getTlCommandInterface()->setPhaseDuration(remainingDuration);
192  simtime_t newValueInSumo = tlCommandInterface->getAssumedNextSwitchTime();
193  ASSERT(newValueInSumo == nextSwitch);
194  }
195  if (nextSwitchTime != nextSwitch) {
196  sendChangeMsg(TrafficLightAtrributeType::SWITCHTIME, std::to_string(nextSwitch.inUnit(SIMTIME_MS)), std::to_string(nextSwitchTime.inUnit(SIMTIME_MS)));
197  }
198  nextSwitchTime = nextSwitch;
199 }

201 void TraCITrafficLightInterface::setRemainingDuration(const simtime_t& rawRemainingDuration, bool setSumo)
202 {
203  ASSERT(rawRemainingDuration >= 0);
204  // round (up) to match sumo time steps
205  simtime_t veinsTimeNow(simTime());
206  simtime_t sumoTimeNow(ceil(veinsTimeNow, updateInterval) - updateInterval);
207  simtime_t roundedRemainingDuration = ceil(rawRemainingDuration, updateInterval, 0);
208  // simtime_t nextSwitchInVeins = floor(simTime() + roundedRemainingDuration, updateInterval, 0) - updateInterval;
209  simtime_t nextSwitchInVeins = sumoTimeNow + roundedRemainingDuration;
210  if (setSumo) {
211  // set value to sumo
212  getTlCommandInterface()->setPhaseDuration(roundedRemainingDuration);
213  // check that everything is consistent
214  simtime_t nextSwitchInSumo = tlCommandInterface->getAssumedNextSwitchTime();
215  ASSERT(nextSwitchInSumo == nextSwitchInVeins);
216  }
217  if (nextSwitchTime != nextSwitchInVeins) {
218  sendChangeMsg(TrafficLightAtrributeType::SWITCHTIME, std::to_string(nextSwitchInVeins.inUnit(SIMTIME_MS)), std::to_string(nextSwitchTime.inUnit(SIMTIME_MS)));
219  }
220  nextSwitchTime = nextSwitchInVeins;
221 }

246 void TraCITrafficLightInterface::handleChangeCommandMessage(cMessage* msg)
247 {
248  TraCITrafficLightMessage* tmMsg = check_and_cast<TraCITrafficLightMessage*>(msg);
249  switch (tmMsg->getChangedAttribute()) {
250  case TrafficLightAtrributeType::LOGICID:
251  setCurrentLogicById(tmMsg->getNewValue(), true);
252  break;
253 
254  case TrafficLightAtrributeType::PHASEID:
255  setCurrentPhaseByNr(std::stoi(tmMsg->getNewValue()), true);
256  break;
257 
258  case TrafficLightAtrributeType::SWITCHTIME:
259  setNextSwitch(SimTime(std::stoi(tmMsg->getNewValue()), SIMTIME_MS), true);
260  break;
261 
262  case TrafficLightAtrributeType::STATE:
263  setCurrentState(tmMsg->getNewValue(), true);
264  break;
265  }
266 }
