83 void ChannelAccess::sendToChannel(cPacket* msg)
84 {
85  const NicEntry::GateList& gateList = cc->getGateList(getParentModule()->getId());
86  NicEntry::GateList::const_iterator i = gateList.begin();
87 
88  if (useSendDirect) {
89  // use Andras stuff
90  if (i != gateList.end()) {
91  simtime_t delay = SIMTIME_ZERO;
92  for (; i != --gateList.end(); ++i) {
93  // calculate delay (Propagation) to this receiving nic
94  delay = calculatePropagationDelay(i->first);
95 
96  int radioStart = i->second->getId();
97  int radioEnd = radioStart + i->second->size();
98  for (int g = radioStart; g != radioEnd; ++g) sendDirect(static_cast<cPacket*>(msg->dup()), delay, msg->getDuration(), i->second->getOwnerModule(), g);
99  }
100  // calculate delay (Propagation) to this receiving nic
101  delay = calculatePropagationDelay(i->first);
102 
103  int radioStart = i->second->getId();
104  int radioEnd = radioStart + i->second->size();
105  for (int g = radioStart; g != --radioEnd; ++g) sendDirect(static_cast<cPacket*>(msg->dup()), delay, msg->getDuration(), i->second->getOwnerModule(), g);
106 
107  sendDirect(msg, delay, msg->getDuration(), i->second->getOwnerModule(), radioEnd);
108  }
109  else {
110  EV_WARN << "Nic is not connected to any gates!" << endl;
111  delete msg;
112  }
113  }
114  else {
115  // use our stuff
116  EV_TRACE << "sendToChannel: sending to gates\n";
117  if (i != gateList.end()) {
118  simtime_t delay = SIMTIME_ZERO;
119  for (; i != --gateList.end(); ++i) {
120  // calculate delay (Propagation) to this receiving nic
121  delay = calculatePropagationDelay(i->first);
122 
123  sendDelayed(static_cast<cPacket*>(msg->dup()), delay, i->second);
124  }
125  // calculate delay (Propagation) to this receiving nic
126  delay = calculatePropagationDelay(i->first);
127 
128  sendDelayed(msg, delay, i->second);
129  }
130  else {
131  EV_WARN << "Nic is not connected to any gates!" << endl;
132  delete msg;
133  }
134  }
135 }

137 simtime_t ChannelAccess::calculatePropagationDelay(const NicEntry* nic)
138 {
139  if (!usePropagationDelay) return 0;
140 
141  ChannelAccess* const senderModule = this;
142  ChannelAccess* const receiverModule = nic->chAccess;
143  // const simtime_t_cref sStart = simTime();
144 
145  ASSERT(senderModule);
146  ASSERT(receiverModule);
147 
148  /** claim the Move pattern of the sender from the Signal */
149  Coord senderPos = senderModule->antennaPosition.getPositionAt();
150  Coord receiverPos = receiverModule->antennaPosition.getPositionAt();
151 
152  // this time-point is used to calculate the distance between sending and receiving host
153  return receiverPos.distance(senderPos) / BaseWorldUtility::speedOfLight();
154 }

156 void ChannelAccess::receiveSignal(cComponent* source, simsignal_t signalID, cObject* obj, cObject* details)
157 {
158  if (signalID == BaseMobility::mobilityStateChangedSignal) {
159  ChannelMobilityPtrType const mobility = check_and_cast<ChannelMobilityPtrType>(obj);
160 
161  auto heading = Heading::fromCoord(mobility->getCurrentOrientation());
162  antennaPosition = AntennaPosition(getId(), mobility->getPositionAt(simTime()) + antennaOffset.rotatedYaw(-heading.getRad()), mobility->getCurrentSpeed(), simTime());
163  antennaHeading = Heading(heading.getRad() + antennaOffsetYaw);
164 
165  if (isRegistered) {
166  cc->updateNicPos(getParentModule()->getId(), antennaPosition.getPositionAt(), antennaHeading);
167  }
168  else {
169  // register the nic with ConnectionManager
170  // returns true, if sendDirect is used
171  useSendDirect = cc->registerNic(getParentModule(), this, antennaPosition.getPositionAt(), antennaHeading);
172  isRegistered = true;
173  }
174  }
175 }
