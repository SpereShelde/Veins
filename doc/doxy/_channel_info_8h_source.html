37 /**
38  * @brief This class is used by the BasePhyLayer to keep track of the AirFrames
39  * on the channel.
40  *
41  * ChannelInfo is able to return every AirFrame which intersects with a
42  * specified interval. This is mainly used to get the noise for a received
43  * signal.
44  *
45  * ChannelInfo is a passive class meaning the user has to tell it when a new
46  * AirFrame starts and an existing ends.
47  *
48  * Once an AirFrame has been added to the ChannelInfo the ChannelInfo holds the
49  * ownership of this AirFrame even if the AirFrame is removed again from the
50  * ChannelInfo. This is necessary because the ChannelInfo has to be able to
51  * store also the AirFrames which are over but still intersect with an currently
52  * running AirFrame.
53  *
54  * Note: ChannelInfo assumes that the AirFrames are added and removed
55  * chronologically. This means every time you add an AirFrame with a
56  * specific start time ChannelInfo assumes that start time as the current
57  * time and assumes that every following action happens after that moment.
58  * The same goes for "removeAirFrame". When removing an AirFrames,
59  * ChannelInfo assumes the start time plus the duration of the AirFrame as
60  * the current time.
61  * This also affects "getAirFrames" in the way that you may only ask for
62  * intervals which lie before the "current time" of ChannelInfo.
63  *
64  * @ingroup phyLayer
65  */
66 class VEINS_API ChannelInfo {

79  /**
80  * @brief Iterator for every intersection of a specific interval in a
81  * AirFrameMatrix.
82  *
83  * A time interval A_start to A_end intersects with another interval B_start
84  * to B_end iff the following two conditions are fulfilled:
85  *
86  * 1. A_end >= B_start.
87  * 2. A_start <= B_end and
88  *
89  * Or the defined by the
90  * opposite: The two intervals do not intersect iff A_end < B_start or
91  * A_start > B_end.
92  *
93  * To iterate over a two dimensional (end-time x start-time) container
94  * of AirFrames this class iterates over all AirFrame-lists whose end-time
95  * is bigger or equal than the interval start (intersect condition 1) and in
96  * each of these lists iterates over all entries but only stops at and
97  * returns entries which start-time is smaller or equal than the interval
98  * end (intersection condition 2).
99  *
100  * In template form to work as const- and non-const iterator.
101  */
102  template <class C, class ItMatrix, class ItList>
103  class VEINS_API BaseIntersectionIterator {
104  public:
105  /** @brief Pointer to the matrix holding the intervals.*/
106  C* intervals;
107 
108  /** @brief Point in time to start iterating over intersections.*/
109  simtime_t from;
110 
111  /** @brief Point in time to end iterating over intersections.*/
112  simtime_t to;
113 
114  /** @brief Iterator over AirFrame end times.*/
115  ItMatrix endIt;
116 
117  /** @brief Iterator over AirFrame start times.*/
118  ItList startIt;
119 
120  /** @brief True if the we are already pointing to the next entry.*/
121  bool alreadyNext;
122 
123  public:
124  /**
125  * @brief Creates an iterator for the specified interval at the
126  * specified AirFrameMatrix.
127  */
128  BaseIntersectionIterator(C* airFrames, simtime_t_cref from, simtime_t_cref to)
129  : intervals(airFrames)
130  , from(from)
131  , to(to)
132  {
133  // begin at the smallest end-time-entry fulfilling the intersection
134  // condition 1
135  endIt = intervals->lower_bound(from);
136 
137  if (endIt != intervals->end()) {
138  startIt = endIt->second.begin();
139  }
140  // we are already pointing at the first unchecked interval
141  alreadyNext = true;
142  }
143 
144  /**
145  * @brief Increases the iterator to the next intersecting AirFrame and
146  * returns a pointer to this AirFrame.
147  */
148  AirFrame* next()
149  {
150  if (endIt == intervals->end()) return nullptr;
151 
152  // "alreadyNext" indicates that some previous iterator function has
153  // already increased the intern iterators to a yet unchecked values.
154  // This happens after initialization of the iterator and when an
155  // element is erased.
156  if (alreadyNext)
157  alreadyNext = false;
158  else
159  startIt++;
160 
161  while (endIt != intervals->end()) {
162  // while there are entries left at the current end-time
163  while (startIt != endIt->second.end()) {
164  // check if this entry fulfilles the intersection condition
165  // 2 (condition 1 is already fulfilled in the constructor by
166  // the start-value of the end-time-iterator)
167  if (startIt->first <= to) {
168  return startIt->second;
169  }
170  startIt++;
171  }
172 
173  endIt++;
174  if (endIt == intervals->end()) return nullptr;
175 
176  startIt = endIt->second.begin();
177  }
178 
179  return nullptr;
180  }
181  };

194  typedef BaseIntersectionIterator<AirFrameMatrix, AirFrameMatrix::iterator, AirFrameTimeList::iterator> Base;

198  * @brief Creates an iterator for the specified interval at the
199  * specified AirFrameMatrix.
200  */
201  IntersectionIterator(AirFrameMatrix* airFrames, simtime_t_cref from, simtime_t_cref to)
202  : Base(airFrames, from, to)
203  {
204  }

206  /**
207  * @brief Erases the AirFrame the iterator currently points to from the
208  * AirFrameMatrix.
209  *
210  * After the erase the iterator points to an invalid value and "next()"
211  * should be called.
212  */
213  void eraseAirFrame()
214  {
215  ASSERT(endIt != intervals->end());
216  ASSERT(startIt != endIt->second.end());
217 
218  // erase AirFrame from list
219  startIt = endIt->second.erase(startIt);
220 
221  // check if we've deleted the last entry in the list
222  if (startIt == endIt->second.end()) {
223  // check if we deleted the only entry in the list
224  if (endIt->second.empty()) {
225  intervals->erase(endIt++); // delete list from map
226  }
227  else {
228  endIt++;
229  }
230 
231  // increase to a valid value if we are not done
232  if (endIt != intervals->end()) {
233  startIt = endIt->second.begin();
234  }
235  }
236  alreadyNext = true;
237  }
238  };

415  /**
416  * @brief Tells ChannelInfo to keep from now on all channel information
417  * until the passed time in history.
418  *
419  * The passed start-time should be the current simulation time, otherwise
420  * ChannelInfo can't assure that it hasn't already thrown away some
421  * information for that passed time period.
422  *
423  * Subsequent calls to this method will update the recording start time and
424  * information from old start times is thrown away.
425  *
426  * @param start The point in time from which to keep all channel information
427  * stored.
428  */
429  void startRecording(simtime_t_cref start)
430  {
431  // clean up until old record start
432  if (recordStartTime > -1) {
433  recordStartTime = start;
434  checkAndCleanInterval(0, recordStartTime);
435  }
436  else {
437  recordStartTime = start;
438  }
439  }
440 
441  /**
442  * @brief Tells ChannelInfo to stop recording Information from now on.
443  *
444  * Its up to ChannelInfo to decide when to actually throw away the
445  * information it doesn't need to store anymore now.
446  */
447  void stopRecording()
448  {
449  if (recordStartTime > -1) {
450  simtime_t old = recordStartTime;
451  recordStartTime = -1;
452  checkAndCleanFrom(old);
453  }
454  }
455 
456  /**
457  * @brief Returns true if ChannelInfo is currently recording.
458  * @return true if ChannelInfo is recording
459  */
460  bool isRecording() const
461  {
462  return recordStartTime > -1;
463  }
464 
465  /**
466  * @brief Returns true if there are currently no active or inactive
467  * AirFrames on the channel.
468  */
469  bool isChannelEmpty() const
470  {
471  ASSERT(recordStartTime != -1 || activeAirFrames.empty() == airFrameStarts.empty());
472 
473  return airFrameStarts.empty();
474  }
