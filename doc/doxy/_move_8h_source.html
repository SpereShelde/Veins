160  /**
161  * @brief Sets the direction to the passed vector,
162  * which must be already normalized or the 0-vector.
163  */
164  void setDirectionByVector(const Coord& direction)
165  {
166  ASSERT(math::almost_equal(direction.squareLength(), 1.0) || math::almost_equal(direction.squareLength(), 0.0));
167  this->direction = direction;
168  }
169 
170  /**
171  * @brief Sets the direction to the normalized vector pointing
172  * from the current start position to the passed target point.
173  *
174  * NOTE: The target point must not be the current start position
175  * or too close to it.
176  */
177  void setDirectionByTarget(const Coord& target)
178  {
179  direction = target - startPos;
180 
181  ASSERT(!math::almost_equal(direction.length(), 0.0));
182  direction /= direction.length();
183  }
184 
185  /**
186  * @brief Returns the position of the Move (Host) at the specified point in time.
187  * It is intended to be passed simTime() as actualTime and returns the actual position.
188  *
189  * Assumes that direction represents a normalized vector (length equals 1.0).
190  * Further this function does not check whether the given time point is before
191  * the startTime of the actual move pattern. So in this case one might obtain
192  * an unintended result.
193  *
194  */
195  virtual Coord getPositionAt(simtime_t_cref actualTime = simTime()) const
196  {
197  // if speed is very close to 0.0, the host is practically standing still
198  if (math::almost_equal(speed, 0.0)) return startPos;
199 
200  // otherwise: actualPos = startPos + ( direction * v * t )
201  return startPos + (direction * speed * SIMTIME_DBL(actualTime - startTime));
202  }
203  virtual const Coord& getStartPosition() const
204  {
205  if (lastPos.z != DBL_MAX) return lastPos;
206  return startPos;
207  }
208 
209 public:
210  /**

212  */
213  std::string info() const override

216  ost << " HostMove "
217  << " startPos: " << startPos.info() << " direction: " << direction.info() << " orientation: " << orientation.info() << " startTime: " << startTime << " speed: " << speed;
218  return ost.str();
219  }
