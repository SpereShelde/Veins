42 simtime_t Decider80211p::processNewSignal(AirFrame* msg)
43 {
44 
45  AirFrame11p* frame = check_and_cast<AirFrame11p*>(msg);
46 
47  // get the receiving power of the Signal at start-time and center frequency
48  Signal& signal = frame->getSignal();
49 
50  signalStates[frame] = EXPECT_END;
51 
52  if (signal.smallerAtCenterFrequency(minPowerLevel)) {
53 
54  // annotate the frame, so that we won't try decoding it at its end
55  frame->setUnderMinPowerLevel(true);
56  // check channel busy status. a superposition of low power frames might turn channel status to busy
57  if (cca(simTime(), nullptr) == false) {
58  setChannelIdleStatus(false);
59  }
60  return signal.getReceptionEnd();
61  }
62  else {
63 
64  // This value might be just an intermediate result (due to short circuiting)
65  double recvPower = signal.getAtCenterFrequency();
66  setChannelIdleStatus(false);
67 
68  if (phy11p->getRadioState() == Radio::TX) {
69  frame->setBitError(true);
70  frame->setWasTransmitting(true);
71  EV_TRACE << "AirFrame: " << frame->getId() << " (" << recvPower << ") received, while already sending. Setting BitErrors to true" << std::endl;
72  }
73  else {
74 
75  if (!currentSignal.first) {
76  // NIC is not yet synced to any frame, so lock and try to decode this frame
77  currentSignal.first = frame;
78  EV_TRACE << "AirFrame: " << frame->getId() << " with (" << recvPower << " > " << minPowerLevel << ") -> Trying to receive AirFrame." << std::endl;
79  if (notifyRxStart) {
80  phy->sendControlMsgToMac(new cMessage("RxStartStatus", MacToPhyInterface::PHY_RX_START));
81  }
82  }
83  else {
84  // NIC is currently trying to decode another frame. this frame will be simply treated as interference
85  EV_TRACE << "AirFrame: " << frame->getId() << " with (" << recvPower << " > " << minPowerLevel << ") -> Already synced to another AirFrame. Treating AirFrame as interference." << std::endl;
86  }
87 
88  // channel turned busy
89  // measure communication density
90  myBusyTime += signal.getDuration().dbl();
91  }
92  return signal.getReceptionEnd();
93  }
94 }

107 DeciderResult* Decider80211p::checkIfSignalOk(AirFrame* frame)
108 {
109  auto frame11p = check_and_cast<AirFrame11p*>(frame);
110 
111  Signal& s = frame->getSignal();
112  simtime_t start = s.getReceptionStart();
113  simtime_t end = s.getReceptionEnd();
114 
115  // compute receive power
116  double recvPower_dBm = 10 * log10(s.getAtCenterFrequency());
117 
118  start = start + PHY_HDR_PREAMBLE_DURATION; // its ok if something in the training phase is broken
119 
120  AirFrameVector airFrames;
121  getChannelInfo(start, end, airFrames);
122 
123  double noise = phy->getNoiseFloorValue();
124 
125  // Make sure to use the adjusted starting-point (which ignores the preamble)
126  double sinrMin = SignalUtils::getMinSINR(start, end, frame, airFrames, noise);
127  double snrMin;
128  if (collectCollisionStats) {
129  // snrMin = SignalUtils::getMinSNR(start, end, frame, noise);
130  snrMin = s.getDataMin() / noise;
131  }
132  else {
133  // just set to any value. if collectCollisionStats != true
134  // it will be ignored by packetOk
135  snrMin = 1e200;
136  }
137 
138  double payloadBitrate = getOfdmDatarate(static_cast<MCS>(frame11p->getMcs()), BANDWIDTH_11P);
139 
140  DeciderResult80211* result = nullptr;
141 
142  switch (packetOk(sinrMin, snrMin, frame->getBitLength(), payloadBitrate)) {
143 
144  case DECODED:
145  EV_TRACE << "Packet is fine! We can decode it" << std::endl;
146  result = new DeciderResult80211(true, payloadBitrate, sinrMin, recvPower_dBm, false);
147  break;
148 
149  case NOT_DECODED:
150  if (!collectCollisionStats) {
151  EV_TRACE << "Packet has bit Errors. Lost " << std::endl;
152  }
153  else {
154  EV_TRACE << "Packet has bit Errors due to low power. Lost " << std::endl;
155  }
156  result = new DeciderResult80211(false, payloadBitrate, sinrMin, recvPower_dBm, false);
157  break;
158 
159  case COLLISION:
160  EV_TRACE << "Packet has bit Errors due to collision. Lost " << std::endl;
161  collisions++;
162  result = new DeciderResult80211(false, payloadBitrate, sinrMin, recvPower_dBm, true);
163  break;
164 
165  default:
166  ASSERT2(false, "Impossible packet result returned by packetOk(). Check the code.");
167  break;
168  }
169 
170  return result;
171 }

173 enum Decider80211p::PACKET_OK_RESULT Decider80211p::packetOk(double sinrMin, double snrMin, int lengthMPDU, double bitrate)
174 {
175  double packetOkSinr;
176  double packetOkSnr;
177 
178  // compute success rate depending on mcs and bw
179  packetOkSinr = NistErrorRate::getChunkSuccessRate(bitrate, BANDWIDTH_11P, sinrMin, PHY_HDR_SERVICE_LENGTH + lengthMPDU + PHY_TAIL_LENGTH);
180 
181  // check if header is broken
182  double headerNoError = NistErrorRate::getChunkSuccessRate(PHY_HDR_BITRATE, BANDWIDTH_11P, sinrMin, PHY_HDR_PLCPSIGNAL_LENGTH);
183 
184  double headerNoErrorSnr;
185  // compute PER also for SNR only
186  if (collectCollisionStats) {
187 
188  packetOkSnr = NistErrorRate::getChunkSuccessRate(bitrate, BANDWIDTH_11P, snrMin, PHY_HDR_SERVICE_LENGTH + lengthMPDU + PHY_TAIL_LENGTH);
189  headerNoErrorSnr = NistErrorRate::getChunkSuccessRate(PHY_HDR_BITRATE, BANDWIDTH_11P, snrMin, PHY_HDR_PLCPSIGNAL_LENGTH);
190 
191  // the probability of correct reception without considering the interference
192  // MUST be greater or equal than when consider it
193  ASSERT(packetOkSnr >= packetOkSinr);
194  ASSERT(headerNoErrorSnr >= headerNoError);
195  }
196 
197  // probability of no bit error in the PLCP header
198 
199  double rand = RNGCONTEXT dblrand();
200 
201  if (!collectCollisionStats) {
202  if (rand > headerNoError) return NOT_DECODED;
203  }
204  else {
205 
206  if (rand > headerNoError) {
207  // ups, we have a header error. is that due to interference?
208  if (rand > headerNoErrorSnr) {
209  // no. we would have not been able to receive that even
210  // without interference
211  return NOT_DECODED;
212  }
213  else {
214  // yes. we would have decoded that without interference
215  return COLLISION;
216  }
217  }
218  }
219 
220  // probability of no bit error in the rest of the packet
221 
222  rand = RNGCONTEXT dblrand();
223 
224  if (!collectCollisionStats) {
225  if (rand > packetOkSinr) {
226  return NOT_DECODED;
227  }
228  else {
229  return DECODED;
230  }
231  }
232  else {
233 
234  if (rand > packetOkSinr) {
235  // ups, we have an error in the payload. is that due to interference?
236  if (rand > packetOkSnr) {
237  // no. we would have not been able to receive that even
238  // without interference
239  return NOT_DECODED;
240  }
241  else {
242  // yes. we would have decoded that without interference
243  return COLLISION;
244  }
245  }
246  else {
247  return DECODED;
248  }
249  }

252 bool Decider80211p::cca(simtime_t_cref time, AirFrame* exclude)
253 {
254 
255  AirFrameVector airFrames;
256 
257  // collect all AirFrames that intersect with [start, end]
258  getChannelInfo(time, time, airFrames);
259 
260  // In the reference implementation only centerFrequenvy - 5e6 (half bandwidth) is checked!
261  // Although this is wrong, the same is done here to reproduce original results
262  double minPower = phy->getNoiseFloorValue();
263  bool isChannelIdle = minPower < ccaThreshold;
264  if (airFrames.size() > 0) {
265  size_t usedFreqIndex = airFrames.front()->getSignal().getSpectrum().indexOf(centerFrequency - 5e6);
266  isChannelIdle = SignalUtils::isChannelPowerBelowThreshold(time, airFrames, usedFreqIndex, ccaThreshold - minPower, exclude);
267  }
268 
269  return isChannelIdle;
270 }

272 simtime_t Decider80211p::processSignalEnd(AirFrame* msg)
273 {
274 
275  AirFrame11p* frame = check_and_cast<AirFrame11p*>(msg);
276 
277  // here the Signal is finally processed
278  Signal& signal = frame->getSignal();
279 
280  double recvPower_dBm = 10 * log10(signal.getMax());
281 
282  bool whileSending = false;
283 
284  // remove this frame from our current signals
285  signalStates.erase(frame);
286 
287  DeciderResult* result;
288 
289  if (frame->getUnderMinPowerLevel()) {
290  // this frame was not even detected by the radio card
291  result = new DeciderResult80211(false, 0, 0, recvPower_dBm);
292  }
293  else if (frame->getWasTransmitting() || phy11p->getRadioState() == Radio::TX) {
294  // this frame was received while sending
295  whileSending = true;
296  result = new DeciderResult80211(false, 0, 0, recvPower_dBm);
297  }
298  else {
299 
300  // first check whether this is the frame NIC is currently synced on
301  if (frame == currentSignal.first) {
302  // check if the snr is above the Decider's specific threshold,
303  // i.e. the Decider has received it correctly
304  result = checkIfSignalOk(frame);
305 
306  // after having tried to decode the frame, the NIC is no more synced to the frame
307  // and it is ready for syncing on a new one
308  currentSignal.first = 0;
309  }
310  else {
311  // if this is not the frame we are synced on, we cannot receive it
312  result = new DeciderResult80211(false, 0, 0, recvPower_dBm);
313  }
314  }
315 
316  if (result->isSignalCorrect()) {
317  EV_TRACE << "packet was received correctly, it is now handed to upper layer...\n";
318  // go on with processing this AirFrame, send it to the Mac-Layer
319  if (notifyRxStart) {
320  phy->sendControlMsgToMac(new cMessage("RxStartStatus", MacToPhyInterface::PHY_RX_END_WITH_SUCCESS));
321  }
322  phy->sendUp(frame, result);
323  }
324  else {
325  if (frame->getUnderMinPowerLevel()) {
326  EV_TRACE << "packet was not detected by the card. power was under minPowerLevel threshold\n";
327  }
328  else if (whileSending) {
329  EV_TRACE << "packet was received while sending, sending it as control message to upper layer\n";
330  phy->sendControlMsgToMac(new cMessage("Error", RECWHILESEND));
331  }
332  else {
333  EV_TRACE << "packet was not received correctly, sending it as control message to upper layer\n";
334  if (notifyRxStart) {
335  phy->sendControlMsgToMac(new cMessage("RxStartStatus", MacToPhyInterface::PHY_RX_END_WITH_FAILURE));
336  }
337 
338  if (((DeciderResult80211*) result)->isCollision()) {
339  phy->sendControlMsgToMac(new cMessage("Error", Decider80211p::COLLISION));
340  }
341  else {
342  phy->sendControlMsgToMac(new cMessage("Error", BITERROR));
343  }
344  }
345  delete result;
346  }
347 
348  if (phy11p->getRadioState() == Radio::TX) {
349  EV_TRACE << "I'm currently sending\n";
350  }
351  // check if channel is idle now
352  // we declare channel busy if CCA tells us so, or if we are currently
353  // decoding a frame
354  else if (cca(simTime(), frame) == false || currentSignal.first != 0) {
355  EV_TRACE << "Channel not yet idle!\n";
356  }
357  else {
358  // might have been idle before (when the packet rxpower was below sens)
359  if (isChannelIdle != true) {
360  EV_TRACE << "Channel idle now!\n";
361  setChannelIdleStatus(true);
362  }
363  }
364  return notAgain;
365 }

396 void Decider80211p::switchToTx()
397 {
398  if (currentSignal.first != 0) {
399  // we are currently trying to receive a frame.
400  if (allowTxDuringRx) {
401  // if the above layer decides to transmit anyhow, we need to abort reception
402  AirFrame11p* currentFrame = dynamic_cast<AirFrame11p*>(currentSignal.first);
403  ASSERT(currentFrame);
404  // flag the frame as "while transmitting"
405  currentFrame->setWasTransmitting(true);
406  currentFrame->setBitError(true);
407  // forget about the signal
408  currentSignal.first = 0;
409  }
410  else {
411  throw cRuntimeError("Decider80211p: mac layer requested phy to transmit a frame while currently receiving another");
412  }
413  }
414 }
