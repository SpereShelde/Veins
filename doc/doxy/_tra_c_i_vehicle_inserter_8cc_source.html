104 void TraCIVehicleInserter::receiveSignal(cComponent* source, simsignal_t signalID, const SimTime& t, cObject* details)
105 {
106  if (signalID == TraCIScenarioManager::traciTimestepBeginSignal) {
107  ASSERT(manager->isConnected());
108  if (simTime() > 1) {
109  if (vehicleTypeIds.size() == 0) {
110  std::list<std::string> vehTypes = manager->getCommandInterface()->getVehicleTypeIds();
111  for (std::list<std::string>::const_iterator i = vehTypes.begin(); i != vehTypes.end(); ++i) {
112  if (i->substr(0, 8).compare("DEFAULT_") != 0) {
113  EV_DEBUG << *i << std::endl;
114  vehicleTypeIds.push_back(*i);
115  }
116  }
117  }
118  if (routeIds.size() == 0) {
119  std::list<std::string> routes = manager->getCommandInterface()->getRouteIds();
120  for (std::list<std::string>::const_iterator i = routes.begin(); i != routes.end(); ++i) {
121  std::string routeId = *i;
122  if (par("useRouteDistributions").boolValue() == true) {
123  if (std::count(routeId.begin(), routeId.end(), '#') >= 1) {
124  EV_DEBUG << "Omitting route " << routeId << " as it seems to be a member of a route distribution (found '#' in name)" << std::endl;
125  continue;
126  }
127  }
128  EV_DEBUG << "Adding " << routeId << " to list of possible routes" << std::endl;
129  routeIds.push_back(routeId);
130  }
131  }
132  for (int i = manager->getManagedHosts().size() + queuedVehicles.size(); i < numVehicles; i++) {
133  insertNewVehicle();
134  }
135  }
136 
137  insertVehicles();
138  }
139 }

141 void TraCIVehicleInserter::insertNewVehicle()
142 {
143  std::string type;
144  if (vehicleTypeIds.size()) {
145  int vehTypeId = mobRng->intRand(vehicleTypeIds.size());
146  type = vehicleTypeIds[vehTypeId];
147  }
148  else {
149  type = "DEFAULT_VEHTYPE";
150  }
151  int routeId = mobRng->intRand(routeIds.size());
152  vehicleInsertQueue[routeId].push(type);
153 }

155 void TraCIVehicleInserter::insertVehicles()
156 {
157 
158  for (std::map<int, std::queue<std::string>>::iterator i = vehicleInsertQueue.begin(); i != vehicleInsertQueue.end();) {
159  std::string route = routeIds[i->first];
160  EV_DEBUG << "process " << route << std::endl;
161  std::queue<std::string> vehicles = i->second;
162  while (!i->second.empty()) {
163  bool suc = false;
164  std::string type = i->second.front();
165  std::stringstream veh;
166  veh << type << "_" << vehicleNameCounter;
167  EV_DEBUG << "trying to add " << veh.str() << " with " << route << " vehicle type " << type << std::endl;
168 
169  suc = manager->getCommandInterface()->addVehicle(veh.str(), type, route, simTime());
170  if (!suc) {
171  i->second.pop();
172  }
173  else {
174  EV_DEBUG << "successful inserted " << veh.str() << std::endl;
175  queuedVehicles.insert(veh.str());
176  i->second.pop();
177  vehicleNameCounter++;
178  }
179  }
180  std::map<int, std::queue<std::string>>::iterator tmp = i;
181  ++tmp;
182  vehicleInsertQueue.erase(i);
183  i = tmp;
184  }
185 }
