32 void BreakpointPathlossModel::filterSignal(Signal* signal)
33 {
34  auto senderPos = signal->getSenderPoa().pos.getPositionAt();
35  auto receiverPos = signal->getReceiverPoa().pos.getPositionAt();
36 
37  /** Calculate the distance factor */
38  double distance = useTorus ? receiverPos.sqrTorusDist(senderPos, playgroundSize) : receiverPos.sqrdist(senderPos);
39  distance = sqrt(distance);
40  EV_TRACE << "distance is: " << distance << endl;
41 
42  if (distance <= 1.0) {
43  // attenuation is negligible
44  return;
45  }
46 
47  double attenuation = 1;
48  // PL(d) = PL0 + 10 alpha log10 (d/d0)
49  // 10 ^ { PL(d)/10 } = 10 ^{PL0 + 10 alpha log10 (d/d0)}/10
50  // 10 ^ { PL(d)/10 } = 10 ^ PL0/10 * 10 ^ { 10 log10 (d/d0)^alpha }/10
51  // 10 ^ { PL(d)/10 } = 10 ^ PL0/10 * 10 ^ { log10 (d/d0)^alpha }
52  // 10 ^ { PL(d)/10 } = 10 ^ PL0/10 * (d/d0)^alpha
53  if (distance < breakpointDistance) {
54  attenuation = attenuation * PL01_real;
55  attenuation = attenuation * pow(distance, alpha1);
56  }
57  else {
58  attenuation = attenuation * PL02_real;
59  attenuation = attenuation * pow(distance / breakpointDistance, alpha2);
60  }
61  attenuation = 1 / attenuation;
62  EV_TRACE << "attenuation is: " << attenuation << endl;
63 
64  pathlosses.record(10 * log10(attenuation)); // in dB
65 
66  *signal *= attenuation;
67 }
