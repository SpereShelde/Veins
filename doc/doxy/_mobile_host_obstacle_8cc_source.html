34 bool isPointInObstacle(Coord point, const MobileHostObstacle::Coords& shape)
35 {
36  bool isInside = false;
37  auto i = shape.begin();
38  auto j = (shape.rbegin() + 1).base();
39  for (; i != shape.end(); j = i++) {
40  bool inYRangeUp = (point.y >= i->y) && (point.y < j->y);
41  bool inYRangeDown = (point.y >= j->y) && (point.y < i->y);
42  bool inYRange = inYRangeUp || inYRangeDown;
43  if (!inYRange) continue;
44  bool intersects = point.x < (i->x + ((point.y - i->y) * (j->x - i->x) / (j->y - i->y)));
45  if (!intersects) continue;
46  isInside = !isInside;
47  }
48  return isInside;
49 }

51 double segmentsIntersectAt(Coord p1From, Coord p1To, Coord p2From, Coord p2To)
52 {
53  Coord p1Vec = p1To - p1From;
54  Coord p2Vec = p2To - p2From;
55  Coord p1p2 = p1From - p2From;
56 
57  double D = (p1Vec.x * p2Vec.y - p1Vec.y * p2Vec.x);
58 
59  double p1Frac = (p2Vec.x * p1p2.y - p2Vec.y * p1p2.x) / D;
60  if (p1Frac < 0 || p1Frac > 1) return -1;
61 
62  double p2Frac = (p1Vec.x * p1p2.y - p1Vec.y * p1p2.x) / D;
63  if (p2Frac < 0 || p2Frac > 1) return -1;
64 
65  return p1Frac;
66 }

70 MobileHostObstacle::Coords MobileHostObstacle::getShape(simtime_t t) const
71 {
72  double l = getLength();
73  double o = getHostPositionOffset(); // this is the shift we have to undo in order to (given the OMNeT++ host position) get the car's front bumper position
74  double w = getWidth() / 2;
75  const BaseMobility* m = getMobility();
76  Coord p = m->getPositionAt(t);
77  double a = Heading::fromCoord(m->getCurrentOrientation()).getRad();
78 
79  Coords shape;
80  shape.push_back(p + Coord(-(l - o), -w).rotatedYaw(-a));
81  shape.push_back(p + Coord(+o, -w).rotatedYaw(-a));
82  shape.push_back(p + Coord(+o, +w).rotatedYaw(-a));
83  shape.push_back(p + Coord(-(l - o), +w).rotatedYaw(-a));
84 
85  return shape;
86 }

88 bool MobileHostObstacle::maybeInBounds(double x1, double y1, double x2, double y2, simtime_t t) const
89 {
90  double l = getLength();
91  double o = getHostPositionOffset(); // this is the shift we have to undo in order to (given the OMNeT++ host position) get the car's front bumper position
92  double w = getWidth() / 2;
93  const BaseMobility* m = getMobility();
94  Coord p = m->getPositionAt(t);
95 
96  double lw = std::max(l, w);
97 
98  double xx1 = p.x - std::abs(o) - lw;
99  double xx2 = p.x + std::abs(o) + lw;
100  double yy1 = p.y - std::abs(o) - lw;
101  double yy2 = p.y + std::abs(o) + lw;
102 
103  if (xx2 < x1) return false;
104  if (xx1 > x2) return false;
105  if (yy2 < y1) return false;
106  if (yy1 > y2) return false;
107 
108  return true;
109 }

111 double MobileHostObstacle::getIntersectionPoint(const Coord& senderPos, const Coord& receiverPos, simtime_t t) const
112 {
113  const double not_a_number = std::numeric_limits<double>::quiet_NaN();
114 
115  MobileHostObstacle::Coords shape = getShape(t);
116 
117  // shortcut if sender is inside
118  bool senderInside = isPointInObstacle(senderPos, shape);
119  if (senderInside) return 0;
120 
121  // get a list of points (in [0, 1]) along the line between sender and receiver where the beam intersects with this obstacle
122  std::multiset<double> intersectAt;
123  bool doesIntersect = false;
124  MobileHostObstacle::Coords::const_iterator i = shape.begin();
125  MobileHostObstacle::Coords::const_iterator j = (shape.rbegin() + 1).base();
126  for (; i != shape.end(); j = i++) {
127  Coord c1 = *i;
128  Coord c2 = *j;
129 
130  double inter = segmentsIntersectAt(senderPos, receiverPos, c1, c2);
131  if (inter != -1) {
132  doesIntersect = true;
133  EV << "intersect: " << inter << endl;
134  intersectAt.insert(inter);
135  }
136  }
137 
138  // shortcut if no intersections
139  if (!doesIntersect) {
140  bool receiverInside = isPointInObstacle(receiverPos, shape);
141  if (receiverInside) return senderPos.distance(receiverPos);
142  return not_a_number;
143  }
144 
145  return (*intersectAt.begin() * senderPos.distance(receiverPos));
146 }
