65 void BasePhyLayer::initialize(int stage)
66 {
67 
68  ChannelAccess::initialize(stage);
69 
70  if (stage == 0) {
71  // if using sendDirect, make sure that messages arrive without delay
72  gate("radioIn")->setDeliverOnReceptionStart(true);
73 
74  upperLayerIn = findGate("upperLayerIn");
75  upperLayerOut = findGate("upperLayerOut");
76  upperControlOut = findGate("upperControlOut");
77  upperControlIn = findGate("upperControlIn");
78 
79  if (par("useNoiseFloor").boolValue()) {
80  noiseFloorValue = FWMath::dBm2mW(par("noiseFloor").doubleValue());
81  }
82  else {
83  noiseFloorValue = 0;
84  }
85  minPowerLevel = par("minPowerLevel").doubleValue();
86  minPowerLevel = FWMath::dBm2mW(minPowerLevel);
87 
88  recordStats = par("recordStats").boolValue();
89 
90  radio = initializeRadio();
91 
92  world = FindModule<BaseWorldUtility*>::findGlobalModule();
93  if (world == nullptr) {
94  throw cRuntimeError("Could not find BaseWorldUtility module");
95  }
96 
97  if (cc->hasPar("sat") && (minPowerLevel - FWMath::dBm2mW(cc->par("sat").doubleValue())) < -0.000001) {
98  throw cRuntimeError("minPowerLevel can't be smaller than the signal attenuation threshold (sat) in ConnectionManager. Please adjust your omnetpp.ini file accordingly.");
99  }
100 
101  initializeAnalogueModels(par("analogueModels").xmlValue());
102  initializeDecider(par("decider").xmlValue());
103  initializeAntenna(par("antenna").xmlValue());
104 
105  radioSwitchingOverTimer = new cMessage("radio switching over", RADIO_SWITCHING_OVER);
106  txOverTimer = new cMessage("transmission over", TX_OVER);
107  }
108 }

139 void BasePhyLayer::getParametersFromXML(cXMLElement* xmlData, ParameterMap& outputMap)
140 {
141  cXMLElementList parameters = xmlData->getElementsByTagName("Parameter");
142 
143  for (cXMLElementList::const_iterator it = parameters.begin(); it != parameters.end(); it++) {
144 
145  const char* name = (*it)->getAttribute("name");
146  const char* type = (*it)->getAttribute("type");
147  const char* value = (*it)->getAttribute("value");
148  if (name == nullptr || type == nullptr || value == nullptr) throw cRuntimeError("Invalid parameter, could not find name, type or value");
149 
150  std::string sType = type; // needed for easier comparision
151  std::string sValue = value; // needed for easier comparision
152 
153  cMsgPar param(name);
154 
155  // parse type of parameter and set value
156  if (sType == "bool") {
157  param.setBoolValue(sValue == "true" || sValue == "1");
158  }
159  else if (sType == "double") {
160  param.setDoubleValue(strtod(value, nullptr));
161  }
162  else if (sType == "string") {
163  param.setStringValue(value);
164  }
165  else if (sType == "long") {
166  param.setLongValue(strtol(value, nullptr, 0));
167  }
168  else {
169  throw cRuntimeError("Unknown parameter type: '%s'", sType.c_str());
170  }
171 
172  // add parameter to output map
173  outputMap[name] = param;
174  }

281 std::shared_ptr<Antenna> BasePhyLayer::initializeSampledAntenna1D(ParameterMap& params)
282 {
283  // get samples of the modeled antenna and put them in a vector
284  ParameterMap::iterator it = params.find("samples");
285  if (it == params.end()) {
286  throw cRuntimeError("BasePhyLayer::initializeSampledAntenna1D(): No samples specified for this antenna. \
287  Please adjust your xml file accordingly.");
288  }
289 
290  std::vector<double> values;
291  std::stringstream samplesStream(it->second.stringValue());
292  std::copy(std::istream_iterator<double>(samplesStream), std::istream_iterator<double>(), std::back_inserter(values));
293 
294  // get optional random offsets for the antenna's samples
295  std::string offsetType = "";
296  std::vector<double> offsetParams;
297  it = params.find("random-offsets");
298  if (it != params.end()) {
299  std::string buf;
300  std::stringstream offsetStream(it->second.stringValue());
301  offsetStream >> offsetType;
302  std::copy(std::istream_iterator<double>(offsetStream), std::istream_iterator<double>(), std::back_inserter(offsetParams));
303  }
304 
305  // get optional random rotation of the whole pattern
306  std::string rotationType = "";
307  std::vector<double> rotationParams;
308  it = params.find("random-rotation");
309  if (it != params.end()) {
310  std::string buf;
311  std::stringstream rotationStream(it->second.stringValue());
312  rotationStream >> rotationType;
313  std::copy(std::istream_iterator<double>(rotationStream), std::istream_iterator<double>(), std::back_inserter(rotationParams));
314  }
315 
316  return std::make_shared<SampledAntenna1D>(values, offsetType, offsetParams, rotationType, rotationParams, this->getRNG(0));
317 }

321 void BasePhyLayer::initializeAnalogueModels(cXMLElement* xmlConfig)
322 {
323 
324  if (xmlConfig == nullptr) {
325  throw cRuntimeError("No analogue models configuration file specified.");
326  }
327 
328  cXMLElementList analogueModelList = xmlConfig->getElementsByTagName("AnalogueModel");
329 
330  if (analogueModelList.empty()) {
331  throw cRuntimeError("No analogue models configuration found in configuration file.");
332  }
333 
334  // iterate over all AnalogueModel-entries, get a new AnalogueModel instance and add
335  // it to analogueModels
336  for (auto&& analogueModelData : analogueModelList) {
337  const char* name = analogueModelData->getAttribute("type");
338  const char* thresholdingFlag = analogueModelData->getAttribute("thresholding");
339 
340  if (name == nullptr) {
341  throw cRuntimeError("Could not read name of analogue model.");
342  }
343 
344  ParameterMap params;
345  getParametersFromXML(analogueModelData, params);
346 
347  auto newAnalogueModel = getAnalogueModelFromName(name, params);
348 
349  if (!newAnalogueModel) {
350  throw cRuntimeError("Could not find an analogue model with the name \"%s\".", name);
351  }
352 
353  // attach the new AnalogueModel to the AnalogueModelList
354  if (thresholdingFlag && std::string(thresholdingFlag) == "true") {
355  if (!newAnalogueModel->neverIncreasesPower()) {
356  throw cRuntimeError("Tried to instantiate analogue model \"%s\" with tresholding=true, but model does not support this.", name);
357  }
358  analogueModelsThresholding.push_back(std::move(newAnalogueModel));
359  }
360  else {
361  analogueModels.push_back(std::move(newAnalogueModel));
362  }
363 
364  EV_TRACE << "AnalogueModel \"" << name << "\" loaded." << endl;
365  }
366 }

370 void BasePhyLayer::handleMessage(cMessage* msg)
371 {
372 
373  // self messages
374  if (msg->isSelfMessage()) {
375  handleSelfMessage(msg);
376 
377  // MacPkts <- MacToPhyControlInfo
378  }
379  else if (msg->getArrivalGateId() == upperLayerIn) {
380  handleUpperMessage(msg);
381 
382  // controlmessages
383  }
384  else if (msg->getArrivalGateId() == upperControlIn) {
385  handleUpperControlMessage(msg);
386 
387  // AirFrames
388  }
389  else if (msg->getKind() == AIR_FRAME) {
390  handleAirFrame(static_cast<AirFrame*>(msg));
391 
392  // unknown message
393  }
394  else {
395  EV << "Unknown message received." << endl;
396  delete msg;
397  }

400 void BasePhyLayer::handleAirFrame(AirFrame* frame)
401 {
402  // TODO: ask jerome to set air frame priority in his UWBIRPhy
403  // ASSERT(frame->getSchedulingPriority() == airFramePriority());
404 
405  switch (static_cast<AirFrameState>(frame->getState())) {
406  case AirFrameState::start_receive:
407  handleAirFrameStartReceive(frame);
408  break;
409 
410  case AirFrameState::receiving: {
411  handleAirFrameReceiving(frame);
412  break;
413  }
414  case AirFrameState::end_receive:
415  handleAirFrameEndReceive(frame);
416  break;
417 
418  default:
419  throw cRuntimeError("Unknown AirFrame state: %s", frame->getState());
420  }

423 void BasePhyLayer::handleAirFrameStartReceive(AirFrame* frame)
424 {
425  EV_TRACE << "Received new AirFrame " << frame << " from channel." << endl;
426 
427  channelInfo.addAirFrame(frame, simTime());
428  ASSERT(!channelInfo.isChannelEmpty());
429 
430  if (usePropagationDelay) {
431  Signal& s = frame->getSignal();
432  simtime_t delay = simTime() - s.getSendingStart();
433  s.setPropagationDelay(delay);
434  }
435  ASSERT(frame->getSignal().getReceptionStart() == simTime());
436 
437  filterSignal(frame);
438 
439  if (decider && isKnownProtocolId(frame->getProtocolId())) {
440  frame->setState(static_cast<int>(AirFrameState::receiving));
441 
442  // pass the AirFrame the first time to the Decider
443  handleAirFrameReceiving(frame);
444 
445  // if no decider is defined we will schedule the message directly to its end
446  }
447  else {
448  Signal& signal = frame->getSignal();
449 
450  simtime_t signalEndTime = signal.getReceptionStart() + frame->getDuration();
451  frame->setState(static_cast<int>(AirFrameState::end_receive));
452 
453  sendSelfMessage(frame, signalEndTime);
454  }
455 }
456 
457 void BasePhyLayer::handleAirFrameReceiving(AirFrame* frame)
458 {
459 
460  Signal& signal = frame->getSignal();
461  simtime_t nextHandleTime = decider->processSignal(frame);
462 
463  ASSERT(signal.getDuration() == frame->getDuration());
464  simtime_t signalEndTime = signal.getReceptionStart() + frame->getDuration();
465 
466  // check if this is the end of the receiving process
467  if (simTime() >= signalEndTime) {
468  frame->setState(static_cast<int>(AirFrameState::end_receive));
469  handleAirFrameEndReceive(frame);
470  return;
471  }
472 
473  // smaller zero means don't give it to me again
474  if (nextHandleTime < 0) {
475  nextHandleTime = signalEndTime;
476  frame->setState(static_cast<int>(AirFrameState::end_receive));
477 
478  // invalid point in time
479  }
480  else if (nextHandleTime < simTime() || nextHandleTime > signalEndTime) {
481  throw cRuntimeError("Invalid next handle time returned by Decider. Expected a value between current simulation time (%.2f) and end of signal (%.2f) but got %.2f", SIMTIME_DBL(simTime()), SIMTIME_DBL(signalEndTime), SIMTIME_DBL(nextHandleTime));
482  }
483 
484  EV_TRACE << "Handed AirFrame with ID " << frame->getId() << " to Decider. Next handling in " << nextHandleTime - simTime() << "s." << endl;
485 
486  sendSelfMessage(frame, nextHandleTime);
487 }

489 void BasePhyLayer::handleAirFrameEndReceive(AirFrame* frame)
490 {
491  EV_TRACE << "End of Airframe with ID " << frame->getId() << "." << endl;
492 
493  simtime_t earliestInfoPoint = channelInfo.removeAirFrame(frame);

495  /* clean information in the radio until earliest time-point
496  * of information in the ChannelInfo,
497  * since this time-point might have changed due to removal of
498  * the AirFrame
499  */
500  if (channelInfo.isChannelEmpty()) {
501  earliestInfoPoint = simTime();
502  }

505 void BasePhyLayer::handleUpperMessage(cMessage* msg)
506 {
507 
508  // check if Radio is in TX state
509  if (radio->getCurrentState() != Radio::TX) {
510  delete msg;
511  msg = nullptr;
512  throw cRuntimeError("Error: message for sending received, but radio not in state TX");
513  }
514 
515  // check if not already sending
516  if (txOverTimer->isScheduled()) {
517  delete msg;
518  msg = nullptr;
519  throw cRuntimeError("Error: message for sending received, but radio already sending");
520  }
521 
522  // build the AirFrame to send
523  ASSERT(dynamic_cast<cPacket*>(msg) != nullptr);
524 
525  unique_ptr<AirFrame> frame = encapsMsg(static_cast<cPacket*>(msg));
526 
527  // Prepare a POA object and attach it to the created Airframe
528  AntennaPosition pos = antennaPosition;
529  Coord orient = antennaHeading.toCoord();
530  frame->setPoa({pos, orient, antenna});
531 
532  // make sure there is no self message of kind TX_OVER scheduled
533  // and schedule the actual one
534  ASSERT(!txOverTimer->isScheduled());
535  sendSelfMessage(txOverTimer, simTime() + frame->getDuration());
536 
537  sendMessageDown(frame.release());
538 }

545 unique_ptr<AirFrame> BasePhyLayer::encapsMsg(cPacket* macPkt)
546 {
547  // the cMessage passed must be a MacPacket... but no cast needed here
548  // MacPkt* pkt = static_cast<MacPkt*>(msg);
549 
550  // ...and must always have a ControlInfo attached (contains Signal)
551  auto ctrlInfo = unique_ptr<cObject>(macPkt->removeControlInfo());
552  ASSERT(ctrlInfo);
553 
554  // create the new AirFrame
555  auto frame = createAirFrame(macPkt);
556 
557  // set the members
558  // set priority of AirFrames above the normal priority to ensure
559  // channel consistency (before any thing else happens at a time
560  // point t make sure that the channel has removed every AirFrame
561  // ended at t and added every AirFrame started at t)
562  frame->setSchedulingPriority(airFramePriority());
563  frame->setProtocolId(myProtocolId());
564  frame->setId(world->getUniqueAirFrameId());
565  frame->setChannel(radio->getCurrentChannel());
566  frame->setBitLength(0); // as the PHY might mix multiple different encoding rates (and things that might not be bits at all)
567 
568  frame->encapsulate(macPkt);
569 
570  // attach the spectrum-dependent Signal to the airFrame
571  attachSignal(frame.get(), ctrlInfo.get());
572 
573  // --- from here on, the AirFrame is the owner of the MacPacket ---
574  macPkt = nullptr;
575  EV_TRACE << "AirFrame encapsulated, length: " << frame->getBitLength() << "\n";
576 
577  return frame;
578 }

587 {

638 void BasePhyLayer::filterSignal(AirFrame* frame)
639 {
640  ASSERT(dynamic_cast<ChannelAccess* const>(frame->getArrivalModule()) == this);
641  ASSERT(dynamic_cast<ChannelAccess* const>(frame->getSenderModule()));
642  Signal& signal = frame->getSignal();
643 
644  // Extract position and orientation of sender and receiver (this module) first
645  const AntennaPosition receiverPosition = antennaPosition;
646  const Coord receiverOrientation = antennaHeading.toCoord();
647  // get POA from frame with the sender's position, orientation and antenna
648  POA& senderPOA = frame->getPoa();
649  const AntennaPosition senderPosition = senderPOA.pos;
650  const Coord senderOrientation = senderPOA.orientation;
651 
652  // add position information to signal
653  signal.setSenderPoa(senderPOA);
654  signal.setReceiverPoa({receiverPosition, receiverOrientation, antenna});
655 
656  // compute gains at sender and receiver antenna
657  double receiverGain = antenna->getGain(receiverPosition.getPositionAt(), receiverOrientation, senderPosition.getPositionAt());
658  double senderGain = senderPOA.antenna->getGain(senderPosition.getPositionAt(), senderOrientation, receiverPosition.getPositionAt());
659 
660  // add the resulting total gain to the attenuations list
661  EV_TRACE << "Sender's antenna gain: " << senderGain << endl;
662  EV_TRACE << "Own (receiver's) antenna gain: " << receiverGain << endl;
663  signal *= receiverGain * senderGain;
664 
665  // go on with AnalogueModels
666  // attach analogue models suitable for thresholding to signal (for later evaluation)
667  signal.setAnalogueModelList(&analogueModelsThresholding);
668 
669  // apply all analouge models that are *not* suitable for thresholding now
670  for (auto& analogueModel : analogueModels) {

715 simtime_t BasePhyLayer::setRadioState(int rs)
716 {
717  Enter_Method_Silent();
718  ASSERT(radio);
719 
720  if (txOverTimer && txOverTimer->isScheduled()) {
721  EV_WARN << "Switched radio while sending an AirFrame. The effects this would have on the transmission are not simulated by the BasePhyLayer!";
722  }
723 
724  simtime_t switchTime = radio->switchTo(rs, simTime());
725 
726  // invalid switch time, we are probably already switching
727  if (switchTime < 0) return switchTime;
728 
729  // if switching is done in exactly zero-time no extra self-message is scheduled
730  if (switchTime == 0.0) {
731  // TODO: in case of zero-time-switch, send no control-message to mac!
732  // maybe call a method finishRadioSwitchingSilent()
733  finishRadioSwitching();
734  }
735  else {
736  sendSelfMessage(radioSwitchingOverTimer, simTime() + switchTime);
737  }
738 
739  return switchTime;
740 }
