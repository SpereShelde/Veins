46  template <typename T>
47  T read()
48  {
49  T buf_to_return;
50  unsigned char* p_buf_to_return = reinterpret_cast<unsigned char*>(&buf_to_return);
51 
52  if (isBigEndian()) {
53  for (size_t i = 0; i < sizeof(buf_to_return); ++i) {
54  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
55  p_buf_to_return[i] = buf[buf_index++];
56  }
57  }
58  else {
59  for (size_t i = 0; i < sizeof(buf_to_return); ++i) {
60  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
61  p_buf_to_return[sizeof(buf_to_return) - 1 - i] = buf[buf_index++];
62  }
63  }
64 
65  return buf_to_return;
66  }

68  template <typename T>
69  void write(T inv)
70  {
71  unsigned char* p_buf_to_send = reinterpret_cast<unsigned char*>(&inv);
72 
73  if (isBigEndian()) {
74  for (size_t i = 0; i < sizeof(inv); ++i) {
75  buf += p_buf_to_send[i];
76  }
77  }
78  else {
79  for (size_t i = 0; i < sizeof(inv); ++i) {
80  buf += p_buf_to_send[sizeof(inv) - 1 - i];
81  }
82  }
83  }

85  void readBuffer(unsigned char* buffer, size_t size)
86  {
87  if (isBigEndian()) {
88  for (size_t i = 0; i < size; ++i) {
89  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
90  buffer[i] = buf[buf_index++];

92  }
93  else {
94  for (size_t i = 0; i < size; ++i) {
95  if (eof()) throw cRuntimeError("Attempted to read past end of byte buffer");
96  buffer[size - 1 - i] = buf[buf_index++];
97  }
98  }
