44 /**
45  * First we have to initialize the module from which we derived ours,
46  * in this case BaseLayer.
47  *
48  **/
49 void BaseMacLayer::initialize(int stage)
50 {
51  BaseLayer::initialize(stage);
52 
53  if (stage == 0) {
54  // get handle to phy layer
55  if ((phy = FindModule<MacToPhyInterface*>::findSubModule(getParentModule())) == nullptr) {
56  throw cRuntimeError("Could not find a PHY module.");
57  }
58 
59  headerLength = par("headerLength");
60  }
61  if (myMacAddr == LAddress::L2NULL()) {
62  // see if there is an addressing module available
63  // otherwise use NIC modules id as MAC address
64  AddressingInterface* addrScheme = FindModule<AddressingInterface*>::findSubModule(findHost());
65  if (addrScheme) {
66  myMacAddr = addrScheme->myMacAddr(this);
67  }
68  else {
69  const std::string addressString = par("address").stringValue();
70  if (addressString.empty() || addressString == "auto")
71  myMacAddr = LAddress::L2Type(getParentModule()->getId());
72  else
73  myMacAddr = 0;
74  // use streaming operator for string conversion, this makes it more
75  // independent from the myMacAddr type
76  std::ostringstream oSS;
77  oSS << myMacAddr;
78  par("address").setStringValue(oSS.str());
79  }
80  registerInterface();
81  }
82 }

101 /**
102  * Encapsulates the received NetwPkt into a MacPkt and set all needed
103  * header fields.
104  **/
105 MacPkt* BaseMacLayer::encapsMsg(cPacket* netwPkt)
106 {
107  MacPkt* pkt = new MacPkt(netwPkt->getName(), netwPkt->getKind());
108  pkt->setBitLength(headerLength);
109 
110  // copy dest address from the Control Info attached to the network
111  // message by the network layer
112  cObject* cInfo = netwPkt->removeControlInfo();
113 
114  EV_TRACE << "CInfo removed, mac addr=" << getUpperDestinationFromControlInfo(cInfo) << endl;
115  pkt->setDestAddr(getUpperDestinationFromControlInfo(cInfo));
116 
117  // delete the control info
118  delete cInfo;
119 
120  // set the src address to own mac address (nic module getId())
121  pkt->setSrcAddr(myMacAddr);
122 
123  // encapsulate the network packet
124  pkt->encapsulate(netwPkt);
125  EV_TRACE << "pkt encapsulated\n";
126 
127  return pkt;
128 }

130 /**
131  * Redefine this function if you want to process messages from upper

139  ASSERT(dynamic_cast<cPacket*>(mac));
140  sendDown(encapsMsg(static_cast<cPacket*>(mac)));

143 /**
144  * This basic implementation just forwards all message that are
145  * broadcast (destAddr = L2BROADCAST) or destined for this node
146  * (destAddr = nic module getId()) to the network layer
147  *
148  * @sa sendUp
149  **/
150 
151 void BaseMacLayer::handleLowerMsg(cMessage* msg)
152 {
153  MacPkt* mac = static_cast<MacPkt*>(msg);
154  LAddress::L2Type dest = mac->getDestAddr();
155  LAddress::L2Type src = mac->getSrcAddr();
156 
157  // only foward to upper layer if message is for me or broadcast
158  if ((dest == myMacAddr) || LAddress::isL2Broadcast(dest)) {
159  EV_TRACE << "message with mac addr " << src << " for me (dest=" << dest << ") -> forward packet to upper layer\n";
160  sendUp(decapsMsg(mac));
161  }
162  else {
163  EV_TRACE << "message with mac addr " << src << " not for me (dest=" << dest << ") -> delete (my MAC=" << myMacAddr << ")\n";
164  delete mac;
165  }
166 }

168 void BaseMacLayer::handleLowerControl(cMessage* msg)
169 {
170  switch (msg->getKind()) {
171  case MacToPhyInterface::TX_OVER:
172  msg->setKind(TX_OVER);
173  sendControlUp(msg);
174  break;
175  default:
176  EV << "BaseMacLayer does not handle control messages of this type (name was " << msg->getName() << ")\n";
177  delete msg;
178  break;
179  }
180 }
