44 void TraCITrafficLightAbstractLogic::handleMessage(cMessage* msg)
45 {
46  if (msg->isSelfMessage()) {
47  handleSelfMsg(msg);
48  }
49  else if (msg->arrivedOn("interface$i")) {
50  TraCITrafficLightMessage* tlMsg = check_and_cast<TraCITrafficLightMessage*>(msg);
51  // always check for changed switch time and (re-)schedule switch handler if so
52  if (tlMsg->getChangedAttribute() == TrafficLightAtrributeType::SWITCHTIME) {
53  // schedule handler right before the switch
54  cancelEvent(switchTimer);
55  // make sure the message is not scheduled to the past
56  simtime_t nextTick = std::max(SimTime(std::stoi(tlMsg->getNewValue()), SIMTIME_MS), simTime());
57  scheduleAt(nextTick, switchTimer);
58  }
59  // defer further handling to subclass implementation
60  handleTlIfMsg(tlMsg);
61  }
62  else if (msg->arrivedOn("applLayer$i")) {
63  handleApplMsg(msg);
64  }
65  else {
66  throw cRuntimeError("Unknown message arrived on %s", msg->getArrivalGate()->getName());
67  }
68 }
