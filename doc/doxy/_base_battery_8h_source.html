34 /**
35  * @brief Defines the amount of power drawn by a device from
36  * a power source.
37  *
38  * Used as generic amount parameter for BaseBatteries "draw"-method.
39  *
40  * Can be either an instantaneous draw of a certain energy amount
41  * in mWs (type=ENERGY) or a draw of a certain current in mA over
42  * time (type=CURRENT).
43  *
44  * Can be sub-classed for more complex power draws.
45  *
46  * @ingroup baseModules
47  * @ingroup power
48  */
49 class VEINS_API DrawAmount {
50 public:
51  /** @brief The type of the amount to draw.*/
52  enum PowerType {
53  CURRENT, /** @brief Current in mA over time. */
54  ENERGY /** @brief Single fixed energy draw in mWs */
55  };
56 
57 protected:
58  /** @brief Stores the type of the amount.*/
59  int type;
60 
61  /** @brief Stores the actual amount.*/
62  double value;
63 
64 public:
65  /** @brief Initializes with passed type and value.*/
66  DrawAmount(int type = CURRENT, double value = 0)
67  : type(type)
68  , value(value)
69  {
70  }
71  virtual ~DrawAmount()
72  {
73  }
74 
75  /** @brief Returns the type of power drawn as PowerType. */
76  virtual int getType() const
77  {
78  return type;
79  }
80  /** @brief Returns the actual amount of power drawn. */
81  virtual double getValue() const
82  {
83  return value;
84  }
85 
86  /** @brief Sets the type of power drawn. */
87  virtual void setType(int t)
88  {
89  type = t;
90  }
91  /** @brief Sets the actual amount of power drawn. */
92  virtual void setValue(double v)
93  {
94  value = v;
95  }
96 };

98 /**
99  * @brief Base class for any power source.
100  *
101  * See "SimpleBattery" for an example implementation.
102  *
103  * @ingroup baseModules
104  * @ingroup power
105  * @see SimpleBattery
106  */
107 class VEINS_API BaseBattery : public BaseModule {
108 public:
109  BaseBattery()
110  : BaseModule()
111  {
112  }
113  BaseBattery(unsigned stacksize)
114  : BaseModule(stacksize)
115  {
116  }
117  /**
118  * @brief Registers a power draining device with this battery.
119  *
120  * Takes the name of the device as well as a number of accounts
121  * the devices draws power for (like rx, tx, idle for a radio device).
122  *
123  * Returns an ID by which the device can identify itself to the
124  * battery.
125  *
126  * Has to be implemented by actual battery implementations.
127  */
128  virtual int registerDevice(const std::string& name, int numAccounts) = 0;
129 
130  /**
131  * @brief Draws power from the battery.
132  *
133  * The actual amount and type of power drawn is defined by the passed
134  * DrawAmount parameter. Can be an fixed single amount or an amount
135  * drawn over time.
136  * The drainID identifies the device which drains the power.
137  * "Account" identifies the account the power is drawn from. It is
138  * used for statistical evaluation only to see which activity of a
139  * device has used how much power. It does not affect functionality.
140  */
141  virtual void draw(int drainID, DrawAmount& amount, int account) = 0;
142 
143  /**
144  * @name State-of-charge interface
145  *
146  * @brief Other host modules should use these interfaces to obtain
147  * the state-of-charge of the battery. Do NOT use BatteryState
148  * interfaces, which should be used only by Battery Stats modules.
149  */
150  /*@{*/
151  /** @brief get voltage (future support for non-voltage regulated h/w */
152  virtual double getVoltage() const = 0;
153  /** @brief current state of charge of the battery, relative to its
154  * rated nominal capacity [0..1]
155  */
156  virtual double estimateResidualRelative() const = 0;
157  /** @brief current state of charge of the battery (mW-s) */
158  virtual double estimateResidualAbs() const = 0;
159  /** @brief Current state of the battery. */
160  virtual HostState::States getState() const = 0;
161  /*@}*/
162 };
